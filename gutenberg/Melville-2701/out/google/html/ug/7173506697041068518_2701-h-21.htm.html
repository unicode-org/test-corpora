بىچارە Queequeg! كېمە يېرىمى يېرىلىپ كەتكەندە ، سىز چوقۇم ئۆڭكۈرنىڭ ئۈستىدە ئېگىلىپ ، ئۇنىڭغا قاراپ تۇرغان بولۇشىڭىز كېرەك ئىدى. بۇ يەردە ، ئۇنىڭ يۇڭ تارتمىلىرىغا سېلىنىپ ، چەكمە ۋەھشىي قۇدۇقنىڭ ئاستىدىكى يېشىل داغلىق كەسلەنچۈككە ئوخشاش ئاشۇ نەملىك ۋە پاتقاق ئىچىدە ئۆمىلەپ يۈردى. قۇدۇق ياكى مۇز ئۆي ، قانداقتۇر ئۇنىڭغا نامرات بۇتپەرەسلىكنى ئىسپاتلىدى. غەلىتە يېرى شۇكى ، ئۇنىڭ تەرلىرىنىڭ پۈتۈن ئىسسىقلىقى ئۈچۈن ، ئۇ قىزىتمىغا پېتىپ قالغان قورقۇنچلۇق سوغۇقنى تۇتتى. ئاخىرى ، بىر نەچچە كۈنلۈك ئازاب-ئوقۇبەتلەردىن كېيىن ، ئۇنى ئۆلۈم دەرۋازىسىنىڭ تۈۋىگە يېقىن جايدا قومۇلغا قويدى. ئۇ ئۇزۇنغا سوزۇلغان ئاشۇ بىر نەچچە كۈندە قانداق ئىسراپ بولۇپ ئىسراپ بولۇپ كەتتى ، تاكى ئۇنىڭ رامكىسى ۋە چەكمە چېكىشتىن باشقا نەرسە قالمىدى. ئەمما ئۇنىڭدىكى باشقا نەرسىلەرنىڭ ھەممىسى شالاڭلىشىپ ، مەڭزى سۆڭىكى تېخىمۇ ئۆتكۈرلەشكەنلىكتىن ، ئۇنىڭ كۆزلىرى قارىماققا بارغانسىرى تولۇپ تاشقاندەك قىلاتتى. ئۇلار پارقىراق بىر غەلىتە يۇمشاقلىققا ئايلاندى. ھەمدە ئۇنىڭ كېسەللىكىدىن مۇلايىم ، ئەمما چوڭقۇر سىزگە نەزەر سالدى ، ئۇنىڭدىكى ئۆلمەس ياكى ئاجىزلاپ كېتەلمەيدىغان بۇ ئۆلمەس ساغلاملىقنىڭ ئاجايىپ گۇۋاھلىقى. سۇ ئۈستىدىكى چەمبىرەكلەرگە ئوخشاش ، ئۇلار بارغانسىرى سۇسلاشقاندا كېڭىيىدۇ. شۇڭا ئۇنىڭ كۆزلىرى ئەبەدىي ئۈزۈككە ئوخشاش يۇمىلاق ۋە يۇمىلاق كۆرۈندى. ئىسمىنى ئاشكارىلاشنى خالىمايدىغان بىر قورقۇنچ بۇ سۇسلاپ كەتكەن ۋەھشىينىڭ يېنىدا ئولتۇرۇپ ، ئۇنىڭ چىرايىدا غەلىتە نەرسىلەرنى كۆردىڭىز ، خۇددى زوروئاستېر ئۆلگەندە ئەتراپتىكى كىشىلەردەك. چۈنكى ئىنساندا ھەقىقەتەن ھەيران قالارلىق ۋە قورقۇنچلۇق نەرسە ئەزەلدىن سۆز ياكى كىتابقا قويۇلمىغان. ئۆلۈمنىڭ يېقىنلىشىشى ، ھەممە قەۋەتنى كۆرۈش ، ئەڭ ئاخىرقى ۋەھىي بىلەن ھەممىنى تەسىرلەندۈرىدۇ ، بۇنى پەقەت ئۆلۈكلەرلا ئېيتالايدۇ. دېمەك ، ئۇنى يەنە بىر قېتىم ئېيتايلى - ئۆلۈۋاتقان چالدى ياكى گرېتسىيەلىكلەرنىڭكىدىنمۇ يۇقىرى ۋە تېخىمۇ مۇقەددەس خىياللار يوق ئىدى ، ئۇلارنىڭ سىرلىق سايىسى بىچارە كۋېكېگنىڭ يۈزىدە سىيرىلىپ چۈشكەنلىكىنى كۆردى ، ئۇ جىمجىت لەيلەپ تۇرغان قومۇلدا جىمجىت ياتقاندەك قىلاتتى ، دومىلاپ تۇرغان دېڭىز ئۇنى ئاستا-ئاستا ئاخىرقى ئارامگاھىغا تىترىگەندەك قىلاتتى ، ئوكياننىڭ كۆرۈنمەيدىغان كەلكۈن دولقۇنى ئۇنى مەنزىلدىكى جەننەتكە كۆتۈردى.

خىزمەت ئەترىتىدىكىلەر ئەمەس ، بەلكى ئۇنىڭدىن ۋاز كەچتى كۋېكېگنىڭ ئۆزىگە كەلسەك ، ئۇنىڭ بۇ دېلوغا بولغان كۆز قارىشىنى ئۇ سورىغان قىزىقىش بىلەن زورلۇق بىلەن كۆرسەتتى. ئۇ كۈلرەڭ ئەتىگەنلىك سائەتتە ئۇنىڭغا بىرنى چاقىردى ، ئۇ كۈن ئەمدىلا يېتىپ كېلىپ ، ئۇنىڭ قولىنى تۇتۇپ ، نەنتۇكېتتىكى چېغىدا ئۆزىنىڭ ئانا ئارىلىدىكى مول ئۇرۇش ياغىچىغا ئوخشاش قېنىق ياغاچنىڭ بەزى كىچىك قولۋاقلىرىنى كۆرۈش پۇرسىتىگە ئېرىشكەنلىكىنى ئېيتتى. ھەمدە سۈرۈشتۈرگەندىن كېيىن ، ئۇ نەنتۇكېتتا قازا قىلغان بارلىق كىتلارنىڭ ئاشۇ قاراڭغۇ قولۋاقتا ياتقانلىقىنى ، بۇنداق ياتقۇزۇش خىيالىنىڭ ئۇنى بەك خۇشال قىلغانلىقىنى بىلدى. چۈنكى ئۇ ئۆز ئىرقىنىڭ ئادىتىگە ئوخشىمايتتى ، ئۇ ئۆلۈك جەڭچىنى جەسەتكە سالغاندىن كېيىن ، ئۇنى قولۋاققا سوزۇپ ، ئۇنى يۇلتۇزلۇق تاقىم ئاراللارغا لەيلەپ قويۇپ قويدى. چۈنكى ئۇلار يۇلتۇزلارنىڭ ئارال ئىكەنلىكىگە ئىشىنىپلا قالماستىن ، بەلكى بارلىق كۆرۈنگەن ئۇپۇقلاردىن ھالقىپ ، ئۆزىنىڭ مۇلايىم ، ماسلاشمىغان دېڭىزلىرى كۆك ئاسمانغا تۇتىشىدۇ. شۇنىڭ بىلەن سۈت يولىنىڭ ئاق پارچىلىرىنى شەكىللەندۈرىدۇ. ئۇ يەنە ئۆزىنىڭ ئادەتتىكى دېڭىز ئادىتى بويىچە قومۇلغا دەپنە قىلىنىشنى ئويلاپ تىترەپ كەتكەنلىكىنى ، ئۆلۈمنى يەيدىغان لەھەڭگە سەت ئىشتەك تاشلانغانلىقىنى ئېيتتى. ياق: ئۇ نەنتۇكېتقا ئوخشاش قولۋاقنى ئارزۇ قىلدى ، ئۇنىڭغا تېخىمۇ تۇغما ، كىتچى بولۇش سۈپىتى بىلەن ، بۇ جەسەت ساندۇقىنىڭ كىت كېمىسىگە ئوخشاش ئاچقۇچ يوق ئىدى. گەرچە بۇ چېتىشلىق ، ئەمما ئېنىق بولمىغان رول ۋە نۇرغۇن غۇۋا يوللار سۇس دەۋرنى زىننەتلەيدۇ.

ھازىر ، بۇ غەلىتە ئەھۋال ئاشكارىلانغاندا ، ياغاچچى ئۆز ئىچىگە ئالغان ھەر قانداق ئىشنى بىراقلا كۋېكېگنىڭ بۇيرۇقىنى قىلىشقا بۇيرۇدى. بۇ يەردە بىر قىسىم يات ، جەسەت ساندۇقى رەڭلىك كونا ياغاچلار بار بولۇپ ، ئۇلار ئۇزۇن سەپەردە لاكاداي ئارىلىنىڭ يەرلىك قەبىلىسىدىن كېسىۋېتىلگەن بولۇپ ، بۇ قاراڭغۇ تاختايلاردىن جەسەت ساندۇقى ياساش تەۋسىيە قىلىنغان. ياغاچچى بۇ بۇيرۇقنى بالدۇرراق ئىگىلىدى ، ئۇنىڭ ھۆكۈمرانلىقىنى ئالغاندىن كېيىن ، ئۇ دەرھال خاراكتېرىنىڭ پەرۋاسىز تېزلىكى بىلەن ، ئالدىن پەرەزگە قەدەم قويدى ۋە كۇئېگېگنىڭ تەدبىرلىرىنى ناھايىتى توغرىلىق بىلەن قوللاندى ، ئۇ قائىدىنى يۆتكىگەندە دائىم كۋېكېگنىڭ ئادىمىنى قوغلىدى.

ئۇزۇن ئارال ماتروسى: «ئاھ! بىچارە تورداش! ئۇ ھازىر ئۆلۈپ كېتىشى كېرەك» دېدى.

ئۇنىڭ مۇئاۋىن ئورۇندۇقىغا بارغاندا ، ياغاچچى قۇلايلىق ۋە ئادەتتىكى پايدىلىنىش ئۈچۈن ، ھازىر ئۇنىڭغا يۆتكىلىپ جەسەت ساندۇقىنىڭ قانچىلىك ئۇزۇنلۇقتا ئىكەنلىكىنى ئۆلچەپ چىقتى ، ئاندىن ئۇنىڭ ئۇچىدىكى ئىككى نوقتىنى كېسىپ يۆتكىلىشنى مەڭگۈلۈك قىلدى. ئۇ شۇنداق قىلىپ ، تاختاي ۋە ئۇنىڭ سايمانلىرىنى مارشاپ ، ئىشلەشكە باشلىدى.

ئەڭ ئاخىرقى مىخ ھەيدىلىپ ، قاپاق مۇۋاپىق پىلانلانغان ۋە ماسلاشتۇرۇلغاندىن كېيىن ، جەسەت ساندۇقىنى يېنىك مۈرىسىگە ئېلىپ ئۇنىڭ بىلەن بىللە مېڭىپ ، ئۇلارنىڭ بۇ تەرەپكە تېخى تەييار ياكى ئەمەسلىكىنى سۈرۈشتۈردى.

پالۇبىدىكى كىشىلەر جەسەت ساندۇقىنى ھەيدەشكە باشلىغان ئاچچىق ، ئەمما يېرىم يۇمۇرلۇق پەريادنى ئاڭلىغان كۋېكېگنى ھەممەيلەننى ھەيران قالدۇردى ، ئۇ ئىشنى دەرھال ئۇنىڭغا ئەكېلىشكە بۇيرۇدى ، شۇنداقلا ئۇنى ئىنكار قىلىدىغانلارمۇ يوق بارلىق ئىنسانلار ئىچىدە ، ئۆلۈپ كېتىۋاتقان بەزى ئەرلەرنىڭ ئەڭ زالىم ئىكەنلىكىنى كۆرۈپ شۈبھىسىزكى ، ئۇلار ئۇزۇن ئۆتمەي بىزگە ئازغىنە ئاۋارىچىلىك ئېلىپ كېلىدىغان بولغاچقا ، بىچارە تورداشلار بېرىلىشى كېرەك.

قومۇلغا يۆلىنىپ تۇرغان كۋېكېگ ئۇزۇن ۋاقىت دىققەت بىلەن جەسەت ساندۇقىنى كۆزەتكەن. ئاندىن ئۇ ھارپىسىنى چاقىردى ، ئۇنىڭدىن ياغاچ پايپاقنى تارتىپ چىقاردى ، ئاندىن تۆمۈر قىسمىنى كېمىنىڭ بىر پاچىقى بىلەن بىللە جەسەت ساندۇقىغا قويدى. ئۇنىڭ تەلىپىگە ئاساسەن ، پېچىنە-پىرەنىكلەر ئاندىن ئىچىگە يان تەرەپكە توغرىلاندى: بېشىغا بىر تەخسە تاتلىق سۇ قويۇلدى ، بىر كىچىك خالتا ياغاچ توپا پۇتتىكى تۇتۇپ قېلىندى. ھەمدە بىر پارچە يەلكەنلىك رەخت ياستۇققا ئۆرۈلۈپ كېلىندى ، كۋېكېگ ھازىر ئەڭ ئاخىرقى كارىۋىتىغا كۆتۈرۈلۈپ كېتىشنى ئۆتۈندى ، ئەگەر ئۇ بار بولسا ، ئۇنىڭ راھەتلىكىنى سىناپ بېقىشى كېرەك. ئۇ بىر نەچچە مىنۇت مىدىرلىمايلا يېتىپ قالدى ، ئاندىن بىرىگە سومكىسىغا بېرىپ كىچىك ئىلاھى يوجونى ئېلىپ كېتىشنى ئېيتتى. ئاندىن يوجو بىلەن قوللىرىنى كۆكرىكىگە توغرىلاپ ، جەسەت ساندۇقىنىڭ قېپىنى (ئۇ دەپ ئاتىغان تۇخۇم) ئۇنىڭ ئۈستىگە قويۇشقا چاقىردى. باش قىسمى خۇرۇم ھالقا بىلەن بۇرۇلۇپ كەتتى ، ئۇنىڭ جەسەت ساندۇقىدا كۋېكېگ ياتاتتى ، ئەمما چىرايىنى كۆزدە تۇتۇپ. «رارماي» (قىلىدۇ ؛ ئاسان بولىدۇ) ، ئۇ ئاخىرى زارلىنىپ ، قومۇلغا ئالماشتۇرۇشقا ئىمزا قويدى.

ئەمما بۇ ئىش يۈز بېرىشتىن ئىلگىرى ، بۇ ئىشلارنىڭ ھەممىسىدە ئاستا-ئاستا ئايلىنىپ يۈرگەن پىپ ئۇنىڭ ياتقان يېرىگە يېقىنلاشتى ۋە يۇمشاق يىغلاپ تۇرۇپ ئۇنى قولىغا ئالدى. يەنە بىرىدە ، تەمبۇرنى تۇتۇپ.

«بىچارە ماشىنا ئادەم! سىز ئەزەلدىن بۇ ھارغىنلىق بىلەن ھەرىكەت قىلىپ باقمىغانمۇ؟ ھازىر نەگە بارىسىز؟ ئۇ. Rig-a-dig, dig, dig!

- مەن ئاڭلىدىم ، - دېدى ستارباك ، - دەپ قاتتىق غەزەپلىنىپ ، - زوراۋان قىزىتمىلاردا ، ئەرلەر ، بارلىق نادانلار قەدىمكى تىللاردا سۆزلەشتى ، ۋە بۇ سىر تەكشۈرۈلگەندە ، ئۇلارنىڭ قەدىمكى ئۇنتۇلغان بالىلىق دەۋرىدە بەزى قەدىمكى ئالىملارنىڭ ئاڭلىشىدا سۆزلەنگەنلىكى ، شۇڭا مېنىڭ سۆيۈملۈك ئېتىقادىم ، بىچارە پىپ ، بىزنىڭ بۇ غەلىتە تاتلىقلىقىمىز. ئۇ ، ئەمما ئۇ يەردە؟ - قاراڭ ، ئۇ يەنە سۆزلەيدۇ ، ئەمما ھازىر تېخىمۇ ۋەھشىي ».

«ئىككى ۋە ئىككىنى شەكىللەندۈرۈڭ! بىز ئۇنىڭ گېنىرالى قىلايلى! خو ، ئۇنىڭ ھارپىسى نەدە؟ بۇ يەردە ياتقۇزۇڭ. ھەممە ئادەم! - ئەگەر Pip! كىت كېمىسى!

بۇلارنىڭ ھەممىسىدە ، كۋېك چۈشىدىكىدەك كۆزلىرىنى يۇمۇپ ياتاتتى. تۇرۇبا ئېلىپ كېلىندى ، كېسەل ئادەم ئۇنىڭ قومۇلغا ئالماشتۇرۇلدى.

ئەمما ھازىر ئۇ ئېنىقكى ئۆلۈمگە تەييارلىق قىلدى. ھازىر ئۇنىڭ جەسەت ساندۇقىنىڭ ماس كېلىدىغانلىقى ئىسپاتلانغاندىن كېيىن ، كۋېك تۇيۇقسىز يىغىلىپ قالدى. ئۇزۇن ئۆتمەي ياغاچچى ساندۇقىنىڭ ھاجىتى يوقتەك قىلاتتى: شۇنىڭ بىلەن ، بەزىلەر خۇشاللىقىدىن ھەيران قالغانلىقىنى ئىپادىلىگەندە ، ئۇ ماھىيەتتە ئۇنىڭ تۇيۇقسىز قېتىپ قېلىشىنىڭ سەۋەبىنىڭ مۇنداق ئىكەنلىكىنى ئېيتتى ؛ - ھالقىلىق پەيتتە ، ئۇ ئەمدىلا ئەمەلدىن قالدۇرماقچى بولغان بىر ئاز قىرغاقنى ئەسلەپ ئۆتتى. شۇڭلاشقا ئۇ ئۆلۈش توغرىسىدىكى كۆز قارىشىنى ئۆزگەرتكەن: ئۇ تېخى ئۆلەلمەيتتى. ئۇلار ئۇنىڭدىن ، ياشاش ياكى ئۆلۈشنىڭ ئۆزىنىڭ ئىگىلىك ھوقۇقى ۋە خۇشاللىقى مەسىلىسى ئىكەنلىكىنى سورىدى. ئۇ ئەلۋەتتە جاۋاب بەردى. بىر ئېغىز سۆز بىلەن ئېيتقاندا ، ئەگەر بىر ئادەم ياشاشنى قارار قىلسا ، نوقۇل كېسەللىك ئۇنى ئۆلتۈرەلمەيدۇ: كىت ياكى گالدىن باشقا نەرسە ئەمەس ، ياكى بۇ خىلدىكى زوراۋان ، كونترول قىلغىلى بولمايدىغان ، ئەقىلسىز بۇزغۇچىدىن باشقا نەرسە يوق.

ھازىر ، ۋەھشىيلەر بىلەن مەدەنىيەتلىك بۇ كۆرۈنەرلىك پەرق بار گەرچە كېسەل ، مەدەنىيەتلىك ئادەم ئالتە ئاي ماسلىشىدىغان بولۇشى مۇمكىن ، ئومۇمەن قىلىپ ئېيتقاندا ، كېسەل ۋەھشىي بىر كۈندە يەنە يېرىمغا يېقىنلىشىدۇ. شۇنداق قىلىپ ، ياخشى پەيتتە مېنىڭ كۋېكېگ كۈچلەندى. ھەمدە ئۇزۇن ۋاقىت شامال ئەينىكىدە ئولتۇرغاندىن كېيىن (ئەمما كۈچلۈك ئىشتىھا بىلەن غىزالىنىش) تۇيۇقسىز پۇتىغا سەكرەپ ، قول ۋە پۇتىنى تاشلىدى ، ئۆزىگە ياخشى سوزۇلدى ، ئازراق ئەسندى ، ئاندىن كۆتۈرۈلگەن كېمىنىڭ بېشىغا باھار كەلدى ۋە بىر ھارپا ئۇرۇپ ، ئۆزىنى ئۇرۇشقا ماس كېلىدىغانلىقىنى جاكارلىدى.

ياۋايى خاھىش بىلەن ئۇ ھازىر جەسەت ساندۇقىنى دېڭىز-كۆكرەككە ئىشلەتتى ھەمدە ئۇنىڭ كەندىر سومكىسىنى بوشىتىپ ، شۇ يەرگە تىزىڭ. ئۇ نۇرغۇن بوش ۋاقىتلارنى سەرپ قىلىپ ، ھەر خىل سۈرەتلىك رەسىملەر ۋە سىزمىلار بىلەن قاپاقنى ئويۇپ چىقتى. ھەمدە ئۇ بۇ ئارقىلىق قوپال ئۇسۇلدا بەدىنىگە بۇرمىلانغان چەكمە رەسىملەرنى كۆچۈرمەكچى بولۇۋاتقاندەك قىلاتتى. ھەمدە بۇ چەكمە چېكىش ئۇنىڭ ئارىلىدىكى ئايرىلغان پەيغەمبەر ۋە كۆرگۈچىنىڭ ئەسىرى بولۇپ ، ئۇ بۇ قاتلاملىق بەلگىلەر بىلەن بەدىنىگە ئاسمان-زېمىننىڭ تولۇق نەزەرىيىسىنى ۋە ھەقىقەتكە ئېرىشىش سەنئىتى ھەققىدە سىرلىق رىسالى يازغان. شۇنداق قىلىپ كۇئېگېگ ئۆزىنىڭ مۇۋاپىق ئادەمدە ئاشكارىلىنىدىغان تېپىشماق ئىدى. بىر توملۇق ئاجايىپ ئەسەر گەرچە ئۇنىڭ تىرىك يۈرىكى ئۇلارغا قارشى ئۇرغان بولسىمۇ ، ئۇنىڭ سىرلىرىنى ھەتتا ئۆزى ئوقۇيالمىغان. شۇڭلاشقا بۇ سىرلار ئاخىرىدا ئۇلار يېزىلغان تىرىك تېرىدىن يىراقلىشىپ ، ئاخىرقىسىغىچە ھەل قىلىنمىغان. بۇ ئوي چوقۇم ئاخابقا ئۆزىنىڭ ياۋايى خىتابنامىسىنى ئوتتۇرىغا قويغان بولسا كېرەك ، بىر كۈنى ئەتىگەندە بىچارە كۋېكېگنى تەكشۈرۈشتىن يۈز ئۆرۈگەندە - «ئاھ ، ئىلاھلارنى شەيتانغا مايىل قىلىش!».





111- باب تىنچ ئوكيان.

باشې ئارىلىدىن سىيرىلغاندا بىز ئاخىرى چوڭ جەنۇبىي دېڭىزدا پەيدا بولدۇق. باشقا ئىشلار بولمىغان بولسا ، مەن سۆيۈملۈك تېنچ ئوكياننى سانسىز رەھمەت بىلەن كۈتۈۋالغان بولاتتىم ، چۈنكى ھازىر ياشلىقىمنىڭ ئۇزۇن دۇئالىرى جاۋابقا ئېرىشتى. ئۇ خاتىرجەم ئوكيان مەندىن شەرققە قاراپ ئۆرۈلۈپ كەتتى.

بۇ يەردە ، بۇ دېڭىز ھەققىدە نېمىدېگەن تاتلىق سىر بارلىقىنى بىلمەيدۇ ، ئۇنىڭ ئاستا-ئاستا قورقۇنچلۇق غىدىقلىشى ئاستىدىكى بەزى يوشۇرۇن روھلارنى سۆزلەيدىغاندەك قىلىدۇ. ئەفەسلىك سودنىڭ كۆمۈلگەن خۇش خەۋەرچى ساينىت جون ئۈستىدىكى ئاشۇ ئەپسانىلەرگە ئوخشاش. شۇنى كۆرۈۋېلىشقا بولىدۇكى ، بۇ تۆت يايلاقتىكى بۇ يايلاق ، كەڭرى ئېقىن سۇ يايلىقى ۋە كۇلالچىلار ئېتىزىدا ، دولقۇن كۆتۈرۈلۈپ يىقىلىشى ، توختىماي ئېقىشى كېرەك. چۈنكى بۇ يەردە مىليونلىغان ئارىلاشما سايە ۋە سايە ، سۇغا غەرق بولغان چۈش ، بىر قىسىم ئادەتلەر ، ھۆرمەتلەر بىز ھايات ۋە جان دەپ ئاتايدىغانلارنىڭ ھەممىسى يالغان ، خىيال قىلىش ، چۈش كۆرۈش ، يەنىلا كارىۋاتلىرىدىكى ئۇيقۇغا ئوخشاش تاشلاش توختىماي لەرزان دولقۇن ، ئەمما ئۇلارنىڭ بىئاراملىقى بىلەن شۇنداق قىلدى.

ھەر قانداق تەپەككۇر قىلىدىغان Magian rover غا نىسبەتەن ، بۇ تىنچ ئوكيان تىنچ ئوكياننى كۆرگەندىن كېيىن ، چوقۇم ئۇنى بېقىۋالغان دېڭىز بولۇشى كېرەك. ئۇ دۇنيادىكى ئەڭ ئوتتۇرا سۇنى ، ھىندى ئوكيان ۋە ئاتلانتىك ئوكياننى ئۆز ئىچىگە ئالىدۇ. ئوخشاش دولقۇن كالىفورنىيەنىڭ يېڭىدىن سېلىنغان شەھەرلىرىنىڭ موللاقلىرىنى يۇيۇپ بېرىدۇ ، ئەمما تۈنۈگۈن ئەڭ يېقىنقى ئەرلەر مۇسابىقىسى تېرىدى ، ھەمدە ئاسىيا زېمىنىنىڭ سۇسلاشقان ، ئەمما يەنىلا ئېسىل يوپكىلىرىنى ئىبراھىمدىنمۇ چوڭ قىلدى. ھالبۇكى مارجان ئارىلىنىڭ لەيلىمە سۈت يولى بىلەن تۆۋەن يالغان ، تۈگىمەس ، نامەلۇم تاقىم ئارىلى ۋە ئۆتكىلى بولمايدىغان ياپونلار ئارىسىدا. شۇنداق قىلىپ بۇ سىرلىق ، ئىلاھىي تىنچ ئوكيان رايونى پۈتۈن دۇنيانى قاپلىدى. بارلىق دېڭىز قىرغاقلىرىنى ئۇنىڭغا بىر قولتۇق قىلىدۇ. يەرشارىنىڭ دولقۇن ئۇرغان يۈرىكىدەك قىلىدۇ. ئاشۇ مەڭگۈلۈك ئىششىقتىن كۆتۈرۈلگەندە ، چوقۇم جەلپكار ئىلاھقا ئىگە بولۇشىڭىز كېرەك ، بېشىنى پانغا ئېگىشىڭىز كېرەك.

ئەمما پان توغرىسىدىكى ئازغىنە خىياللار ئاخابنىڭ مېڭىسىنى لەرزىگە سالدى ، خۇددى مىزاجنىڭ يېنىدا كۆنۈپ قالغان جايدا تۆمۈر ھەيكەلدەك تۇرغاندەك ، بىر بۇرۇن تۆشۈكى بىلەن ئۇ باشې ئارىلىدىكى شېكەر مۇسكۇلى (ئۇ تاتلىق ئورمانلىقتا مۇلايىم ئاشىق-مەشۇقلار مېڭىشى كېرەك) ، يەنە بىرى ئاڭلىق ھالدا يېڭى تېپىلغان دېڭىزنىڭ تۇز نەپسىنى سۈمۈردى. نەپرەتلىنىدىغان ئاق كىت ھەتتا ئۇ ۋاقىتتا چوقۇم سۇ ئۈزۈشى كېرەك. ئۇزۇنغا سوزۇلغان بۇ ئاخىرقى سۇلارغا ئۇزۇن قويۇپ بېرىلىپ ، ياپون ساياھەت پاراخوتىغا قاراپ سىيرىلىپ ، بۇ بوۋاينىڭ مەقسىتى تېخىمۇ كۈچەيدى. ئۇنىڭ مۇستەھكەم لەۋلىرى بىر لەۋنىڭ لەۋلىرىدەك ئۇچراشتى. پېشانىسىدىكى تومۇر دېلتىسى قاپلانغان ئېقىندەك يوغىناپ كەتتى ئۇ ناھايىتى ئۇخلاۋاتقاندا ، ئۇنىڭ جاراڭلىق ئاۋازى ۋارقىراپ ئۆڭكۈردىن يۈگۈرۈپ: «ھەممىنى ستېرلىڭ! ئاق كىت قېلىن قان تۆكتى!» دېدى.





112-باب. تۆمۈرچى.

ھازىر بۇ كەڭلىكلەردە ھۆكۈمرانلىق قىلىۋاتقان مۇلايىم ، يازدىكى سالقىن ھاۋارايىدىن ئۆزىنى قاچۇرۇپ ، تەقەززالىق بىلەن ئۇزۇن ئۆتمەيلا ئالاھىدە ئاكتىپ قوغلىشىشقا تەييارلىق قىلىش ئۈچۈن ، تىلەمچىلىك قىلغان ، قاپارتقان كونا تۆمۈرچى پېرت ئېلىپ يۈرۈشكە ئەپلىك ئويغىنىنى يەنە تۇتۇپ تۇرالمىدى ، ئەمما ئاخابنىڭ پۇتىغا تۆھپە قوشۇشنى ئاخىرلاشتۇرغاندىن كېيىن ، ئۇنى يەنىلا پەلەمپەيدە ساقلاپ قويدى. ھازىر ئۇلار باشلىق ، ھارپا چالغۇچى ۋە ئوقياچىلار تەرىپىدىن توختىماي چاقىرىق قىلىنىۋاتىدۇ. ئۇلارنىڭ ھەر خىل قوراللىرى ۋە كېمە ئۆي جاھازىلىرىنى ئۆزگەرتىش ، رېمونت قىلىش ياكى يېڭى شەكىللەندۈرۈش. ھەمىشە ئۇنى ھەۋەس قىلىدىغان چەمبىرەك ئوراپ تۇراتتى ، ھەممىسى مۇلازىمەت قىلىشنى ساقلايتتى كېمە كالتەك ، پايپاق بېشى ، ھارپا ۋە نەيزە تۇتۇپ ، جاپالىق ئىشلەۋاتقاندا ئۇنىڭ ھەر بىر تىنىچ ھەرىكىتىنى ھەسەت بىلەن كۆزىتىۋاتىدۇ. قانداقلا بولمىسۇن ، بۇ بوۋاينىڭ سەۋرچان بولقا بىلەن سەۋرچان بولقا ئىدى. ئۇنىڭدىن زارلىنىش ، تاقەتسىزلىك ، ھاياجانلىنىش يوق. جىمجىت ، ئاستا ۋە تەنتەنىلىك ئۇ ئاستا-ئاستا سۇنغان دۈمبىسىنى يەنە بىر قېتىم ئېگىلىپ ، جاپا تارتتى ، خۇددى جاپالىق ھاياتنىڭ ئۆزىدەك ، بولقانىڭ قاتتىق ئۇرۇلۇشى ئۇنىڭ يۈرىكىنىڭ قاتتىق سوقۇشى. شۇنداق بولدى. - ئەڭ ئېچىنىشلىق!

بۇ بوۋايدا ئۆزىگە خاس سەيلە قىلىش ، مېڭىشىدا بەلگىلىك سەل ئەمما ئازابلىق كۆرۈنگەن ئەسكىرى سەپەرنىڭ دەسلەپكى مەزگىلىدە دېڭىزچىلارنىڭ قىزىقىشىنى قوزغىدى. ئۇلارنىڭ توختىماي سورىغان سوئاللىرىنىڭ مۇھىملىقىغا ئۇ ئاخىرى بەردى. شۇنداق قىلىپ ھەممە ئادەم ئۇنىڭ ئېچىنىشلىق تەقدىرىنىڭ نومۇسسىز ھېكايىسىنى بىلدى.

كېچىكىپ ، گۇناھسىز ئەمەس ، بىر ئاچچىق قىش يېرىم كېچىدە ، ئىككى يېزا-بازار ئارىلىقىدا كېتىۋاتقان يولدا ، تۆمۈرچى يېرىم ئەخمەقلىق بىلەن ئۇنىڭ ئۈستىدىن ئەجەللىك غۇۋالىقنى ئوغرىلىغانلىقىنى ھېس قىلدى ۋە ئېڭىشىپ كەتكەن ، كونىراپ كەتكەن قوتاندىن پاناھلىق تىلىدى. مەسىلە ، ئىككى پۇتنىڭ ئۇچىنى يوقىتىش ئىدى. بۇ ۋەھىيدىن قىسمەن ھالدا خۇشاللىقنىڭ تۆت خىل ھەرىكىتى چىقتى ، يەنە بىرى ئۇنىڭ ھاياتىدىكى دراما قايغۇسىنىڭ ئۇزۇن ، ئەمما تېخى ئىسپاتلانمىغان بەشىنچى ھەرىكىتى.

ئۇ بىر بوۋاي بولۇپ ، ئۇ ئاتمىش ياشقا كىرگەندە قايغۇ-ھەسرەت تېخنىكىسىدا بۇ نەرسىنى كېچىكتۈرگەن. ئۇ داڭلىق مۇنەۋۋەر ھۈنەرۋەن بولۇپ ، قىلىدىغان ئىشلىرى كۆپ ئىدى. ئۆي ۋە باغقا ئىگە بولغان ياشلىق ، قىزىغا ئوخشاش ، سۆيۈملۈك ئايالى ۋە ئۈچ ئاقكۆڭۈل ، قوپال بالىلارنى قۇچاقلىدى ھەر يەكشەنبە كۈنى دەرەخزارلىققا تىكىلگەن خۇشخۇي چىركاۋغا باراتتى. ئەمما بىر كۈنى كەچتە ، قاراڭغۇلۇقنىڭ ئاستىدا ، ۋە تېخىمۇ ھىيلىگەر قىياپەتتە تېخىمۇ يوشۇرۇنۇپ ، ئۈمىدسىز ئوغرى ئۇنىڭ خۇشال ئۆيىگە سىيرىلىپ كىرىپ ، ئۇلارنىڭ ھەممىسىنى بۇلىدى. تېخىچە قاراڭغۇ ، تۆمۈرچىنىڭ ئۆزى نادانلارچە بۇ ئوغرىنى ئائىلىسىنىڭ قەلبىگە ئېلىپ باردى. بۇ بوتۇلكا ئۇلىغۇچ! ئاشۇ ئەجەللىك قورۇما ئېچىلغاندىن كېيىن ، ئۇچۇپ يۈرۈپ ، ئۆيىنى قىستىدى. ھازىر ، ئاقىلانە ، ئەڭ دانا ۋە ئىقتىسادىي سەۋەبلەر تۈپەيلىدىن ، تۆمۈرچىنىڭ دۇكىنى ئۇنىڭ تۇرالغۇسىنىڭ يەر ئاستى ئۆيىدە ئىدى ، ئەمما ئۇنىڭغا ئايرىم كىرىش ئېغىزى بار ئىدى. شۇنداق قىلىپ ياش ۋە سۆيۈملۈك ساغلام ئايالىنى بەختسىز جىددىيلىك بىلەن ئاڭلىماي ، بەلكى كۈچلۈك خۇشاللىق بىلەن ياش قوراللىق ياشانغان ئېرىنىڭ بولقانىڭ جاراڭلىق ئاۋازى ئاڭلىدى. ئۇنىڭ قاينام-تاشقىنلىقلىرى پول ۋە تاملاردىن ئۆتۈپ گاڭگىراپ ، بالىلار باغچىسىدا تاتلىق ئەمەس ، ئۇنىڭ يېنىغا كەلدى. شۇنداق قىلىپ ، ئەمگەكچىلەرنىڭ تۆمۈر لۆڭگىسىنى پۇختا قىلىش ئۈچۈن ، تۆمۈرچىنىڭ بوۋاقلىرى ئۇيقۇغا كەتتى.

ۋاي ۋاي! ئەي ئۆلۈم ، نېمىشقا بەزىدە ۋاقتىدا بولالمايسەن؟ ئەگەر سىز بۇ كونا تۆمۈرچىنى پۈتۈنلەي ھالاك بولۇشتىن بۇرۇن ئۆزىڭىزگە ئېلىپ كەتكەن بولسىڭىز ، ئۇنداقتا بۇ ياش تۇل ئايالنىڭ مەززىلىك قايغۇسى بولغان بولسا ، يېتىم بالىلىرى ھەقىقىي ھۆرمەتكە سازاۋەر ، رىۋايەتلىك سىيىر بولۇپ ، ئۇلارنىڭ كېيىنكى يىللىرىدا ئارزۇ قىلاتتى. ئۇلارنىڭ ھەممىسىنىڭ ھالىدىن خەۋەر ئېلىش ئىقتىدارى بار. ئەمما ئۆلۈم بەزى پەزىلەتلىك چوڭ ئاكا-ئۇكىلارنى يىرتىپ تاشلىدى ، ئۇلارنىڭ كۈندىلىك پۇشتەك قىلىشى باشقا ئائىلىلەرنىڭ مەسئۇلىيىتىنىلا ئېسىپ قويدى ، ھەمدە پايدىسىز بوۋايدىنمۇ ناچارسىنى تۇرغۇزدى ، تاكى ھاياتنىڭ يوشۇرۇن چىرىتىشى ئۇنى يىغىۋېلىشقا قولايلىق ياراتتى.

نېمىشقا كۆپچىلىككە دەيسىز؟ يەر ئاستى بولقانىڭ ئۇرۇلۇشى كۈندىن-كۈنگە كۆپەيدى. ھەر بىر قېتىملىق زەربە ھەر كۈنىگە قارىغاندا سۇسلاپ كەتتى. ئايالى دېرىزىدە مۇزلاپ ئولتۇردى ، كۆز ياشلىرى بىلەن پارقىراپ بالىلىرىنىڭ يىغلاۋاتقان چىرايىغا تىكىلدى. قوڭغۇراق چۈشۈپ كەتتى. ساختا ساندۇقلار بىلەن بوغۇلدى. ئۆي سېتىلدى ئانا ئۇزۇن چېركاۋ ھويلىسىدىكى ئوت-چۆپلەرگە شۇڭغۇدى. ئۇنىڭ بالىلىرى ئىككى قېتىم ئۇنىڭغا ئەگىشىپ ماڭدى. ئۆي-ماكانسىز ، ئۆي-ماكانسىز بوۋاي كەركىداندا سەرگەردان بولۇپ كەتتى. ئۇنىڭ بارلىق ۋايلىرى ھۆرمەتسىز. ئۇنىڭ كۈلرەڭ بېشى زىغىر بۈدرە ئۈچۈن مەسخىرە!

ئۆلۈم مۇشۇنىڭغا ئوخشاش كەسىپ ئۈچۈن بىردىنبىر كۆڭۈلدىكىدەك داۋامىدەك قىلىدۇ. ئەمما ئۆلۈم پەقەت غەلىتە سىناق قىلىنمىغان رايونغا قويۇپ بېرىش. ئۇ پەقەت غايەت زور يىراق ، ياۋا ، سۇ ، قۇرۇقلۇقنىڭ مۇمكىنچىلىكىنىڭ تۇنجى سالىمى. شۇڭلاشقا ، ئۇلاردا يەنىلا ئۆزىنى ئۆلتۈرىۋېلىشقا قارشى بىر قىسىم ئىچكى ئىقتىدارلارنى قالدۇرۇپ قويغان بۇنداق كىشىلەرنىڭ ئۆلۈمگە ئىنتىلىدىغان كۆزىگە ، بارلىق تۆھپىلەرنى قوشقان ۋە قوبۇل قىلىدىغان دېڭىز-ئوكيان كىشىنى ھەيران قالدۇرغۇدەك دەرىجىدە تۈزلەڭلىكنى كېڭەيتتى ، ۋەھىمە ئېلىپ ، ئاجايىپ ، يېڭى ھايات سەرگۈزەشتىلىرىنى قىلدى. چەكسىز تىنچ ئوكياننىڭ قەلبىدىن مىڭ سۇ پەرىسى ئۇلارغا ناخشا ئېيتىدۇ - «بۇ يەرگە كېلىڭلار ، يۈرىكى سۇنۇپ كەتتى ؛ بۇ يەردە ئارىلىقتىكى ئۆلۈم گۇناھسىز يەنە بىر ھايات بار ؛ بۇ يەردە ئادەتتىن تاشقىرى مۆجىزىلەر بار ، ئۇلار ئۈچۈن ئۆلمەيسىز . بۇ يەرگە كېلىڭلار! تاكى بىز سىز بىلەن توي قىلغۇچە! ».

بۇ ئاۋازلارنى ئاڭلاپ ، شەرق ۋە غەرب ، بالدۇر كۈن چىقىش ۋە كەچ كۈزدە ، تۆمۈرچىنىڭ روھى جاۋاب بېرىپ: «ھە ، مەن كەلدىم! شۇنداق قىلىپ پېرت كىت كىتتى.





113-باب.

ساقال قويغان ، ھەمدە پارقىراپ تۇرغان لەھەڭ تېرىسىدىن ياسالغان پەلەيگە چۆمگەن بولۇپ ، يېرىم كۈن ئەتراپىدا ، پېرت ئۇنىڭ ئويمانلىقى بىلەن مۈڭگۈزىنىڭ ئوتتۇرىسىدا تۇرغان ، كېيىنكىسى تۆمۈر ياغاچ ياغىچىغا قويۇلغان ، بىر قولى كۆمۈرگە پىكاپ بېشىنى تۇتۇۋالغان ، يەنە بىر قولى بىلەن ئۇنىڭ ئۆپكىسىدە ، كاپىتان ئاخاب بىللە كەلگەندە ، قولىدا كىچىك داتلىشىپ كەتكەن خۇرۇم سومكىنى كۆتۈرگەن. قورشاۋدىن سەل يىراق بولسىمۇ ، مىجەزى يوق ئاخاب توختاپ قالدى ئاخىرىغا قەدەر ، پېرت تۆمۈرنى ئوتتىن تارتىپ ، ئۇنى مۈڭگۈزگە ئۇرۇپ ئۇرۇشقا باشلىدى - قىزىل ماسسا ئۇچقاندەك ئۇچۇپ ئۇچۇپ ئۇچۇپ كەتتى ، ئۇلارنىڭ بەزىلىرى ئاخابقا يېقىن ئۇچتى.

«بۇلار ئاناڭ كارىينىڭ توخۇلىرىمۇ ، پېرت؟ ئۇلار ھەمىشە سېنىڭ ئويغىنىشىڭدا ئۇچۇۋاتىدۇ ؛ ياخشى بېشارەت قۇشلىرىمۇ ، ئەمما ھەممىگە ئەمەس ؛ - بۇ يەرگە قاراڭ ، ئۇلار كۆيۈپ كەتتى ؛ ئەمما سەن - ئۇلارنىڭ ئارىسىدا كۆيدۈرمەي ياشايسەن».

پېرت جاۋاب بېرىپ: «مەن ھەممىلا جايدا كۆيۈپ كەتتىم ، كاپىتان ئاخاب» دېدى. «مەن كۆيۈپ ئۆتۈپ كەتتىم ، ئاسانلا داغنى كۆيدۈرەلمەيسىز».

"ياخشى ، ياخشى ؛ ئەمدى بولمايدۇ. كىچىكلىگەن ئاۋازىڭىز ماڭا بەك تەمكىن ، ساراڭلارچە ۋاي. مەن ھېچقانداق جەننەتتە ئەمەس ، مەن ساراڭ ئەمەس باشقىلاردىكى بارلىق ئازاب-ئوقۇبەتلەرگە تاقەت قىلالمايمەن. سىز ساراڭ بولۇپ قالماسلىقىڭىز كېرەك.

«ئەپەندىم كونا پىلە بېشىنى كەپشەرلە ، ئەپەندى ؛ ئۇنىڭدا يوچۇق ۋە چىش بار ئىدى».

«تۆمۈرچى ، بۇرۇنقىدەك قاتتىق ئىشلىتىلگەندىن كېيىن ، ئۇنى ھەممىنى ئوڭۇشلۇق قىلالامسىز؟».

- ئەپەندىم ، مېنىڭچە شۇنداق.

«مېنىڭچە ، سىز ھەر قانداق يوچۇق ۋە چىشلارنى دېگۈدەك راۋانلاشتۇرالايسىز دەپ ئويلايمەن ؛ مېتال ، تۆمۈرچىنىڭ قانچىلىك جاپالىق ئىكەنلىكىنى ھەرگىز ئۇنتۇپ قالماڭ».

«ھەئە ، ئەپەندىم ، مېنىڭچە مەن قىلالايمەن ؛ بارلىق يوچۇق ۋە چىشلار بىر».

ئاخاب ھاياجانلانغان ھالدا: «بۇ يەرگە قاراڭلار» دەپ ۋارقىرىدى ۋە ئىككى قولى بىلەن پېرتنىڭ مۈرىسىگە يۆلىنىپ تۇرۇپ قالدى. «بۇ يەرگە قاراڭ - مانا ، تۆمۈرچى ، مۇشۇنداق بىر يوچۇقنى راۋانلاشتۇرالامسىز؟» بىر قولىنى قوۋۇرغىسىنىڭ ئۇچىدىن سۈپۈردى. «ئەگەر سەن قىلالايدىغان بولساڭ ، تۆمۈرچى ، خۇشاللىقىمدىن بېشىمنى مۈڭگۈزۈمگە قويۇپ ، كۆزلىرىمنىڭ ئارىسىدىكى ئەڭ ئېغىر بولقانى ھېس قىلاتتىم. جاۋاب! بۇ تامنى راۋانلاشتۇرالامسەن؟».

- ھە ، ئۇ شۇ ، خوجايىن!

«ئەي ، تۆمۈرچى ، ئۇ شۇ ؛ ھەي ، ئادەم ، ئۇ كىشىنىڭ ئەقلى يەتمەيدۇ ؛ چۈنكى سىز ئۇنى پەقەت مېنىڭ تېنىمدىلا كۆرسىڭىزمۇ ، ئۇ مېنىڭ باش سۆڭىكىمنىڭ سۆڭىكىگە چۈشۈپ كەتتى - مانا بۇلارنىڭ ھەممىسى قورۇق! خۇرۇم سومكىنى خۇددى ئالتۇن پۇللار بىلەن تولغاندەك. «مەنمۇ بىر ھارپا ياساپ چىقىشنى خالايمەن ؛ مىڭ بويۇن ئومۇرتقىسى پارچىلىنىپ كېتەلمەيدىغان پېرت ؛ ئۆزىنىڭ ئىنچىكە سۆڭىكىگە ئوخشاش كىتتا قالىدىغان نەرسە بار. «قاراڭ ، تۆمۈرچى ، بۇلار بەيگە ئاتلىرىنىڭ پولات ئاياغلىرىنىڭ يىغىلغان مىخ پاچىقى».

- ئەپەندىم ، ئات ئاياغ پاچىقى ، ئەپەندى ، نېمىشقا كاپىتان ئاخاب ، سىز بۇ يەردە ، بىز تۆمۈرچىلەر ئىشلىگەن ئەڭ ياخشى ۋە جاھىل نەرسىلەر بار؟

"مەن بىلىمەن ، بوۋاي ؛ بۇ پاخاللار قاتىللارنىڭ ئېرىگەن سۆڭىكىدىن يېلىمدەك بىللە كەپشەرلىنىدۇ. تېز! ماڭا ھارپا ياساپ بېرىڭ. ھەمدە مېنى ئالدى بىلەن ئون ئىككى كالتەك بىلەن ياسىڭ ؛ ئاندىن شامال ، بۇرمىلاپ ، بۇ ئون ئىككىسىنى سۆرەش لىنىيىسىنىڭ يىپلىرى ۋە تاياقلىرىدەك بولقا بىلەن ئۇرۇڭ. تېز! مەن ئوتنى ئۇراي."

ئاخىرى ئون ئىككى تاياق ياسالغاندا ، ئاخاب ئۇزۇن ، ئېغىر تۆمۈر بولتىنى چۆرىدىگەن ھالدا ئۆز قولى بىلەن ئۇلارنى بىر-بىرلەپ سىلاپ سىناپ باقتى. «نۇقسان!». ئاخىرقىسىنى رەت قىلىش. «قايتا ئىشلەڭ ، پېرت».

بۇ ئىش تاماملانغاندىن كېيىن ، پېرت ئون ئىككىنى كەپشەرلەشكە باشلىماقچى بولغاندا ، ئاخاب قولىنى ساقلاپ ، ئۆزىنىڭ تۆمۈرنى كەپشەرلەيدىغانلىقىنى ئېيتتى. شۇنىڭدىن كېيىن ، ئۇ دائىملىق ، گازلىق پاقالچاقلار بىلەن ئۇۋاققا بولقا ئۇردى ، پېرت كەينى-كەينىدىن پارقىراپ تۇرغان تاياقلارنى ئۇنىڭغا يەتكۈزدى ، قاتتىق بېسىلغان ئويما ئۇنىڭ كۈچلۈك تۈز يالقۇنىنى ئېتىۋەتكەندىن كېيىن ، پارسې ئۈن-تىنسىز ئۆتۈپ كەتتى ۋە بېشىنى ئېگىپ ئوتقا قاراپ باش ئەگدى. ئەمما ، ئاخاب قارىغىنىچە بىر چەتكە قاراپ كەتتى.

«ئۇ يەردە توپ تېپىۋاتقان بىر تۈركۈم لەۋ سۇرۇخلار نېمە؟». - دەپ ۋارقىرىدى Stubb ، ئالدىن پەرەزدىن. «ئۇ پارسېي ئوتنىڭ پۇرىقىدەك پۇرايدۇ ؛ ھەمدە ئۇنىڭ پۇرىقى ئىسسىق مۇسكۇلنىڭ پاراشوكىدەك».

ئاخىرى شانكا ، بىر مۇكەممەل تاياقتا ئاخىرقى ئىسسىقنى تاپشۇرۇۋالغان. پېرت ئۇنى پەسەيتىش ئۈچۈن ، ئۇنىڭ شىۋىرغان ئاۋازىنى يېقىن ئەتراپتىكى سۇ ساندۇقىغا شۇڭغۇدى ، كۆيگەن ھور ئاخابنىڭ ئېگىلىپ كەتكەن يۈزىگە ئېتىلدى.

- پېرت ، سەن مېنى ماركا قىلامسەن؟ ئاغرىق بىلەن بىر ئاز تەۋرىنىش «مەن ئۆزۈمنىڭ ماركا-تۆمۈرنى ياسىغان بولسام ، ئۇنداقتا؟».

«خۇداغا دۇئا قىلىڭ ، ئۇنداق ئەمەس ؛ شۇنداقتىمۇ مەن بىر ئىشتىن قورقىمەن ، كاپىتان ئاخاب. بۇ ئاق كىت ئۈچۈن بۇ ھارپا ئەمەسمۇ؟».

«ئاق تەنلىكلەر ئۈچۈن!

بىر مەزگىلدىن كېيىن ، كونا تۆمۈرچى تىغ ئۇچىنى ئىشلەتمەي ھوشىدىن كەتكەندەك كۆزىگە قارىدى.

«ئۇلارنى ئېلىڭ ، ئىنسان ، مېنىڭ ئۇلارغا ئېھتىياجىم يوق ؛ چۈنكى مەن ھازىر ساقال قويمايمەن ، تاماقمۇ قىلمايمەن ، ياكى دۇئا قىلمايمەن ، ئەمما بۇ يەردە ئىشلەيمەن!».

ئەڭ ئاخىرىدا ئوقيا شەكلىدە ياسالغان ، پېرت تەرىپىدىن كەپشەرلەپ كەپشەرلەنگەن پولات ئۇزۇن ئۆتمەي تۆمۈرنىڭ ئۇچىنى كۆرسەتتى. تۆمۈرچى قاۋاقخانىلارغا ئاخىرقى ئىسسىقلىقنى بەرمەكچى بولغاندا ، ئۇلارنى قىزىتىشتىن بۇرۇن ، ئۇ ئاخابقا ۋارقىراپ سۇ قاچىسىنى يېقىن ئورۇنغا قويدى.

"ياق ، ياق - بۇنىڭ ئۈچۈن سۇ يوق ؛ مەن ئۇنى ھەقىقىي ئۆلۈم خاھىشىدىن ئۈمىد قىلىمەن. ئاخوي ، ئۇ يەردە! تاشتېگو ، كۋېكېگ ، داگگو! نېمە دەيسىز ، مۇشرىكلار! ماڭا بۇ كاۋاپداننى يېپىپ قويغۇدەك قان بەرەمسىز؟" ئۇنى ئېگىز كۆتۈرۈڭ. بىر توپ قاراڭغۇ باش لىڭشىتىپ جاۋاب بەردى. يات گۆشلەردە ئۈچ تۆشۈك ياسالدى ، ئاندىن ئاق كىتنىڭ كاۋاپچىلىرى مىجەز بولدى.

«نامەلۇم پاترىستا Ego non baptizo te ، نامدىكى دىئابولىدا ئولتۇرۇش!». يامان غەرەزلىك تۆمۈر چۆمۈلدۈرۈش قېنىنى يەپ كەتكەچكە ، ئاخاب قەستەن يىغلىدى.

ھازىر ، زاپاس تۈۋرۈكلەرنى تۆۋەندىن يىغىپ ، بىر تال چىۋىقنى تاللاڭ ، پوستى يەنىلا ئۇنىڭغا مەبلەغ سالدى ، ئاخاب ئۇچىنى تۆمۈرنىڭ ئېغىزىغا ماسلاشتۇردى. شۇنىڭدىن كېيىن بىر تال يېڭى سۆرەش لىنىيىسى يارىلانمىدى ، ئۇنىڭ بەزى خىياللىرى شامال ئەينىكىگە ئېلىپ كېلىندى ۋە قاتتىق جىددىيلىككە سوزۇلدى. ئارغامچا ئارغامچادەك غۇڭۇلداپ ، ئاندىن تەقەززالىق بىلەن ئېگىلىپ ، ھېچقانداق تالنى كۆرمىگۈچە ، ئاخاب: «ياخشى! ئەمدى تۇتۇپ قېلىش ئۈچۈن» دەپ ۋارقىرىدى.

بىر ئۇچىدا ئارغامچا ئۈزۈلمەي ، ئايرىم تارقالغان يىپلارنىڭ ھەممىسى ئۆرۈلۈپ ، ھارپوننىڭ ئۇدۇلىغا توقۇلدى. ئاندىن بۇ قۇتۇپ قاتتىق ئوققا تۇتۇلدى. ئاستىنقى ئۇچىدىن ئارغامچا قۇتۇپنىڭ ئۇزۇنلۇقى بىلەن يېرىم يولدا ئىز قوغلانغان ۋە مۇستەھكەم كاپالەتكە ئىگە قىلىنغان بولۇپ ، ئۆز-ئارا گىرەلىشىپ كەتكەن. بۇ ئىش ، قۇتۇپ ، تۆمۈر ۋە ئارغامچا خۇددى ئۈچ تەقدىرگە ئوخشاش بىر-بىرىدىن ئايرىلالمىدى ، ئاخاب كەيپىيات بىلەن قورال بىلەن يىراقلاشتى. ئۇنىڭ پىل چىشىنىڭ پۇتى ۋە چىۋىق قۇتۇبىنىڭ ئاۋازى ھەر ئىككى تاختاينى بويلاپ بوش سايرىدى. ئەمما ئۇ كابىنكىسىغا كىرىشتىن بۇرۇن ، يېنىك ، تەبىئىي ، يېرىم چەكلەش ، ئەمما ئەڭ ئېچىنىشلىق ئاۋاز ئاڭلاندى. ئاھ ، تۇرۇبا! بىچارە كۈلكىڭىز ، بىكار ، ئەمما ئەنسىز كۆزىڭىز. بارلىق غەلىتە مۇمىيالىرىڭىز مۇڭلۇق پاراخوتنىڭ قارا پاجىئەسى بىلەن ئارىلاشماي ، ئۇنى مەسخىرە قىلدى!
