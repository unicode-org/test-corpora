باب 101. ڈیکنٹر۔

اس توں پہلاں کہ انگریزی جہاز نظر توں اوجھل ہو جائے، چاہے اوہ ایتھے رکھی ہووے، کہ اوہ لندن توں آئی سی، تے اس دا ناں مرحوم سیموئیل اینڈربی دے ناں تے رکھیا گیا سی، جو اوس شہر دے تاجر سی، اینڈربی اینڈ سنز دے مشہور وہیلنگ ہاؤس دا اصل؛ اک اجہا گھر جہڑا میرے غریب وہیلمین دی رائے وچ، حقیقی تاریخی دلچسپی دے لحاظ نال، ٹیوڈرز تے بوربن دے متحدہ شاہی گھراں توں بہتا پچھے نئیں اے۔ ساڈے رب دے سال 1775 توں پہلاں، ایہہ عظیم وہیلنگ ہاؤس کدوں تک موجود سی، میرے بے شمار مچھی دستاویزات واضح نئیں کردے; لیکن اس سال (1775) وچ اس نے پہلے انگریزی جہازاں نو فٹ کیتا جہناں نے باقاعدگی نال سپرم وہیل دا شکار کیتا; اگرچہ کجھ سال پہلے (1726 توں) ساڈے بہادر تابوتاں تے نانٹکیٹ تے انگوراں دے باغ دے میسی نے وڈے بیڑیاں چ اس لیویتھن دا تعاقب کیتا سی، لیکن صرف شمالی تے جنوبی بحر اوقیانوس چ: کہتھے ہور نئیں۔ ایتھے ایہہ واضح طور تے درج کیتا گیا اے، کہ نانٹکیٹرز بنی نوع انسان چوں پہلے سن جنہاں نے مہذب فولاد دے نال عظیم سپرم وہیل نوں ہارپون کیتا؛ تے ایہ کہ ادھی صدی تک اوہ پوری دنیا دے واحد لوک سن جنہاں نے اس نوں اینا ماریا۔

1778 وچ، اک چنگا جہاز، امیلیا، ایکسپریس مقصد لئی تیار کیتا گیا، تے زبردست اینڈربیز دے واحد چارج تے، دلیری نال کیپ ہارن نوں گول کیتا، تے ملکاں وچوں پہلا سی جس نے عظیم جنوبی سمندر وچ کسی وی قسم دی وہیل کشتی نوں اتاریا۔ ایہہ سفر اک مہارت تے خوش قسمت سی; تے قیمتی سپرم نال بھری ہولڈ دے نال اپنڑی برتھ تے واپس آندیاں ہوئیاں، امیلیا دی مثال دے بعد جلدی ای دوجے جہاز، انگریزی تے امریکی، تے اس طرح پیسیفک دے وسیع سپرم وہیل گراؤنڈز کھل گئے۔ پر ایس چنگے کم نال مطمئن نہ ہو کے، انتھک گھر نے فیر اپنے آپ نوں ہلایا: سیموئیل تے اوہدے سارے پتر - کنے، اوہناں دی ماں نوں ای پتہ اے - تے اوہناں دی فوری سرپرستی ہیٹھ، تے جزوی طور تے، میرا خیال اے، اوہناں دے خرچے تے، برطانوی حکومت نوں سلوپ آف وار ریٹلر نوں جنوبی سمندر وچ دریافت دے وہیلنگ دے سفر تے گھلن لئی اکسایا گیا۔ اک بحری پوسٹ کیپٹن دی کمانڈ چ، ریٹلر نے اس دا اک کھنگھورا ماریا سفر کیتا، تے کجھ خدمت کیتی؛ کِنا ظاہر نئیں ہوندا۔ پر ایہہ سبھ کجھ نہیں۔ 1819 وچ، اسی گھر نے جاپان دے دور دراز دے پانیاں وچ چکھن آلے کروز تے جان لئی، اپنا اک ڈسکوری وہیل جہاز تیار کیتا۔ اوہ جہاز - جس نوں "سائرن" آکھیا جاندا اے - نے اک عظیم تجرباتی کروز بنایا؛ تے ایہ اس طرح سی کہ عظیم جاپانی وہیلنگ گراؤنڈ سب توں پہلے عام طور تے جانیا جاندا سی۔ اس مشہور سفر وچ سائرن دا کمانڈ اک کیپٹن کافن نے کیتا سی، جو اک نانٹکیٹر سی۔

اس لیی، اینڈربیز دا سارا اعزاز، جہناں دا گھر، میرے خیال چ، اج تک موجود اے; اگرچہ بلاشبہ اصل سیموئیل نے بہت پہلے دوجی دنیا دے عظیم جنوبی سمندر لئی اپنی کیبل پھسل دتی ہووے گی۔

اس دے ناں تے رکھیا گیا جہاز اس اعزاز دے قابل سی، کیونکہ اوہ اک بہت تیز جہازران سی تے ہر لحاظ نال اک عظیم جہاز سی۔ میں اک واری ادھی رات نوں پیٹاگونیا دے ساحل توں دور کدھرے اس تے سوار ہویا، تے فورکاسٹل چ چنگا فلپ ڈاؤن پیتا۔ ایہہ ساڈے کول اک چنگا گیم سی، تے اوہ سارے ٹرمپ سن—جہاز تے ہر روح۔ اوہناں لئی اک چھوٹی جہی زندگی، تے اک خوشگوار موت۔ تے اوہ چنگا گیم جہڑا میرے کول سی—بڈھے اخاب دے اپنی ہاتھی داند دی ایڑی نال اوہدے تختیاں نوں چھون دے بہت عرصے بعد—ایہہ مینوں اوس جہاز دی عظیم، ٹھوس، سیکسن مہمان نوازی دا یاد دلاندا اے؛ تے میرا پادری مینوں بھل جائے، تے شیطان مینوں یاد رکھے، جے میں کدی ایہنوں نظر انداز کر لواں۔ فلپ کرو؟ کی میں آکھیا سی کہ ساڈے کول فلپ سی؟ ہاں، تے اسی اینوں دس گیلن فی گھنٹہ دی شرح نال پلٹیا; تے جدوں طوفان آیا (کیونکہ ایہہ اوتھے پیٹاگونیا دے کول طوفان اے)، تے سارے ہتھاں - زائرین تے سارے - ریف ٹاپ سیلز لئی بلایا گیا، اسی اینے ٹاپ ہیوی ساں کہ سانوں باؤلائنز چ اک دوجے نو اُچا جھولنا پیا؛ تے اسی نادانستہ طور تے اپنڑیاں جیکٹاں دے سکرٹ نو بادباناں چ لپیٹ دتا، تاکہ اسی اوتھے لٹک گئے، تیزی نال ہاکاں ماردے ہوئے جھکھڑ چ ڈبے ہوئے، سارے شرابی ٹاراں لئی اک انتباہ دی مثال۔ تاہم، مستول سمندر توں باہر نئیں گئے; تے ہولی ہولی اسی اتر گئے، اینے ہوشیار، کہ سانوں فیر پلٹنا پیا، حالانکہ فورکاسٹل سکٹل دے تھلے پھٹن آلا وحشی نمک دا چھڑکاؤ، میرے ذوق دے مطابق بہوں زیادہ پتلا تے اچار بنا دتا۔

گائے دا گوشت ٹھیک سی - سخت، لیکن اس وچ جسم دے نال۔ اوہناں نے آکھیا کہ ایہہ بیل دا گوشت سی; دوجے، کہ ایہہ ڈرومیڈری بیف سی; لیکن میں نئیں جانندا، پک نال، کہ ایہ کیویں سی۔ اوہناں کول پکوڑے وی سن؛ نکے، لیکن کافی، سڈول گلوبل، تے ناقابل تباہی پکوڑے۔ میں سوچیا کہ تسی اوہناں نو محسوس کر سکدے او، تے اوہناں نو نگل جان دے بعد اپنے اندر رول کر سکدے او۔ جے تسی بہت اگے جھک گئے، تے تسی اوہناں دی پچنگ نو بلئرڈ گینداں دی طرح خطرے چ پا دتا۔ روٹی—پر اس نال کوئی مدد نئیں کیتی جا سکدی سی; اس توں علاوہ، ایہ اینٹی سکوربوٹک سی; مختصر طور تے، روٹی چ صرف تازہ کرایہ سی جہڑا اوہناں کول سی۔ لیکن فورکاسٹل بہوں ہلکا نئیں سی، تے جدوں تسی ایہنوں کھاندے او تے اک ہنیرے کونے چ قدم رکھنا بہوں سوکھا سی۔ لیکن سب کجھ، اس نوں ٹرک توں پتوار تک لے جانا، باورچی دے بوائلراں دے طول و عرض تے غور کرنا، جس چ اس دے اپنے زندہ چرمپتر بوائلر وی شامل سن; اگے تے پچھے، میں کہنا واں، سیموئیل اینڈربی اک خوشگوار جہاز سی; چنگا کرایہ تے کافی مقدار دا; ٹھیک پلٹنا تے مضبوط؛ کریک فیلوز سب، تے بوٹ ہیلس توں لے کے ہیٹ بینڈ تک سرمایہ۔

لیکن ایہہ کیوں سی، تُسی سوچدے او، کہ سیموئیل اینڈربی، تے کجھ ہور انگریزی وہیلرز جنہاں دے بارے چ میں جاننا واں - اگرچہ سارے نئیں - اینے مشہور، مہمان نواز جہاز سن; جہڑا گائے دے گوشت، تے روٹی، تے ڈبہ، تے مذاق دے آلے دوالے لنگھیا; تے کھان پین تے ہسن توں چھیتی ای تھک نئیں گئے؟ میں تہانوں دساں گا۔ ایہناں انگریزی وہیلرز دی بھرپور خوشی تاریخی تحقیق لئی معاملہ اے۔ نہ ای میں تاریخی وہیل دی تحقیق توں بالکل بچیا اے، جدوں کہ ایہدی لوڑ لگدی اے۔

انگریزاں توں پہلاں وہیل ماہی گیری چ ہالینڈرز، زیلینڈرز تے ڈینش لوک سن; جہناں توں اوہناں نے کئی اصطلاحات حاصل کیتیاں جہڑیاں ہلے وی ماہی گیری چ موجود نیں; تے ہور کی اے، اوہناں دے موٹے پرانے فیشن، کھان پین لئی بہت کجھ چھوہنا۔ کیونکہ، اک عام گل دے طور تے، انگریزی تجارتی جہاز اپنڑے عملے نو گھٹ کردا اے۔ پر انگریزی وہیلر انج نئیں۔ اس لیی، انگریزی چ، وہیلنگ چنگی خوشی دی ایہ گل عام تے فطری نئیں اے، بلکہ حادثاتی تے خاص اے; تے، اس لیی، اس دی کوئی خاص اصل ہونی چاہیدی اے، جس دی نشاندہی ایتھے کیتی گئی اے، تے فیر وی ہور وضاحت کیتی جائے گی۔

لیویتھینک تریخاں وچ اپنی تحقیق دے دوران، میں اک قدیم ڈچ جلد تے ٹھوکر کھادی، جہڑی، اس دی کھنڈ آلی وہیلنگ بو توں، میں جاننا سی کہ وہیلرز دے بارے وچ ہونا چاہیدا اے۔ عنوان سی، "ڈین کوپمین"، اس لیی میں ایہ نتیجہ اخذ کیتا کہ ایہ ماہی گیری چ کجھ ایمسٹرڈیم کوپر دی انمول یادداشتاں ہون گیاں، کیونکہ ہر وہیل جہاز نو اپنا کوپر لے جانا چاہیدا اے۔ میں ایس رائے وچ ایہہ ویکھ کے تقویت حاصل کیتی کہ ایہہ اک "فٹز سواک ہیمر" دی تیاری سی۔ لیکن میرا دوست ڈاکٹر سنوڈ ہیڈ، اک بہوں پڑھیا لکھیا بندہ، سانتا کلاز تے سینٹ پوٹ دے کالج چ لو ڈچ تے ہائی جرمن دے پروفیسر، جہنوں میں ترجمے لئی کم دتا، اوہنوں اوہدی مصیبت لئی سپرم موم بتیاں دا اک ڈبہ دتا - اوہی ڈاکٹر سنوڈ ہیڈ، جیویں ای اوہنے کتاب دی جاسوسی کیتی، مینوں یقین دوایا کہ "ڈی کوپر" دا مطلب "مرچنٹ" نئیں سی۔ مختصر طور تے، اس قدیم تے سیکھے ہوئے لو ڈچ کتاب نے ہالینڈ دی تجارت دا علاج کیتا; تے، دوجے مضامین دے نال، اس دی وہیل ماہی گیری دا اک بہوں دلچسپ بیان شامل سی۔ تے اس باب وچ ایہ سی، جس دا سرناواں سی، "سمیر"، یا "موٹا"، کہ میں ڈچ وہیلمین دے 180 بادباناں دے لارڈرز تے تہہ خانیاں دے لئی لباس دی اک لمبی تفصیلی فہرست ملی؛ جس فہرست چوں، جویں کہ ڈاکٹر سنوڈ ہیڈ نے ترجمہ کیتا اے، میں درج ذیل نو نقل کردا ہاں:

400,000 پونڈ۔ گوشت دا۔ 60,000 پونڈ۔ فریس لینڈ سور دا گوشت۔ 150,000 پونڈ۔ اسٹاک مچھی دا۔ 550,000 پونڈ۔ بسکٹ دا۔ 72,000 پونڈ۔ نرم روٹی دا۔ 2,800 فرکن مکھن۔ 20,000 پونڈ۔ ٹیکسل اینڈ لیڈن پنیر۔ 144,000 پونڈ۔ پنیر (شاید اک گھٹیا مضمون)۔ جنیوا دے 550 اینکرز۔ 10,800 بیرل بیئر۔

زیادہ تر شماریاتی جدولاں پڑھن چ سختی نال خشک ہوندیاں نیں; تاہم، موجودہ معاملے وچ انج نئیں اے، جتھے قاری نوں چنگی جن تے چنگی خوشی دے پورے پائپاں، بیرلاں، کوارٹس تے گلاں نال بھر دتا جاندا اے۔

اوس ویلے، میں تن دن ایس ساری بیئر، گوشت تے روٹی دے مطالعے نال ہضم کرن لئی وقف کیتے، جس دے دوران مینوں اتفاقی طور تے کئی ڈوہنگے خیالات تجویز کیتے گئے، جہڑے اک ماورائی تے افلاطونی اطلاق دے قابل سن؛ تے، مزید برآں، میں اپنے ضمنی جدولاں مرتب کیتیاں، جس چ اسٹاک مچھی وغیرہ دی ممکنہ مقدار نوں چھویا گیا، جہڑا اس قدیم گرین لینڈ تے سپٹزبرگن وہیل فشری چ ہر لو ڈچ ہارپونیئر ولوں کھادا جاندا سی۔ پہلی تھاں تے، مکھن دی مقدار، تے ٹیکسل تے لیڈن پنیر کھادا جاندا اے، حیرت انگیز لگدا اے۔ میں اینوں، اگرچہ، اوہناں دی قدرتی طور تے چکنائی آلی فطرت نال منسوب کردا ہاں، جہڑا کہ اوہناں دے پیشے دی نوعیت دی وجہ توں، تے خاص طور تے اوہناں دے اوہناں ٹھنڈے قطبی سمندراں وچ، اوس ایسکیموکس ملک دے ساحلاں تے، اپنے کھیڈ دا پیچھا کرن نال، جتھے ملوث مقامی لوک ٹرین دے تیل دے بمپراں وچ اک دوجے نوں عہد کردے نیں، ہور وی چکنائی نال پیش کیتا جاندا اے۔

بیئر دی مقدار وی بہت وڈی اے، 10,800 بیرل۔ ہن، کیونکہ اوہ قطبی ماہی گیری صرف اس آب و ہوا دے مختصر موسم گرما وچ ہی مقدمہ چلائی جا سکدی سی، تاں جے ایہناں ڈچ وہیلمیناں وچوں اک دی پوری کروز، بشمول سپٹزبرگن سمندر تک تے اس توں مختصر سفر، تن مہینے توں زیادہ نئیں سی، کہو، تے 180 بادباناں دے ہر بیڑے وچ 30 آدمیاں دا حساب لاندے ہوئے، ساڈے کول 550، 400 لو ڈچ مین نیں؛ اس لیی، میں کہنا واں، ساڈے کول باراں ہفتیاں دے الاؤنس لئی، فی بندہ دو بیرل بیئر نیں، اس دے 550 اینکر جن دے منصفانہ تناسب نو چھڈ کے۔ ہن، چاہے ایہ جن تے بیئر ہارپونرز، جِنّے کہ کوئی اوہناں نوں سمجھ سکدا اے، کشتی دے سر تے کھڑے ہون لئی، تے اڈن آلیاں وہیلاں تے چنگا نشانہ بنان لئی صحیح قسم دے بندے سن؛ ایہہ کجھ ناقابلِ یقین لگے گا۔ فیر وی اوہناں نے اوہناں تے نشانہ بنایا، تے اوہناں نوں وی ماریا۔ لیکن ایہ بہوں دور شمال سی، یاد رکھو، جتھے بیئر آئین دے نال چنگی طراں اتفاق کردا اے; خط استوا تے، ساڈی جنوبی ماہی گیری چ، بیئر ہارپونیئر نو مستول دے سر تے نیندر تے اپنی کشتی چ شرابی بنان لئی موزوں ہووے گا; تے نانٹکٹ تے نیو بیڈ فورڈ نوں گمبھیر نقصان ہو سکدا اے۔

پر ہور نئیں؛ ایہ ظاہر کرن لئی کافی گلاں کیتیاں گئیاں نیں کہ دو یا تن صدیاں پہلے دے پرانے ڈچ وہیلرز ہائی لیور سن; تے ایہ کہ انگریزی وہیلرز نے اتنی بہترین مثال نو نظر انداز نئیں کیتا۔ کیونکہ، اوہ کہندے نیں، جدوں تُسی خالی جہاز تے کروز کردے او، جے تسی دنیا چوں کجھ وی چنگا نئیں لبھ سکدے، تے گھٹ توں گھٹ ایس چوں اک چنگا رات دا کھانا حاصل کرو۔ تے ایہ ڈیکنٹر خالی کر دیندا اے۔





باب 102. آرساسائڈز وچ اک بوئر۔

ہنڑ تیکر، سپرم وہیل دے بارے چ وضاحتی طور تے پیش آندے ہوئے، میں بنیادی طور تے اوہدے بیرونی پہلو دے عجائبات تے غور کیتا اے۔ یا کجھ اندرونی ساختی خصوصیات تے وکھرا تے تفصیل نال۔ پر اوہدے بارے چ اک وڈی تے مکمل سمجھ لئی، ہن مینوں اوہدے بٹن ہور کھولنا چاہیدا اے، تے اوہدے نلی دے نوکاں نوں کھولنا، اوہدے گارٹر کھولنا، تے اوہدیاں اندرلیاں ہڈیاں دے جوڑاں دے ہک تے اکھاں نوں ڈھیلا کرنا، اوہنوں اپنے الٹی میٹم چ تہاڈے سامنڑے رکھنا؛ کہنا اے، اس دے غیر مشروط کنکال وچ۔

پر ہن کیویں اسماعیل؟ ایہہ کیویں ہویا، کہ تسی، ماہی گیری چ اک محض کشتی باز، وہیل دے زیر زمین حصیاں دے بارے چ کجھ جانن دا دکھاوا کردے او؟ کیہ تہاڈے کیپسٹن تے سوار پانڈت سٹب نے سیٹیشیا دی اناٹومی تے لیکچر دتے؛ تے ونڈلاس دی مدد نال، نمائش لئی اک نمونہ پسلی نوں پھڑو؟ اپنے آپ نوں سمجھاؤ، اسماعیل۔ کی تسی جانچ پڑتال لئی اپنے ڈیک تے اک بھری ہوئی وہیل نوں اتار سکدے او، جویں کہ اک باورچی اک روسٹ سور نوں پکاندا اے؟ یقیناً نئیں۔ تُسی ہن تیکر اک حقیقی گواہ رہے او، اسماعیل؛ پر ایس گل دا خیال رکھو کہ تُسی یونس دے استحقاق نوں اکلے کیویں پھڑدے او؛ جوسٹاں تے بیماں تے گل بات کرن دا استحقاق; رافٹرز، رج پول، سلیپرز، تے انڈر پننگز، جو لیویتھن دا فریم ورک بناندے نیں; تے اوہدے آندراں چ ٹیلو واٹ، ڈیری روم، بٹر تے چیزریز دی طرح۔

میں اقرار کردا ہاں، کہ یونس دے بعد توں، بہت گھٹ وہیل مین بالغ وہیل دی جلد دے تھلے بہت دور گھس چکے نیں; اس دے باوجود، مینوں اس نوں نکی جہی طور تے وکھ کرن دا موقع ملیا اے۔ اک جہاز وچ جہدے نال میرا تعلق سی، اک نکے بچے سپرم وہیل نوں اک وار اپنے پوک یا بیگ لئی، ہارپون دے بارباں لئی میان بنان لئی، تے برچھیاں دے سراں لئی ڈیک تے جسمانی طور تے لہرا دتا گیا سی۔ کی تہاڈا خیال اے کہ میں اپنی کشتی دے ہیچٹ تے جیک نائف دے استعمال کیتے بغیر، تے مہر توڑے تے اس نوجوان بچے دے تمام مواد نو پڑھے بغیر، اوہ موقع چھڈ دتا؟

تے جتھوں تیکر لیویتھن دیاں ہڈیاں دے بارے چ میرا عین مطابق علم اے، اوہناں دی وشال، مکمل نشوونما چ، اس نایاب علم لئی میں اپنے مرحوم شاہی دوست ٹرانکو دا مقروض آں، ٹرانکو دا بادشاہ، جو آرساکائڈز چوں اک اے۔ ٹرانکی چ ہون دے لیی، سال پہلے، جدوں میں الجزائر دے تجارتی جہاز ڈی نال منسلک سی، مینوں ٹرانکی دے مالک دے نال، پوپیلا چ اس دے ریٹائرڈ پام ولا چ، ارساکیڈین چھٹیاں دا کجھ حصہ گزارن دی دعوت دتی گئی سی؛ اک سمندر دے کنارے دی گلین جہڑی ساڈے ملاحاں دے راجگڑھ نوں بانس ٹاؤن کہندے سن، اس توں بہت دور نئیں سی۔

ہور کئی چنگیاں خوبیاں دے نال نال، میرا شاہی دوست ٹرانکو، وحشی ورٹو دے سارے معاملیاں لئی اک متقی محبت نال نوازیا گیا سی، پپیلا وچ اوہ نایاب شیواں اکٹھے کر لیا سی جہڑے اپنے لوکاں وچوں زیادہ ہوشیار ایجاد کر سکدے سن; شاندار آلات دی بنیادی طور تے کھدی ہوئی لکڑی، چھینی آلے گولے، جڑے ہوئے نیزے، مہنگے پیڈل، خوشبودار کینو؛ تے ایہ سب کجھ قدرتی عجوبیاں وچ ونڈیا گیا، حیرت انگیز، خراج تحسین پیش کرن آلی لہراں نے اوہدے ساحلاں تے سٹیا سی۔

ایہناں وچوں سب توں وڈی اک عظیم سپرم وہیل سی، جہڑی غیر معمولی لمی طوفان دے بعد، مردہ تے پھسے ہوئے پایا گیا سی، اس دا سر اک کوکو نٹ دے درخت نال لگیا ہویا سی، جس دے کھنبھاں ورگے، ٹفٹڈ ڈروپنگ اس دے ہرے بھرے جیٹ لگدے سن۔ جدوں آخر کار اس وسیع جسم نو اس دے ڈوہنگے لپیٹاں توں محروم کر دتا گیا سی، تے ہڈیاں دھوپ چ دھوڑ نال سک جاندیاں سن، تے فیر اس ڈھانچے نو احتیاط نال پوپیلا گلین تک پہنچایا گیا سی، جتھے ہنڑ اس نوں پناہ دتی گئی سی۔

پسلیاں نوں ٹرافیاں نال لٹکایا گیا سی; ریڑھ دی ہڈی نوں عجیب و غریب ہیروگلیفاں چ آرساکیڈین اینالز نال تراشیا گیا سی; کھوپڑی وچ، پادریاں نے اک غیر بجھی خوشبودار شعلہ رکھیا، تاکہ صوفیانہ سر نے فیر اپنا بخارات آلا ٹونٹی باہر بھیج دتی؛ جدوں کہ، اک ٹہنی توں لٹکیا ہویا، خوفناک تھلواں جبڑا سارے عقیدت منداں تے کمبدا سی، جویں کہ والاں نال لٹکدی ہوئی تلوار جس نے ڈیموکل نوں ڈرایا سی۔

ایہہ اک شاندار نظارہ سی۔ لکڑی آئسی گلین دے کائی دی طرح ہری سی; درخت اُچے تے تکبر نال کھلوتے سن، اپنے زندہ رس نوں محسوس کر رئے سی; تھلے دی محنتی زمین اک بنکر دے کرگھا دی طرح سی، جس تے اک خوبصورت قالین سی، جس دے زمینی بیل دے ٹنڈاں نے وارپ تے ووف بنایا سی، تے زندہ پُھلاں نے شکلاں بنائیاں سن۔ سارے درخت، اوہناں دیاں ساریاں بھریاں ہوئیاں شاخاں دے نال؛ سارے جھاڑیاں، تے فرن، تے گھاس; سنیہا لے جان آلی ہوا; ایہ سارے مسلسل متحرک سن۔ پتیاں دے لیساں دے ذریعے، عظیم سورج اک اڈن آلی شٹل لگدا سی جہڑی انتھک ہریالی نوں پروندی سی۔ اوہ، مصروف بنکر! ان ڈٹھا بنکر!—رک جاؤ!—اک شبد!—کتھے وگدا اے کپڑا؟ ایہہ کیہڑا محل سج سکدا اے؟ ایہہ ساریاں لگاتار محنتاں کیوں؟ بولو، بنکر!—اپنا ہتھ روک!—پر اک ہی شبد تیرے نال! نئیں - شٹل اڈدی اے - اعداد و شمار کرگھا دے اگے توں تیردے نیں; تازہ نسدا ہویا قالین ہمیشہ لئی دور ہو جاندا اے۔ بنکر-دیوتا، اوہ بندا اے؛ تے اوس بنائی نال اوہ بہرا ہو جاندا اے، کہ اوہ کوئی فانی آواز نئیں سندا؛ تے اس گنگناہٹ نال، اسی وی، جہڑے کرگھا تے ویکھدے آں، بہرے ہو جاندے آں؛ تے صرف جدوں اسی اس توں بچ جاواں گے تے اسی ہزاراں آوازاں سناں گے جہڑیاں اس دے ذریعے گل کردیاں نیں۔ کیونکہ فیر وی ایہ تمام مادی کارخانیاں وچ اے۔ اوہ بولے ہوئے بول جو اڈدے تکلیاں وچ سنائی نہیں دیندے؛ اوہی الفاظ دیواراں دے بغیر واضح طور تے سنے جاندے نیں، کھلے ہوئے کیسمنٹاں توں پھٹدے ہوئے۔ اس طرح ولناں دا پتہ لگ گیا اے۔ آہ، فانی! فیر، محتاط رہو; کیونکہ اس طرح، عظیم دنیا دے کرگھا دے اس سارے دھماکے وچ، تیریاں لطیف ترین سوچاں دوروں سنیاں جا سکدیاں ہن۔

ہن، اس ارساکیڈین لکڑی دے ہرے، بے چین کرگھا دے وچکار، عظیم، چٹا، پوجیا جان آلا ڈھانچہ لیٹیا ہویا سی - اک بہوں وڈا ویہل! فیر وی، جدوں ہمیشہ بنیا ہویا ہرا بھرا وارپ تے ووف آپس وچ رلدے ملدے سن تے اوہدے آلے دوالے گنگنا رہے سن، تے طاقتور ویہل اک چالاکی والا بنکر لگدا سی; اپنے آپ نو بیلاں دے نال پرویا ہویا سی; ہر مہینے ہرا بھرا، تازہ سبزی فرض کردے ہوئے; پر اوہ آپ اک ڈھانچہ اے۔ زندگی جوڑی موت؛ موت نے زندگی نوں گھیر لیا; گھور دیوتا نے جوانی دی زندگی دے نال بیوی کیتی، تے اس نوں گھنگریالے سراں آلی شان پیدا کیتی۔

ہن، جدوں میں شاہی ٹرانکو دے نال اس حیرت انگیز وہیل دا دورہ کیتا، تے کھوپڑی نو اک قربان گاہ ویکھیا، تے مصنوعی دھواں اوتھوں چڑھ رہیا سی جتھے اصلی جیٹ جاری کیتا گیا سی، میں حیران ہویا کہ بادشاہ نوں اک چیپل نو ورٹو دی شے سمجھنا چاہیدا اے۔ اوہ ہسیا۔ پر میں ایس گل تے ہور حیران ہویا کہ پادریاں نوں قسم کھادی کہ اوہدا دھواں دار جیٹ اصلی سی۔ میں ایس ڈھانچے توں پہلاں ایدھر اودھر ٹُر پیا - بیلاں نو اک پاسے کر دتا - پسلیاں نو توڑ دتا - تے ارساکیڈین ٹوئن دی گیند دے نال، گھومدا رہیا، اس دے کئی گھماؤدار، سایہ دار کالونیڈز تے آربراں دے وچکار لمبے عرصے تک گھمدا رہیا۔ پر چھیتی ای میری لائن ختم ہو گئی; تے اوہدے پچھے پچھے، میں اوس افتتاح توں باہر نکلیا جتھے میں داخل ہویا سی۔ میں اندر کوئی جاندار نئیں ویکھیا; ہڈیاں دے علاوہ کجھ وی نئیں سی۔

مینوں اک ہری پیمائش دی چھڑی کٹ کے، میں اک واری فیر ڈھانچے دے اندر ڈبکی لا دتی۔ کھوپڑی وچ تیر دے سلٹ توں، پادریاں نے مینوں آخری پسلی دی اونچائی لیندیاں ویکھیا، "ہن کیویں!" اوہناں نے چیخیا; "کیہ توں ایس نوں ساڈے خدا دی پیمائش نئیں کر سکدا! ایہ ساڈے لئی اے۔" "ہاں، پادری—اچھا، فیر تُسی اوہنوں کدوں تیکر بناؤ گے؟" پر ایتھے اوہناں وچکار پیراں تے انچاں دے بارے وچ اک سخت مقابلہ شروع ہو گیا؛ اوہناں نے اپنیاں یارڈ-سٹکس نال اک دوجے دے سکونس نوں توڑ دتا—وڈی کھوپڑی گونج رہی سی—تے اس خوش قسمت موقعے نوں پھڑدیاں ہوئیاں، میں چھیتی نال اپنیاں پیمائشاں دا نتیجہ کڈھیا۔

ایہ پیمائشاں میں ہن تہاڈے سامنڑے رکھن دی تجویز پیش کرنا واں۔ لیکن پہلے، ایہ ریکارڈ کیتا جائے، کہ، اس معاملے وچ، میں کوئی وی خیالی پیمائش کرن لئی آزاد نئیں آں۔ کیونکہ کنکال اتھارٹیز موجود نیں جنہاں دا تسی حوالہ دے سکدے او، میری درستگی دی جانچ کرن لئی۔ اوہ مینوں دسدے نیں کہ ہل، انگلینڈ چ اک لیویتھنک میوزیم اے، جہڑا کہ اوس ملک دی وہیلنگ بندرگاہاں چوں اک اے، جتھے اوہناں کول فن بیک تے دوجیاں وہیلاں دے کجھ چنگے نمونے موجود ہن۔ ایسے طرحاں، میں سُنیا اے کہ نیو ہیمپشائر دے مانچسٹر دے میوزیم وچ، اوہناں کول اوہ چیز اے جہنوں مالک "امریکہ وچ گرین لینڈ یا دریائے وہیل دا واحد کامل نمونہ" کہندے ہن۔ مزید ایہ کہ، یارکشائر، انگلینڈ دی اک تھاں تے، برٹن کانسٹیبل دے ناں نال، اک خاص سر کلفورڈ کانسٹیبل دے کول اک سپرم وہیل دا کنکال اے، لیکن اعتدال پسند سائز دا، میرے دوست کنگ ٹرانکو دی مکمل شدت دا نئیں۔

دوناں صورتاں وچ، پھسے ہوئے وہیلاں جنہاں نال ایہ دو کنکال تعلق رکھدے سن، اصل وچ اوہناں دے مالکان ولوں اسی طرح دی بنیاداں تے دعوی کیتا گیا سی۔ کنگ ٹرانکو نے اپنا قبضہ کر لیا کیونکہ اوہ چاہندا سی; تے سر کلفورڈ، کیونکہ اوہ انہاں حصیاں دے سیگنوریاں دا مالک سی۔ سر کلفورڈ دی وہیل نوں پورے طور تے بیان کیتا گیا اے; تاں جے، درازاں دے اک وڈے صندوق دی طرح، تسی اوہنوں کھول تے بند کر سکو، اوہدے سارے ہڈیاں دے جوفاں وچ - اوہدیاں پسلیاں نوں اک وشال پنکھے دی طرح کھلار دیو - تے سارا دن اوہدے تھلویں جبڑے تے جھول سکو۔ اس دے کجھ ٹریپ-دروازے تے شٹراں تے تالے لگے جانے نیں; تے اک فٹ مین مستقبل دے زائرین نو اپنے نال چابیاں دا اک گروہ دے نال وکھائے گا۔ سر کلفورڈ ریڑھ دی ہڈی چ سرگوشی کرن آلی گیلری تے جھانکن لئی دو پینس وصول کرن دا سوچدا اے۔ اپنے دماغ دے کھوکھلے وچ گونج سنن لئی تھری پینس؛ تے اوہدے متھے توں بے مثال نظارے لئی چھ پیسے۔

ڈھانچے دے طول و عرض جنہاں نو میں ہن مقرر کرن لئی اگے ودھاں گا اوہ میری سجی بانہہ توں لفظ بہ لفظ کاپی کیتیاں گئیاں نیں، جتھے میں اوہناں نو ٹیٹو کروایا سی; جیویں کہ اوس ویلے میرے جنگلی آوارہ گردی چ، اینج دے قیمتی اعداد و شمار نو محفوظ رکھن دا کوئی ہور محفوظ طریقہ نئیں سی۔ لیکن جدوں میں جگہ لئی بھریا ہویا سی، تے اپنے جسم دے دوجے حصے اک نظم لئی خالی صفحہ رہن دی خواہش کردا سی جہنوں میں اوس ویلے رچ رہیا سی—کم از کم، کیہڑے غیر ٹیٹو والے حصے رہ سکدے نیں—میں اپنے آپ نوں عجیب انچاں نال پریشان نئیں کیتا؛ نہ ای، درحقیقت، انچ بالکل وی وہیل دی اک سہج پیمائش وچ داخل ہونا چاہیدا اے۔





باب 103. وہیل دے کنکال دی پیمائش۔

سب توں پہلاں، میں تہاڈے سامنڑے اک خاص، سادہ بیان رکھنا چاہنا واں، جہڑا اس لیویتھن دے زندہ حصے نو چھوندا اے، جس دے کنکال نو اسی مختصر طور تے نمائش کرن گے۔ ایتھے انج دا بیان کم آ سکدا اے۔

اک محتاط حساب کتاب دے مطابق جہڑا میں کیتا اے، تے جہنوں میں جزوی طور تے کیپٹن سکورسبی دے اندازے تے مبنی کردا ہاں، سٹھ فٹ لمبائی دی سب توں وڈی گرین لینڈ وہیل لئی ستر ٹن؛ میرے محتاط حساب دے مطابق، میں کہنا واں، اک سپرم وہیل دی سب توں وڈی شدت، پچاسی توں نوے فٹ دے وچکار، تے اپنے پورے دائرے وچ چالھی فٹ توں گھٹ کجھ، اجیہی وہیل دا وزن گھٹ توں گھٹ نوے ٹن ہووے گا؛ تاکہ، تیراں بندیاں نو اک ٹن دے حساب نال، اوہ اک ہزار سو باشندیاں دے پورے پنڈ دی مجموعی آبادی توں کافی زیادہ ہو جائے گا۔

فیر کی تُسی نئیں سوچدے کہ دماغ، جوئے آلے ڈنگراں دی طرح، ایس لیویتھن نوں لایا جانا چاہیدا اے، تاں جے اوہ کسے وی زمیندار دی کلپنا دے سامنڑے ہل سکے؟

پہلے ای مختلف طریقیاں نال تہاڈے سامنڑے اوہدی کھوپڑی، ٹونٹی دا سوراخ، جبڑا، دند، پونچھ، متھے، چنجاں تے ہور کئی حصیاں نوں پیش کر کے، ہنڑ میں صرف ایہ دساں گا کہ اوہدیاں بے روک ٹوک ہڈیاں دے عام حصے وچ کی سب توں دلچسپ اے۔ لیکن جیویں کہ بہوں وڈی کھوپڑی کنکال دی پوری حد دے بہت وڈے حصے نو اپناندی اے; کیونکہ ایہ سب توں پیچیدہ حصہ اے; تے کیونکہ اس باب وچ اس دے بارے وچ کجھ وی نئیں دہرایا جانا اے، تہانوں اس نوں اپنے دماغ وچ، یا اپنی بانہہ دے تھلے، لے جان وچ ناکام نئیں ہونا چاہیدا، جدوں اسی اگے ودھدے آں، نئیں تے تہانوں اوس عام ڈھانچے دا مکمل اندازہ نئیں ملے گا جہنوں اسی ویکھن آلے آں۔

لمبائی چ، ٹرانکی چ سپرم وہیل دا ڈھانچہ ستر بائی فٹ سی; تاں جے جدوں اوہ زندگی چ پوری طرحاں سرمایہ کاری تے ودھ گیا، تے اوہ نوے فٹ لمبا ہونا چاہیدا اے; کیونکہ وہیل وچ، کنکال زندہ جسم دے مقابلے وچ لمبائی وچ تقریباً پنجواں حصہ کھو دیندا اے۔ اس بائیس فٹ چوں، اوہدی کھوپڑی تے جبڑا تقریباً ویہہ فٹ تے مشتمل سی، جس نال کوئی پنجاہ فٹ سادہ کمر دی ہڈی رہ گئی۔ اس ریڑھ دی ہڈی نال جڑی ہوئی، اس دی لمبائی دے اک تہائی توں وی گھٹ، پسلیاں دی اک طاقتور سرکلر ٹوکری سی جہڑی کدی اس دے حیاتیاتی حصے نو گھیر لیندی سی۔

میرے لیی ایہ وسیع ہاتھی داند دی پسلیاں آلی سینہ، لمی، غیر راحت شدہ ریڑھ دی ہڈی دے نال، اک سدھی لکیر چ اس توں بہت دور تک پھیلی ہوئی، اک عظیم جہاز دے پتوار نال تھوڑا جیہا نئیں ملدا سی جہڑا اسٹاک تے نواں رکھیا گیا سی، جدوں اس دی صرف کوئی ویہہ ننگی کمان دیاں پسلیاں داخل کیتیاں جاندیاں نیں، تے کیل دوجی صورت چ، ویلے لئی، اک لمی، لکڑی، منقطع ہوندی اے۔

پسلیاں اک پاسے دس سن۔ پہلا، گردن توں شروع ہون آلا، تقریباً چھ فٹ لمبا سی; دوجا، تیجا تے چوتھا ہر اک اک توں بعد اک لمبا سی، جدوں تک کہ تسی پنجویں دے عروج تے نئیں پہنچ گئے، یا وچکارلیاں پسلیاں چوں اک، جہدا ناپ اٹھ فٹ تے کجھ انچ سی۔ اس حصے توں، باقی پسلیاں گھٹ گئیاں، دسویں تے آخری صرف پنج فٹ تے کجھ انچ تک پھیلی ہوئی۔ عام موٹائی چ، اوہ سارے اپنی لمبائی دے نال اک مناسب مطابقت رکھدے سن۔ وچکارلیاں پسلیاں سب توں زیادہ محراب آلیاں سن۔ کجھ آرساسائڈز چ اوہ بیماں دے لیی استعمال ہوندے نیں جہدے تے نکے ندیاں تے فٹ پاتھ دے پل بنائے جاندے نیں۔

ایہناں پسلیاں تے غور کردیاں ہویاں، میں ایس حالات نال نویں سرے توں متاثر ہون توں بغیر نئیں رہ سکیا، جہڑا ایس کتاب وچ اینی مختلف انداز وچ دہرایا گیا اے، کہ وہیل دا ڈھانچہ کسی وی صورت وچ اوہدے سرمایہ کاری شدہ شکل دا سڑنا نئیں اے۔ ٹرانکی پسلیاں چوں سب توں وڈی، وچکارلیاں چوں اک، مچھی دے اوس حصے تے قبضہ کردی اے جہڑا، زندگی چ، ڈونگھائی چ سب توں وڈا اے۔ ہن، اس خاص وہیل دے سرمایہ کاری شدہ جسم دی سب توں وڈی گہرائی گھٹ توں گھٹ سولاں فٹ ہونی چاہیدی اے; جد کہ، متعلقہ پسلی دی پیمائش اٹھ فٹ توں تھوڑی زیادہ سی۔ اس لیی اس پسلی نے صرف اس حصے دی زندہ شدت دے حقیقی تصور دا ادھا حصہ ای پہنچایا۔ ایس توں علاوہ، کسی طرح، جتھے میں ہن صرف اک ننگی ریڑھ دی ہڈی ویکھدا سی، اوہ سب کجھ جہڑا کدی گوشت، پٹھیاں، خون تے آندراں چ ٹن وادھو بلک دے نال لپٹیا ہویا سی۔ ہُن وی، کافی چنجاں لئی، میں ایتھے صرف کجھ بے ترتیب جوڑ ویکھے؛ تے بھاری تے شاندار، لیکن ہڈیاں توں پاک فلوکاں دی تھاں، اک بالکل خالی!

فیر، میں سوچیا کہ ڈرپوک بے سفر انسان لئی اس پرامن جنگل وچ پھیلے ہوئے اپنے مردہ کمزور کنکال نو محض ویکھ کے، اس حیرت انگیز وہیل نو صحیح طریقے نال سمجھن دی کوشش کرنا کِنا فضول تے احمقانہ اے۔ نئیں۔ صرف تیز ترین خطریاں دے دل چ؛ صرف اودوں جدوں اوہدے غصے دے جھٹکے دے اندر ہووے؛ صرف ڈوہنگے بے حد سمندر تے، مکمل طور تے سرمایہ کاری شدہ وہیل نو واقعی تے زندہ طور تے پایا جا سکدا اے۔

پر ریڑھ دی ہڈی۔ اس دے لی، اسی اس تے غور کرن دا بہترین طریقہ ایہ اے کہ، اک کرین دے نال، اس دی ہڈیاں نو سرے تے اُچا ڈھیر کرنا۔ کوئی تیز کارروائی نئیں۔ پر ہن ایہہ ہو گیا اے، ایہہ پومپے دے ستون ورگا لگدا اے۔

کل چالی تے اوڈ ریڑھ دی ہڈی اے، جہڑی کنکال چ اک دوجے نال بند نئیں ہوندی۔ اوہ زیادہ تر گوتھک سپائر تے عظیم نوبڈ بلاکس دی طرح پئے ہوندے نیں، جہڑا بھاری چنائی دے ٹھوس کورسز بناندا اے۔ سب توں وڈا، اک وچکارلا، چوڑائی چ تن فٹ توں گھٹ اے، تے ڈونگھائی چ چار توں ودھ اے۔ سب توں نکا، جتھے ریڑھ دی ہڈی پونچھ چ ڈھل جاندی اے، چوڑائی چ صرف دو انچ اے، تے کجھ چٹی بلئرڈ گیند دی طرح لگدا اے۔ مینوں دسیا گیا سی کہ اجے وی چھوٹے-چھوٹے ہن، پر اوہ کجھ نکے-نکے آدم خور ارچناں، پادری دے بچے، گواچ گئے سن، جنہاں نے اوہناں نوں سنگمرمر کھیڈن لئی چوری کر لیا سی۔ اس طرح اسی ویکھنے آں کہ کس طرح سب توں وڈی جانداراں دی ریڑھ دی ہڈی وی آخر کار سادہ بچیاں دے کھیڈاں چ بدل جاندی اے۔





باب 104. جیواشم وہیل۔

اپنے زبردست ڈھیر توں وہیل اک بہوں سہج تھیم پیش کردی اے جہدے تے وڈا کرنا، ودھانا تے عام طور تے ودھانا اے۔ کی تُسی کر سکدے او، تُسی اوہنوں دبا نئیں سکدے۔ چنگے حقاں دے مطابق اوہ صرف شاہی فولیو وچ ہی سلوک کیتا جانا چاہیدا اے۔ گھماؤ توں دم تک اوہدے فرلانگ، تے کمر دے بارے چ اوہ گز ناپدا اے، دوبارہ نئیں دسنا؛ صرف اس دیاں آندراں دے وشال گھماؤ دے بارے وچّ سوچو، جتھے اوہ اس وچّ وڈے کیبلاں اتے ہازراں وانگ پئے ہن جو اک لائن-آف-بیٹل-شپ دے زیر زمین یا لوپ-ڈیک وچّ لپٹے ہوئے ہن۔

چونکہ میں اس لیویتھن نو سنبھالن دا وعدہ کیتا اے، اس لیی میں اپنے آپ نو اس کاروبار وچ ہر چیز جانن آلے تھکاوٹ دی منظوری دینا چاہنا واں; اپنے خون دے معمولی ترین سیمینل جراثیماں نو نظر انداز نئیں کردے ہوئے، تے اس نو اپنی آندراں دے انتہائی کنڈلی تک گھماندے ہوئے۔ اس دے زیادہ تر موجودہ رہائش گاہ تے جسمانی انوکھیاں گلاں وچ اس نوں پہلے ای بیان کر چکے نیں، ہن اس نوں آثار قدیمہ، جیواشم تے اینٹی ڈیلوویئن نقطہ نظر وچ وڈا کرنا باقی اے۔ لیویتھن دے علاوہ کسی وی دوجی مخلوق تے لاگو کیتا جاندا اے - اک چیونٹی یا پسو تے - اس طرح دے پورٹلی اصطلاحات نو غیر جائز طور تے شاندار سمجھیا جا سکدا اے۔ لیکن جدوں لیویتھن متن ہوندا اے، تے کیس بدل جاندا اے۔ میں لغت دے سب توں بھاری لفظاں دے تحت اس عہدے تے ٹُٹ بھج کرن چ ناکام آں۔ تے ایتھے ایہ آکھیا جائے، کہ جدوں وی ایہناں مقالیاں دے دوران کسے نال مشورہ کرنا سوکھا ہویا اے، میں ہمیشہ جانسن دا اک بہت وڈا کوارٹو ایڈیشن استعمال کیتا اے، جہڑا اس مقصد لئی واضح طور تے خریدیا گیا سی؛ کیونکہ اس مشہور لغت نگار دی غیر معمولی ذاتی مقدار نے اس نوں میرے ورگے وہیل مصنف دے ذریعہ استعمال کرن لئی اک لغت مرتب کرن لئی زیادہ موزوں کیتا۔

کوئی اکثر لکھاریاں دے بارے چ سُندا اے جہڑے اپنے مضمون دے نال اُٹھدے تے ودھدے نیں، حالانکہ ایہ اک عام گل لگدی اے۔ فیر، میرے نال، اس لیویتھن دی لکھت کیویں؟ لاشعوری طور تے میری چیروگرافی پلاکارڈ کیپٹلاں وچ پھیل جاندی اے۔ مینوں اک کنڈور دی کھنبھ دے دیو! مینوں اک انکسٹینڈ لئی ویسوویئس دا گڑھا دے دیو! سجنو میریاں باہواں پھڑو! کیونکہ اس لیویتھن دے بارے چ میرے خیالات نو لکھن دے محض عمل چ، اوہ مینوں تھکا دیندے نیں، تے مینوں اپنی جھاڑو دی جامعیت نال بے ہوش کر دیندے نیں، جویں کہ علوم دے پورے دائرے، تے وہیلاں، تے انساناں، تے ماسٹوڈونز دیاں ساریاں نسلاں، ماضی، حال، تے آن آلے، پوری زمین تے سلطنت دے سارے گھمدے ہوئے پینورما دے نال، پوری کائنات چ شامل نئیں، اس دے مضافاتی علاقیاں نو چھڈ کے۔ اجیہا، تے اینا وڈا کرن آلا، اک وڈے تے لبرل تھیم دی خوبی اے! اسی ایہدے وڈے حصے تک ودھاندے آں۔ اک زبردست کتاب تیار کرن دے لی، تہانوں اک زبردست موضوع دا انتخاب کرنا پوے گا۔ پسو تے کدی وی کوئی وڈی تے پائیدار جلد نئیں لکھی جا سکدی، حالانکہ بہت سارے لوک نیں جنہاں نے ایہنوں آزمایا اے۔

جیواشم وہیلاں دے موضوع تے داخل ہون توں پہلاں، میں اک ماہر جیالوجسٹ دے طور تے اپنی اسناد پیش کردا ہاں، ایہ کہہ کے کہ اپنے متفرق ویلے چ میں اک پتھر دا معمار رہیا واں، تے نالے کھائیاں، نہراں تے کھوہاں، شراب دے والٹ، سیلر تے ہر قسم دے حوضاں دا اک وڈا کھودن آلا وی رہیا واں۔ اسی طرح، ابتدائی طور تے، میں قاری نو یاد دلانا چاہنا واں، کہ جد کہ پہلے ارضیاتی پرتاں وچ راکشساں دے جیواشم پائے گئے نیں جہڑے ہن تقریباً مکمل طور تے ختم ہو چکے نیں; ترتیری فارمیشناں وچ دریافت ہون آلے بعد دے اوشیش، اینٹی کرونیکل مخلوق دے وچکار جڑن آلے، یا کسی وی شرح نال روکے ہوئے روابط لگدے نیں، تے اوہناں لوکاں دے وچکار جنہاں دی دور دراز دی اولاد دے بارے وچ آکھیا جاندا اے کہ اوہ صندوق وچ داخل ہوئے نیں; ہنڑ تیکر دریافت کیتے گئے سارے جیواشم وہیل ترتیری دور نال تعلق رکھدے نیں، جہڑا کہ سطحی تشکیل توں پہلے دا آخری دور اے۔ تے اگرچہ انہاں چوں کوئی وی موجودہ ویلے دی کسی وی جانی پہچانی پرجاتی دا قطعی طور تے جواب نئیں دیندا، اوہ فیر وی عام لحاظ نال انہاں دے نال کافی ملدے جلدے نیں، تاکہ سیٹیشین جیواشم دے طور تے انہاں دی درجہ بندی دا جواز پیش کیتا جا سکے۔

پری-ایڈمائٹ وہیلاں دے ٹٹے ہوئے جیواشم، اوہناں دیاں ہڈیاں تے کنکالاں دے ٹکڑے، پچھلے تیس سالاں دے اندر، مختلف وقفیاں تے، الپس دے اڈے تے، لومبارڈی، فرانس، انگلینڈ، سکاٹ لینڈ، تے ریاستاں لوزیانا، مسیسپی تے الاباما چ پائے گئے ہن۔ ایسی باقیات چوں زیادہ متجسس چ اک کھوپڑی دا حصہ اے، جہنوں سال 1779 چ پیرس دے رو ڈوفین چ دفن کیتا گیا سی، اک چھوٹی گلی جہڑی تقریباً براہ راست ٹائلریز دے محل تے کھلدی اے۔ تے نپولین دے ویلے چ اینٹورپ دیاں وڈیاں گودیاں دی کھدائی چ دفن کیتیاں گئیاں ہڈیاں۔ کیویر نے ایہناں ٹکڑیاں نو کجھ بالکل نامعلوم لیویتھن پرجاتیاں نال تعلق رکھن دا اعلان کیتا۔

پر ہنڑ تیکر سارے سیٹیشین اوشیشاں چوں سب توں حیرت انگیز اک معدوم راکشس دا تقریباً مکمل وسیع کنکال سی، جہڑا سال 1842 چ، الاباما چ جج کریگ دی باگ تے پایا گیا سی۔ آسے پاسے دے خوفزدہ غلاماں نے اینوں ڈگے ہوئے فرشتیاں چوں اک دی ہڈی لئی لیا۔ الاباما دے ڈاکٹراں نے اس نوں اک بہت وڈا رینگنے والا جانور قرار دتا، تے اس نوں باسیلوسورس دا ناں دتا۔ لیکن اس دے کجھ نمونے دی ہڈیاں نو سمندر پار اوون، انگریزی اناٹومیسٹ دے کول لے جایا گیا، تے پتہ چلیا کہ ایہ مبینہ رینگنے آلا جانور اک وہیل سی، اگرچہ اک رخصت ہون آلی پرجاتی دا سی۔ اس حقیقت دی اک اہم مثال، اس کتاب وچ بار بار دہرایا گیا اے، کہ وہیل دا کنکال اس دے مکمل طور تے سرمایہ کاری شدہ جسم دی شکل دے بارے وچ بہت گھٹ اشارہ فراہم کردا اے۔ اس لیی اوون نے راکشس زیگلوڈن دا ناں دوبارہ رکھ دتا; تے لندن جیولوجیکل سوسائٹی دے سامنڑے پڑھے گئے اپنے مقالے وچ، اس نے اس نو، مادہ وچ، سب توں غیر معمولی مخلوق وچوں اک قرار دتا جہنوں گلوب دے تغیرات نے وجود توں مٹا دتا اے۔

جدوں میں ایہناں طاقتور لیویتھن دے کنکالاں، کھوپڑیاں، دند، جبڑے، پسلیاں تے ریڑھ دی ہڈی دے وچکار کھلوتا ہاں، جہڑے سمندری راکشساں دی موجودہ نسلاں نال جزوی طور تے مشابہت رکھدے نیں; لیکن نال ای دوجے پاسے فنا ہون آلے اینٹی کرونیکل لیویتھنز، اوہناں دے ناقابل حساب سینئر دے نال اسی طرح دی وابستگی رکھدے ہوئے; میں، اک سیلاب دے ذریعے، اس حیرت انگیز دور وچ واپس لے جایا گیا ہاں، اس توں پہلاں کہ وقت خود شروع ہو گیا اے؛ کیونکہ ویلا انسان توں شروع ہویا۔ ایتھے زحل دی سرمئی افراتفری میرے اتے لڑھکدی اے، تے میں اوہناں قطبی ہمیشگیاں وچ دھندلیاں، کمبدی ہوئی جھلکیاں حاصل کردا ہاں؛ جدوں برف دے پچر آلے گڑھاں نے ہن اشنکٹبندیی علاقےآں تے سختی نال دباؤ پایا; تے ایس دنیا دے سارے 25,000 میل دے گھیر وچ، اک وی رہن جوگا ہتھ دی چوڑائی دی زمین نظر نئیں آ رہی سی۔ فیر ساری دنیا وہیل دی سی; تے، تخلیق دا بادشاہ، اس نے اینڈیز تے ہمالہ دی موجودہ لکیراں دے نال اپنی جاگ چھڈ دتی۔ لیویتھن ورگا شجرہ کون وکھا سکدا اے؟ اخاب دے ہارپون نے فرعون دے ہارپون نالوں وی پرانا خون بہایا سی۔ میتھوسلا اک سکول دا منڈا لگدا اے۔ میں شیم نال ہتھ ملاون لئی آلے دوالے ویکھدا ہاں۔ میں وہیل دے ناقابل بیان دہشت گردی دے اس اینٹیموسیک، غیر ماخذ وجود توں خوفزدہ آں، جہڑا کہ ہر ویلے توں پہلاں ہویا اے، سارے انسانی زمانے ختم ہون دے بعد ضروری اے۔

لیکن صرف اس لیویتھن نے فطرت دے دقیانوسی پلیٹاں وچ اپنے پری ایڈمائٹ نشانات نئیں چھڈے، تے چونے دے پتھر تے مارل وچ اپنا قدیم مجسمہ وصیت کیتا اے; لیکن مصری تختیاں تے، جنہاں دی قدیمیت اوہناں لئی تقریباً جیواشم دے کردار دا دعوی کردی اے، سانوں اوہدے چنج دا غیر مشکوک پرنٹ ملدا اے۔ ڈینڈرہ دے عظیم مندر دے اک اپارٹمنٹ وچ، کوئی پنجاہ سال پہلے، گرینائٹ دی چھت تے اک مجسمہ سازی تے پینٹ کیتا ہویا پلینسفیر دریافت کیتا گیا سی، جس وچ سینٹور، گرفن تے ڈالفن دی بھرمار سی، جو جدید لوکاں دے آسمانی گلوب تے عجیب و غریب شخصیتاں نال ملدے جلدے سن۔ اوہناں دے وچکار گلائڈنگ کردے ہوئے، پرانا لیویتھن پرانے وانگوں تیردا سی; سلیمان دے جھولے توں صدیاں پہلاں، اس پلانسفیر وچ تیراکی کر رئے سی۔

نہ ای وہیل دی قدیمیت دی اک ہور عجیب تصدیق نوں چھڈنا چاہیدا اے، اس دی اپنی ہڈی آلی پوسٹ ڈیلوویئن حقیقت وچ، جویں کہ قابل احترام جان لیو، پرانے باربری مسافر نے طے کیتا اے۔

"سمندر دے کنارے توں کجھ دور نئیں، اوہناں کول اک مندر اے، جہدے رافٹرز تے بیم وہیل بونز نال بنے ہوئے نیں؛ کیونکہ راکشسی سائز دیاں وہیلاں نوں اکثر اوس ساحل تے مردہ سُٹ دتا جاندا اے۔ عام لوک تصور کردے نیں، کہ مندر تے رب ولوں دتی گئی خفیہ طاقت دے ذریعے، کوئی وی وہیل فوری طور تے موت دے بغیر نئیں لنگھ سکدا، پر متھ دی سچائی اے۔ مندر دے دوہاں پاسے، چٹاناں نیں جہڑیاں دو میل سمندر وچ گولی ماردیاں نیں، تے جدوں اوہ اوہناں تے چمکدیاں نیں تے اوہناں نوں زخمی کر دتا جاندا اے۔ اوہ اک معجزے لئی اک ناقابل یقین لمبائی دی وہیل دی پسلی رکھدے نیں، جہڑی زمین تے پئی ہوئی اے تے اوہدا محدب حصہ سب توں اُپر اے، اک محراب بناندا اے، جہدا سر اک بیک دے ذریعے نئیں اپڑ سکدا۔ (جان لیو کہندا اے) کہیا جاندا اے کہ میں ویکھن توں سو سال پہلے اوتھے پیا سی۔

وہیل دے اس افریقی مندر وچ میں تہانوں چھڈدا ہاں، قاری، تے جے تسی اک نانٹکیٹر او، تے اک وہیل مین او، تے تسی چپ کر کے اوتھے عبادت کرو گے۔





باب 105. کی وہیل دی شدت گھٹ جاندی اے؟—کی اوہ مر جائے گا؟

فیر، جدوں تک ایہہ لیویتھن ہمیشگی دے سر دے پانیاں توں ساڈے تے اتردا اے، ایہہ مناسب طور تے پچھیا جا سکدا اے، کہ کیہ، اپنیاں نسلاں دے لمے دوران، اوہ اپنے باپ دادا دے اصل وڈے حصے توں وگڑیا نئیں اے۔

پر تحقیقات کرن تے سانوں پتہ چلدا اے، کہ نہ صرف موجودہ دور دیاں وہیلاں اوہناں توں وڈیائی وچ بہتر نیں جنہاں دے جیواشم باقیات ترتیری نظام وچ پائے جاندے نیں (انسان توں پہلے دے اک وکھرے ارضیاتی دور نوں اپناندے نیں)، بلکہ اوس ترتیری نظام وچ پائے جان آلے وہیلاں وچوں، اوہ جیہڑے ایہدے بعد دے فارمیشناں نال تعلق رکھدے نیں، اوہدے سائز وچ ایہدے پہلے وہیلاں توں ودھ اے۔

اجے تیکر کڈھے گئے سارے پری ایڈمائٹ وہیلاں چوں، سب توں وڈی الاباما اے جہدا ذکر پچھلے باب چ کیتا گیا اے، تے اوہ کنکال چ ستر فٹ توں وی گھٹ لمبا سی۔ جد کہ، اسی پہلے ای ویکھ چکے آں، کہ ٹیپ میجر اک وڈے سائز دی جدید وہیل دے کنکال لئی ستر فٹ دیندا اے۔ تے میں سُنیا اے، وہیلمین دے اختیار تے، کہ سپرم وہیلاں نوں پھڑن دے ویلے سو فٹ دے نیڑے پھڑیا گیا اے۔

لیکن ایہ نئیں ہو سکدا، کہ جد کہ موجودہ ویلے دے وہیل سارے پچھلے ارضیاتی ادوار دے وہیلاں دے مقابلے چ شدت چ ترقی کردے نیں; کی ایہہ نئیں ہو سکدا، کہ آدم دے ویلے توں اوہ زوال پذیر ہو چکے نیں؟

یقیناً، سانوں ایہ نتیجہ اخذ کرنا چاہیدا اے، جے اسی پلینی ورگے حضرات دے کھاتیاں تے عام طور تے قدیم فطرت پسنداں دے بیاناں دا سہرا دینا چاہنے آں۔ کیونکہ پلینی سانوں وہیلاں دے بارے چ دسدا اے جہناں نے ایکڑاں دے جیوندے بلک نوں اپنایا سی، تے دوجیاں دے الڈروونڈس جنہاں دی لمبائی اٹھ سو فٹ سی - روپ واک تے وہیلاں دے تھیمز سرنگاں! تے ایتھوں تیکر کہ بینکس تے سولینڈر، کوک دے قدرتی ماہرین دے زمانے چ وی، اسی اکیڈمی آف سائنسز دے اک ڈینش ممبر نو اک سو ویہہ گز تے بعض آئس لینڈ وہیلز (ریڈن-سسکور، یا جھرڑیاں آلے پیٹ) نو تھلے رکھدے ہوئے ویکھنے آں; یعنی، تن سو سٹھ فٹ۔ تے لاسیپیڈ، فرانسیسی نیچرلسٹ، وہیلاں دی اپنی وسیع تاریخ وچ، اپنے کم دے بالکل شروع وچ (صفحہ 3)، دائیں وہیل نو اک سو میٹر، تن سو اٹھائی فٹ تے رکھدا اے۔ تے ایہہ رچنا 1825 عیسوی وچّ اتنی دیر نال چھپی۔

پر کیہ کوئی وہیل مین ایہناں کہانیاں تے یقین کرے گا؟ نئیں۔ اج دی وہیل پلینی دے ویلے دے اپنے وڈکیاں جنی وڈی اے۔ تے جے میں کدی اوتھے جاواں گا جتھے پلینی اے، میں، اک وہیل مین (اوہدے توں ودھ)، اوہنوں ایہ دسن لئی دلیری کراں گا۔ کیونکہ میں ایہ نئیں سمجھ سکدا کہ ایہ کیویں اے، کہ جدوں کہ مصری ممیاں جہڑیاں کہ پلینی دے جمن توں ہزاراں سال پہلاں دفن کیتیاں گئیاں سن، اپنے تابوتاں وچ اینا ناپ نئیں لیندیاں جنیاں کہ اک جدید کینٹکی اپنے جراباں وچ؛ تے جد کہ سب توں پرانے مصری تے نینوا دی تختیاں تے مورتی بنائے گئے مویشی تے دوجے جانور، اوہناں دے نسبتاً تناسب دے لحاظ نال جہدے وچ اوہ کھچے گئے نیں، بالکل اینج ای واضح طور تے ثابت کردے نیں کہ سمتھ فیلڈ دے اعلیٰ نسل دے، اسٹال فیڈ، انعام آلے مویشی، نہ صرف فرعون دے موٹے موٹے مویشی دے برابر نیں، بلکہ شدت وچ کدرے ودھ نیں؛ ایس سب دے باوجود، میں ایہ نئیں مناں گا کہ سارے جانوراں چوں صرف وہیل نوں ای زوال پذیر ہونا چاہیدا سی۔

پر فیر وی اک ہور انکوائری باقی اے; اک اکثر زیادہ نراشاوادی نانٹکیٹرز ولوں ہلایا جاندا اے۔ چاہے وہیل جہازاں دے مستولاں دے سراں تے تقریباً ہر چیز جانن آلی نظراں دی وجہ توں، ہن بہرنگ دے آبنائے دے ذریعے وی گھس رئے نیں، تے دنیا دے دور دراز دے خفیہ درازاں تے لاکراں وچ وی; تے ہزاراں ہارپون تے برچھے سارے براعظمی ساحلاں تے ٹُر پئے; بحث دا نقطہ ایہ اے کہ کی لیویتھن اینی وڈی تعاقب، تے اینی بے رحم تباہی نوں لمبے عرصے تک برداشت کر سکدا اے۔ چاہے اوہنوں آخر کار پانیاں چوں ختم نئیں کیتا جانا چاہیدا، تے آخری وہیل، آخری بندے دی طرح، اپنا آخری پائپ پیندی اے، تے فیر خود آخری پف چ بخارات بن جاندا اے۔

وہیلاں دے کوبڑے آلے جھنڈاں دا موازنہ بھینساں دے کوبڑے آلے جھنڈاں نال کرنا، جہڑے، چالھی سال پہلے نئیں، ایلی نوائے تے مسوری دے پریریاں تے دسیاں ہزاراں لوکاں نال پھیلے ہوئے سن، تے اپنیاں لوہے دیاں کھنبھاں نوں ہلاندے سن تے اپنی گرج نال بھرے ہوئے بھوآں نال دریا دی دارالحکومت دی آبادی آلی تھاواں تے جھکدے سن، جتھے شائستہ دلال اک ڈالر تے زمین ویچدا اے۔ انچ; اس طرح دے موازنہ وچ اک ناقابل رُکاوٹ دلیل پیش کیتی گئی لگدی اے، تاکہ ایہ ظاہر کیتا جا سکے کہ شکار کیتی گئی وہیل ہن تیزی نال ختم ہون توں نئیں بچ سکدی۔

پر تہانوں ایس معاملے نوں ہر نظر نال ویکھنا چاہیدا اے۔ بھاویں کجھ چر پہلاں اینا چھوٹا سی—اک چنگی زندگی نہیں—ایلی نوائے وچ بھینساں دی مردم شماری ہن لندن وچ مرداں دی مردم شماری توں ودھ گئی سی، اتے بھاویں اجوکے دن اوہناں وچوں اک وی سینگ جاں کھر اس سارے کھیتر وچ نہیں بچیا؛ تے اگرچہ اس حیرت انگیز خاتمے دا سبب انسان دا نیزہ سی؛ فیر وی وہیل دے شکار دی بہوں وکھری نوعیت لیویتھن دے اینے بے شان خاتمے توں منع کردی اے۔ اڑتالیس مہینیاں توں سپرم وہیلاں دا شکار کرن آلے اک جہاز چ چالی بندے سوچدے نیں کہ اوہناں نے بہوں چنگا کم کیتا اے، تے رب دا شکر اے، جے آخر کار اوہ چالی مچھیاں دا تیل گھر لے جاندے نیں۔ جد کہ، مغرب دے پرانے کینیڈین تے ہندوستانی شکاریاں تے ٹریپراں دے زمانے وچ، جدوں دور مغرب (جس دے غروب آفتاب وچ سورج ہلے وی چڑھدے نیں) اک بیابان تے اک کنواری سی، اسی تعداد وچ موکاسین والے آدمی، اسی تعداد وچ مہینیاں لئی، جہازاں وچ سفر کرن دی بجائے گھوڑے تے سوار، ہزاراں لئی نئیں، بلکہ ہزاراں تے زیادہ قتل کرن گے؛ اک حقیقت جہڑی، جے لوڑ ہوندی، تے اعداد و شمار دے لحاظ نال بیان کیتا جا سکدا سی۔

نہ ہی، ٹھیک سمجھیا جاندا اے، ایہہ سپرم وہیل دے ہولی ہولی ختم ہون دے حق وچ کوئی دلیل لگدا اے، مثال دے طور تے، کہ سابقہ ​​سالاں وچ (پچھلی صدی دے آخری حصے، کہو) ایہ لیویتھن، نکے پھلیاں وچ، موجودہ ویلے دے مقابلے وچ بہت زیادہ کثرت نال سامنا کردے سن، تے، نتیجے دے طور تے، سفر اتنے لمبے نئیں سن، تے بہت زیادہ منافع بخش وی سن۔ کیونکہ، جویں کہ کہتھے ہور نوٹ کیتا گیا اے، اوہ وہیل، حفاظت دے کجھ نظریات توں متاثر ہو کے، ہن بے پناہ قافلے وچ سمندراں وچ تیردے نیں، تاں جے وڈی حد تک بکھرے ہوئے تنہا، جوئے، تے پھلیاں، تے دوجے دناں دے سکول ہن وسیع لیکن وسیع پیمانے تے الگ تھلگ، غیر معمولی فوجاں وچ جمع ہو گئے ہن۔ بس اینا ای اے۔ تے اینا ای غلط فخر لگدا اے، کہ کیونکہ نام نہاد وہیل بون وہیل ہن اوہناں نال بھرپور پچھلے سالاں چ بہت سارے میداناں نوں پریشان نئیں کردے، اس لی اوہ پرجاتی وی گھٹدی جا رئی اے۔ کیونکہ اوہناں نوں صرف پرومونٹری توں کیپ تیکر دھکیلیا جا رہیا اے۔ تے جے کوئی ساحل اوہناں دے جیٹاں نال زندہ نئیں اے، تے فیر، پک کرو، کجھ ہور تے دور دراز دے سٹرنڈ نوں حال ای چ انجان تماشے نال حیران کر دتا گیا اے۔

مزید برآں: انہاں آخری ذکر کردہ لیویتھناں دے بارے چ، انہاں دے کول دو پکے قلعے ہن، جہڑے، تمام انسانی امکانات دے مطابق، ہمیشہ دے لیی ناقابل تسخیر رہن گے۔ تے جیویں کہ اوہناں دیاں وادیاں تے حملے تے، ٹھنڈے سوئس اپنے پہاڑاں ول پچھے ہٹ گئے؛ اس لیی، وچکارلے سمندراں دے سوانا تے گلیڈز توں شکار کیتا جاندا اے، وہیل بون وہیل آخر کار اپنے پولر قلعےآں دا سہارا لے سکدے نیں، تے اوتھے دی حتمی شیشے آلی رکاوٹاں تے دیواراں دے تھلے غوطہ لاندے نیں، برفیلی کھیتاں تے فلوز دے وچکار آندے نیں; تے ہمیشگی دے دسمبر دے اک دلکش دائرے وچ، انسان توں ہر تعاقب دی خلاف ورزی کیتی۔

لیکن جویں کہ شاید ایہناں وہیل بون وہیلاں چوں پنجاہ اک کیچالوٹ لئی ہارپون کیتے جاندے نیں، پیش گوئی دے کجھ فلسفیاں نے ایہ نتیجہ اخذ کیتا اے کہ اس مثبت تباہی نے پہلے ای اوہناں دیاں بٹالیناں نوں بہت گمبھیر طور تے گھٹا دتا اے۔ لیکن اگرچہ پچھلے کجھ عرصے توں ایہناں وہیلاں دی اک وڈی تعداد، 13,000 توں گھٹ نئیں، صرف امریکیاں دے ذریعہ شمالی مغربی ساحل تے سالانہ قتل کیتیاں جاندیاں نیں; فیر وی کجھ غور و فکر موجود نیں جہڑے اس معاملے وچ مخالف دلیل دے طور تے بہت گھٹ یا کوئی حساب نئیں دے اس حالات نو وی پیش کردے ہن۔

دنیا دے زیادہ وڈے مخلوق دی آبادی دے بارے چ کجھ ناقابل یقین ہونا فطری اے، فیر وی اسی گوا دے مورخ ہارتو نوں کی آکھئیے، جدوں اوہ سانوں دسدا اے کہ اک شکار چ سیام دے بادشاہ نے 4,000 ہاتھی لئے سن؛ کہ انہاں علاقیاں وچ ہاتھی معتدل موسم وچ مویشیاں دے جھنڈاں دے طور تے بے شمار ہن۔ تے ایس گل تے شک کرن دی کوئی وجہ نئیں لگدی کہ جے ایہ ہاتھی، جہناں دا ہن ہزاراں سالاں توں شکار کیتا جا رہیا اے، سیمیرامس، پورس، ہنیبال، تے مشرق دے سارے بعد دے بادشاہاں ولوں—جے اوہ اجے وی وڈی تعداد وچ اوتھے زندہ رہندے نیں، تے بہوں زیادہ ہو سکدا اے کہ عظیم وہیل سارے شکار توں اگے ودھ سکے، کیونکہ اوہدے کول ایشیا وچ اک چراگاہ اے، جہڑی ایشیا دوناں توں دوگنا اے، دوگنا، عین مطابق یورپ تے افریقہ، نیو ہالینڈ تے سمندر دے سارے جزیرے رل کے۔

مزید برآں: سانوں غور کرنا اے، کہ وہیلاں دی مفروضہ عظیم لمبی عمر توں، انہاں دی عمر اک صدی تے اس توں زیادہ حاصل کرنا، اس لی کسی وی اک عرصے وچ، کئی وکھ بالغ نسلاں ہم عصر ہونیاں چاہیدیاں نیں. تے ایہ کی اے، اسی چھیتی ای کجھ اندازہ لا سکنے آں، تخلیق دے سارے قبرستاناں، قبرستاناں تے خاندانی والٹاں دا تصور کر کے جہڑے سارے مرداں، عورتاں تے بچیاں دے زندہ جسماں نوں پیش کردے نیں جہڑے پچتر سال پہلے زندہ سن؛ تے اس ان گنت میزبان نو دنیا دی موجودہ انسانی آبادی چ شامل کرنا۔

اس لی، انہاں ساریاں شیواں دے لیی، اسی وہیل نو اس دی نسل چ امر سمجھنے آں، چاہے اوہ اس دی انفرادیت چ کِنّا وی خراب ہو جائے۔ براعظماں دے پانی توڑن توں پہلاں اوہنے سمندراں نوں تیریا؛ اوہ اک وار ٹائلریز، تے ونڈسر کیسل، تے کریملن دی تھاں تے تیریا۔ نوح دے سیلاب وچ اُوہنے نوح دے کشتی نوں حقیر سمجھیا؛ تے جے کدی وی دنیا نو نیدرلینڈز دی طرح، اپنڑے چوہے نو مارن دے لیی فیر سیلاب آ جائے، تے فیر وی ابدی وہیل زندہ رہوے گی، تے استوائی سیلاب دی سب توں اُتلی چوٹی تے پال کے، اپنی جھاگ آلی سرکشی نو اسماناں تیکر ٹپکے گی۔





باب 106. اخاب دی ٹنگ۔

کیپٹن اخاب نے لندن دے سیموئیل اینڈربی نو چھڈن دا تیز انداز، اپنے آپ دے نال کجھ نکے جہے تشدد دے نال بے دھیان نئیں سی۔ اوہنے اپنی کشتی دے اک رکاوٹ تے اینی توانائی نال روشنی کیتی سی کہ اوہدی ہاتھی داند دی لت نوں ادھا ٹٹن آلا جھٹکا لگیا سی۔ تے جدوں اپنا ڈیک حاصل کرن دے بعد، تے اوتھے اپنا پیوٹ ہول حاصل کرن دے بعد، اس نے اتنی شدت نال سٹیئر مین نوں اک فوری حکم دے نال چکر لایا (ایہ، ہمیشہ دی طرح، اس دے کافی لچکدار سٹیئرنگ نہ کرن دے بارے چ کجھ سی)؛ فیر، پہلے ای ہلے ہوئے ہاتھی داند نوں اجیہا وادھو موڑ تے رنچ ملیا، کہ بھاویں ایہہ ہُن وی پورا رہیا، تے سارے ظاہری طور تے ہوس بھریا سی، فیر وی اخاب نے اینوں مکمل طور تے قابل اعتماد نئیں سمجھیا۔

تے، واقعی، ایہہ حیرانی لئی نکا جیہا معاملہ لگدا سی، کہ اپنی ساری پھیلی ہوئی، پاگل لاپرواہی دے باوجود، اخاب نے کدی کدی اوس مردہ ہڈی دی حالت تے احتیاط نال دھیان دتا جس تے اوہ جزوی طور تے کھلوتا سی۔ کیونکہ نانٹکیٹ توں پیکوڈ دے جہاز رانی توں بہتا چر پہلاں نئیں ہویا سی، کہ اوہ اک رات زمین تے پیا ہویا سی، تے بے حس سی; کجھ نامعلوم، تے بظاہر ناقابل بیان، ناقابل تصور ہلاکت دے ذریعے، اس دا ہاتھی داند دا انگ اینا تشدد نال بے گھر ہو گیا سی، کہ اس نوں داؤ دے لحاظ نال ماریا گیا سی، تے اس دی گردن نو چھید دتا گیا سی; نہ ای ایہ انتہائی مشکل دے بغیر سی کہ تکلیف دہ زخم مکمل طور تے ٹھیک ہو گیا سی۔

نہ ای، اوس ویلے، ایہ گل اوہدے مونومینیک دماغ وچ داخل ہون وچ ناکام رہی سی، کہ اوس ویلے دے موجودہ دکھ دی ساری تکلیف صرف اک سابقہ ​​افسوس دا براہ راست مسئلہ سی; تے اوہ وی صاف طور تے ویکھدا لگدا سی، کہ جیویں کہ دلدل دا سب توں زہریلا رینگنے آلا جانور، اپنی قسم نو ناگزیر طور تے اینا ای برقرار رکھدا اے جنا کہ گرو دا سب توں پیارا گاناکار؛ اس لیی، ہر خوش قسمتی دے برابر، تمام دکھی واقعات قدرتی طور تے اپنے ورگے پیدا کردے نے۔ ہاں، برابر توں ودھ، اخاب نے سوچیا; کیونکہ غم دی نسل تے نسل دونو خوشی دی نسل تے نسل توں اگے جاندے ہن۔ کیونکہ، اس دا اشارہ نئیں کرنا: کہ ایہ بعض کینن تعلیمات توں اک نتیجہ اے، کہ جد کہ ایتھے کجھ قدرتی لذتاں دا دوجی دنیا لئی کوئی اولاد پیدا نئیں ہووے گا، بلکہ، اس دے برعکس، تمام جہنم دی مایوسی دی خوشی-بے اولاد دے بعد ہووے گا؛ جد کہ، کجھ مجرم فانی مصیبت اجے وی اپنے آپ نوں قبر توں باہر دکھاں دی اک ابدی ترقی پسند اولاد نوں زرخیز طور تے جنم دین گیاں۔ ایس گل دا اشارہ کرن لئی بالکل نئیں، ایس گل دے ڈوہنگے تجزیے وچ ہُن وی عدم مساوات نظر آندی اے۔ کیونکہ، اخاب نے سوچیا، جد کہ اعلیٰ ترین زمینی نعمتاں وچ وی اک خاص غیر معنی خیز معمولی گلاں ہوندیاں نیں، لیکن، تھلے، تمام دل دیاں پریشانیاں، اک صوفیانہ اہمیت، تے، کجھ لوکاں وچ، اک آرچنجیلک عظمت؛ ایس لئی اوہناں دے محنتی ٹریسنگ آؤٹ واضح کٹوتی نوں جھوٹ نئیں کردے۔ ایہناں اعلیٰ فانی مصیبتاں دے شجرے دا پچھا کرنا، سانوں آخر کار دیوتاواں دے بے وسیلے پرائموجینٹرز دے وچکار لے جاندا اے۔ تاں جے، سارے خوش، گھاس بنان آلے سورجاں، تے نرم جھنجھناہٹ، گول فصل دے چناں دے سامنے، سانوں ایس گل نوں ہار مننی پوے گی: کہ دیوتا خود ہمیشہ لئی خوش نئیں ہوندے۔ انسان دے بھوئیں تے ناقابل مٹ، اداس پیدائشی نشان، صرف دستخط کرن آلیاں وچ غم دی مہر اے۔

ایتھے نادانستہ طور تے اک راز ظاہر کیتا گیا اے، جہڑا شاید زیادہ مناسب طریقے نال، مقررہ طریقے نال، پہلے وی ظاہر کیتا گیا ہووے گا۔ اخیاب دے بارے چ کئی ہور تفصیلات دے نال، ہمیشہ کجھ لوکاں لئی ایہہ اک راز بنیا ہویا سی، کہ ایہہ کیوں سی، کہ اک خاص عرصے لئی، پیکوڈ دی جہاز رانی توں پہلاں تے بعد چ، اس نے اپنے آپ نو اجہے گرینڈ لاما ورگی خصوصیت دے نال لُکا لیا سی; تے، اس اک وقفے لئی، مردیاں دی سنگ مرمر دی سینیٹ دے وچکار، جویں کہ ایہہ سی، بے زبان پناہ لئی۔ کیپٹن پیلیگ دی ایس گل دی گھٹیا وجہ کسے وی طرحاں مناسب نئیں لگدی سی; اگرچہ، درحقیقت، اخاب دے سارے ڈوہنگے حصے نو چھون دے طور تے، ہر وحی نے وضاحتی روشنی دے مقابلے چ زیادہ اہم ہنیرے دا حصہ لیا۔ لیکن، آخر چ، ایہ سب کجھ سامنے آیا; کم از کم، اس اک معاملے نے کیتا۔ اوہ بھیانک حادثہ اوہدے عارضی تنہائی دے تھلے سی۔ تے نہ صرف ایہ، بلکہ اس ہمیشہ معاہدہ کرن آلے، ساحل تے گرن آلے دائرے لئی، جہڑا، کسی وی وجہ توں، اوہدے نال گھٹ ممنوعہ نقطہ نظر دا استحقاق رکھدا سی; اس ڈرپوک دائرے لئی اپروکت اشارہ کیتا گیا ہلاکت - رہ گیا، جویں کہ ایہہ سی، اخاب ولوں موڈی طور تے بے حساب - اپنے آپ نوں دہشت گردی نال سرمایہ کاری کیتا، پوری طرحاں روحاں تے ہاکاں دی دھرتی توں محروم نئیں سی۔ ایس لئی کہ، اوہدے لئی اپنے جوش و خروش دے ذریعے، اوہناں ساریاں نے سازش کیتی سی، جتھے تیکر اوہناں دے اندر سی، دوجیاں توں ایس گل دے علم نوں دبا دین لئی؛ تے اس لی ایہ سی، کہ جدوں تک کافی وقفہ نئیں لنگھ گیا، ایہ پیکوڈ دے ڈیکاں تے نئیں آیا۔

لیکن ایہ سب کجھ انج ای ہووے؛ ہوا وچ غیر نظر آن آلے، مبہم سینڈ، یا اگ دے انتقامی شہزادے تے طاقتور، زمینی اخیاب دے نال کرنا یا نئیں کرنا، فیر وی، اپنی ٹنگ دے اس معاملے وچ، اس نے سادہ عملی طریقہ کار اپنایا؛—اس نے بڑھئی نو بلایا۔

تے جدوں اوہ کارکن اوہدے سامنڑے پیش ہویا، تے اوہنے بغیر دیر کیتے اوہنوں اک نویں ٹنگ بنان دا حکم دتا، تے ساتھیاں نوں ہدایت کیتی کہ اوہ اوہنوں جبڑے دے ہاتھی داند (سپرم وہیل) دے سارے سٹڈز تے جوسٹاں نال فراہم کردے ویکھن جہڑے ہن تیکر سفر تے جمع کیتے گئے سن، تاں جے سب توں موٹے، سب توں صاف دانے آلے بھرے ہوئے دا اک محتاط انتخاب محفوظ کیتا جا سکے۔ ایہ کم کرن دے بعد، ترکھان نوں اوس رات ٹانگ مکمل کرن دا حکم ملیا؛ تے اس دے لیی تمام فٹنگز فراہم کرنا، استعمال اچ ناقابل اعتماد توں متعلقہ لوکاں توں آزاد۔ مزید ایہ کہ، جہاز دے فورج نو ہولڈ چ اس دی عارضی بیکاری توں باہر لہران دا حکم دتا گیا سی; تے، معاملے نو تیز کرن دے لی، لوہار نو حکم دتا گیا کہ اوہ فوری طور تے جو وی لوہے دے سازاں دی لوڑ ہووے، اوہناں دی جعلی کاری لئی اگے ودھے۔
