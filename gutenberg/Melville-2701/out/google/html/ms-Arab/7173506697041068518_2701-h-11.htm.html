سبب يڠ ڽات مڠاڤ اهب تيدق مناءيقي كاڤل ڤاوس يڠ تله كامي كاتاكن اداله اين: اڠين دان لاوت مننداكن ريبوت. تتاڤي والاوڤون اين تيدق برلاكو، دي تيدق اكن سلڤس سموا، موڠكين، مناءيقيڽ—مڠيرا دري ڤريلاكوڽ ستروسڽ ڤد كتيك-كتيك يڠ سام—جك بڬيتو، دڠن ڤروسيس مڠوچڤكن سلامت داتڠ، دي تله منداڤتكن جواڤن نيڬاتيف كڤد ڤرتاڽان يڠ دي تاڽكن. كران، سڤرتي يڠ اخيرڽ ترڽات، دي تيدق ڤريهاتين اونتوق برڬادوه، والاوڤون سلاما ليم مينيت، دڠن مان-مان كڤتن يڠ لبيه اسيڠ، كچوالي دي بوليه مڽومبڠ ببراڤ معلومت يڠ دي چاري دڠن بڬيتو مڽرڤ. تتاڤي سموا اين موڠكين تتڤ دأڠڬركن تيدق منچوكوڤي، جك تيدق اد سسواتو يڠ دكاتاكن د سيني تنتڠ كڬونأن كاڤل ڤمبوروان ڤاوس يڠ انيه اڤابيلا برتمو ساتو سام لا؞ين د لاوت اسيڠ، دان تراوتماڽ د ڤادڠ ڤلايرن يڠ سام.

جك دوا اورڠ اسيڠ ملواتي ڤين باررن د نڬري نيو يورق، اتاو ڤادڠ ڤاسير ساليسبوري يڠ سام سوڽي د ايڠڬريس; جك سچارا ريڠن برتمو ساتو سام لا؞ين د هوتن يڠ تيدق راڤت، كدوا-دواڽ، اونتوق كهيدوڤن مريك، تيدق بوليه مڠيلقكن ساليڠ مڠوچڤكن سلامت داتڠ; دان برهنتي سكجڤ اونتوق برتوكر-توكر بريتا; دان، موڠكين، دودوق سكجڤ دان براستراحة دالم كونسرت: كمودين، بتاڤ لبيه سمولا جاديڽ بهاوا د اتس ڤين باررن يڠ تيدق ترباتس دان داتارن ساليسبوري د لاوت، دوا كاڤل ڤنچوري ڤاوس يڠ ساليڠ مڠهينا د هوجوڠ بومي—د لوار ڤولاو فانيڠ يڠ ترسنديري، اتاو كيلڠ راج يڠ جاءوه; بتاڤ لبيه سمولا جادي، ساي كاتاكن، بهوا دالم كادأن سڤرتي اين كاڤل-كاڤل اين تيدق هاڽ برتوكر-توكر هوجن لبت، تتاڤي داتڠ ك دالم هوبوڠن يڠ لبيه راڤت، لبيه راڤت دان برڬاول. دان تراوتماڽ اكن اين دسڠكا سباڬاي ڤركارا يڠ سوده تنتو، دالم حال كاڤل-كاڤل يڠ دميليكي د ساتو ڤلابوهن لاوت، دان يڠ مان كڤتن، ڤڬاواي، دان تيدق سديكيت درڤد للاكي-للاكي ايت دكنلي سچارا ڤريبادي انتارا ساتو سام لا؞ين; دان عقيبةڽ، ممڤوڽاءي ڤلباڬاي ڤركارا دوميستيك يڠ دسايڠي اونتوق دڤركاتاكن.

باڬي كاڤل يڠ سوده لاما تيدق اد، ڤمباتسن لوار، موڠكين، ممڤوڽاءي سورت د اتس كاڤل; ڤد اڤ جوا كادأن، دي اكن ڤستي ممبياركنڽ ممڤوڽاءي ببراڤ كرتس تاريخ ستاهون اتاو دوا لبيه لمبت درڤد يڠ تراخير ڤد فايلڽ يڠ كابور دان دڤاكاي اوليه ايبو جاري. دان سباڬاي بالسن اتس سوڤن ايت، كاڤل يڠ اكن منوجو ك لوار اكن منريما كچرداسن ڤنچارين ​​ڤاوس تربارو دري تانه ڤلايرن يڠ موڠكين دتقديركن، ڤركارا يڠ ڤاليڠ ڤنتيڠ باڬيڽ. دان دالم درجه، سموا اين اكن برلاكو بنر مڠناءي كاڤل ڤنچوري ڤاوس يڠ ملواتي لالوان ماسيڠ-ماسيڠ د ڤادڠ ڤلايرن ايت سنديري، والاوڤون مريك سام-سام لاما تيدق برادا د رومه. كران ساله ساتو درڤد مريك موڠكين تله منريما ڤميندهن سورت درڤد كاڤل كتيڬ، دان كيني جاوه ترجاوه; دان ببراڤ درڤد سورت-سورت ايت موڠكين اونتوق اورڠ-اورڠ د كاڤل يڠ دي تموي سكارڠ. سلاين ايت، مريك اكن برتوكر بريتا ڤمبوروان ڤاوس، دان بربوال-بوال يڠ بوليه دڤرستوجوي. كران بوكن سهاج مريك اكن برتمو دڠن سموا سيمڤاتي ڤارا ڤلاير، تتاڤي جوڬ دڠن سموا كڤرچايأن يڠ انيه يڠ تيمبول درڤد ڤڠجرن يڠ سام دان ككوراڠن دان بهاي يڠ دكوڠسي برسام.

ڤربيذأن نڬارا جوڬ تيدق اكن ممبوات سبارڠ ڤربيذأن يڠ ساڠت ڤنتيڠ; ايايت، سلاڬي كدوا-دوا ڤيهق برتوتور دالم ساتو بهاس، سڤرتي حالڽ دڠن اورڠ اميريك دان ايڠڬريس. والاوڤون، اونتوق دڤستيكن، دري جومله ڤنچوري ڤاوس ايڠڬريس يڠ سديكيت، ڤرتموان ترسبوت تيدق سريڠ برلاكو، دان اڤابيلا اي برلاكو، ترلالو سسواي اونتوق اد سماچم راس مالو د انتارا مريك; كران اورڠ ايڠڬريس اندا اڬق ريزب، دان يانكي اندا، دي تيدق سوكاكن ڤركارا سماچم ايت ڤد سسياڤا ڤون كچوالي ديريڽ سنديري. سلاين ايت، ڤارا ڤنچوري ايڠڬريس تركادڠ منججسكن سماچم كأوڠڬولن ميتروڤوليتن بربنديڠ ڤارا ڤنچوري اميريكا; مڠناءي نانتوچتر يڠ ڤنجڠ دان لمبوت، دڠن ڤروۏينسياليسمىڽ يڠ تيدق دڬامبركن، سباڬاي سماچم ڤتاني لاوت. تتاڤي د مان كأوڠڬولن اين دالم ڤارا ڤمبورو ڤاوس ايڠڬريس بنر-بنر ترديري، سوكر اونتوق دكاتاكن، مليهت بهاوا يانكي دالم ساتو هاري، سچارا كوليكتيف، ممبونوه لبيه باڽق ڤاوس درڤد سموا ايڠڬريس، سچارا كوليكتيف، دالم سڤولوه تاهون. تتاڤي اين اداله كلمهن كچيل يڠ تيدق بربهاي دالم ڤمبورو ڤاوس ايڠڬريس، يڠ مان اورڠ نانتوكيت تيدق مڠمبيل برت; موڠكين، كران دي تاهو بهاوا دي ممڤوڽاءي ببراڤ كلمهن سنديري.

جادي، مك، كيت مليهت بهاوا درڤد سموا كاڤل يڠ برلاير سچارا برسسواين د لاوت، ڤارا ڤنچوري ڤاوس ممڤوڽاءي كبڽقان السن اونتوق برڬاول—دان مريك بڬيتو. سداڠكن، ببراڤ كاڤل داڬڠ يڠ ملواتي ڤارس ساتو سام لا؞ين د ڤرتڠهن اتلانتيق، سريڠ كالي اكن برلالو تنڤا سبارڠ ڤركاتان ڤڠاكوان، ساليڠ مموتوڠ ساتو سام لا؞ين د لاوت تيڠڬي، سڤرتي ڬولوڠن داندي د برودواي; دان سڤنجڠ ماس منجمڤوت، موڠكين، دالم كريتيك فينيكل ترهادڤ ريڬ ماسيڠ-ماسيڠ. باڬي ڤارا ڤڤراڠن، اڤابيلا مريك بركسمڤتن برتمو د لاوت، مريك ترلبيه دهولو ملالوءي رنتاي ڤنوتوڤن دان ڤڠيكيسن يڠ كونچو، ڤنوتوڤن بنديرا، يڠ تيدق نمڤق اد باڽق نية باءيق دان چينتا برساودارا تنتڠڽ سام سكالي. سباڬايمان يڠ مڽنتوه ڤرتموان كاڤل-كاڤل همبا، مڠاڤ، مريك دالم كادأن ترڬسا-ڬسا يڠ لوار بياسا، مريك لاري دري ساتو سام لا؞ين سچڤت موڠكين. دان باڬي ڤارا ڤنچوري، اڤابيلا مريك بركسمڤتن اونتوق ملڠڬر تولڠ سيلڠ ماسيڠ-ماسيڠ، هوجن لبت ڤرتام اداله—“براڤ باڽق تڠكورق؟“—چارا يڠ سام دڠن ڤارا ڤنچوري ڤاوس—“براڤ باڽق توڠ؟“ دان ڤرتاڽان ايت اڤابيلا دجواب، ڤارا ڤنچوري تروس برڤيسه، كران مريك اداله ڤنجاجه نراك د كدوا-دوا بله ڤيهق، دان تيدق سوك مليهت باڽق ڤرسامأن ڤنجاجه ماسيڠ-ماسيڠ.

تتاڤي ليهتله ڤمبورو ڤاوس يڠ برتقوى، جوجور، تيدق برڤواس هاتي، رامه، برڬاول، بيبس دان موده! اڤ يڠ دلاكوكن اوليه ڤمنڠ ڤاوس اڤابيلا دي برتمو دڠن ڤمنڠ ڤاوس يڠ لا؞ين دالم چواچ يڠ باءيق؟ دي ممڤوڽاءي “ ڬم ،“ سسواتو يڠ ساڠت تيدق دكنلي اوليه سموا كاڤل يڠ لا؞ين سهيڠڬ مريك تيدق ڤرنه مندڠر نام ايت ڤون; دان جك سچارا كبتولن مريك مندڠر تنتڠڽ، مريك هاڽ ترسڽوم ڤدڽ، دان مڠولاڠي ڤركارا-ڤركارا يڠ برماءين-ماءين تنتڠ “spouters” دان “blubber-boilers،“ دان سباڬايڽ سڤرتي سروان يڠ چوكوڤ اينده. مڠاڤكه سموا ڤلاير ڤداڬڠ، دان جوڬ سموا ڤارا ڤنچوري دان ڤارا ڤلاير ڤراڠ، دان ڤلاير كاڤل همبا، مڠهرڬاي ڤراسان مڠهينا ترهادڤ كاڤل ڤاوس; اين اداله سوءالن يڠ سوكر اونتوق دجواب. كارنا، دالم حال ڤنچوري، كاتاكن، ساي ايڠين تاهو سام اد ڤروفيسي مريك ايت ممڤوڽاءي سبارڠ كڬميلڠن يڠ انيه تنتڠڽ. اي تركادڠ براخير دڠن كڤنجڠن يڠ تيدق بياسا، ميمڠ; تتاڤي هاڽ د ڬلڠڬڠ. دان سلاين ايت، اڤابيلا سأورڠ للاكي دأڠكت دڠن چارا يڠ ڬنجيل ايت، دي تيدق ممڤوڽاءي اساس يڠ سسواي اونتوق كڤنجڠنڽ يڠ لبيه تيڠڬي. اوليه ايت، ساي مڽيمڤولكن، بهاوا دالم ممبڠڬاكن ديريڽ سباڬاي تيڠڬي د اتس سأورڠ ڤنجاڬ ڤاوس، دالم ڤرڽاتأن ايت ڤنچوري ايت تيدق ممڤوڽاءي اساس يڠ كوكوه اونتوق برديري.

تتاڤي اڤ ايت ڬم؟ اندا موڠكين اكن هابيس جاري تندا اندا برلاري ك اتس دان ك باوه كولوم قاموس، دان تيدق ڤرنه منموكن ڤركاتان ترسبوت. دوكتور جوهنسون تيدق ڤرنه منچاڤأي ڤڠتاهوان ايت; كاڤل نوح ويبستر تيدق ممڬڠڽ. والاو باڬايماناڤون، ڤركاتان ايكسڤريسيف يڠ سام اين كيني تله برتاهون-تاهون دڬوناكن سچارا برتروسن د كالڠن كيرا-كيرا ليما بلس ريبو يانكي يڠ لاهير سبنر. تنتو ساج، اي ممرلوكن دفينيسي، دان ڤرلو دماسوقكن ك دالم ليكسيك. دڠن ڤنداڠن ايت، ايذينكن ساي منديفينيسيكنڽ دڠن بلاجر.

ڬام. كات نام— ڤرتموان سوسيال دوا ( اتاو لبيه ) كاڤل ڤاوس، عمومڽ د ڤادڠ ڤلايرن; اڤابيلا، سلڤس برتوكر هوجن، مريك برتوكر لاوتن اوليه كرجا-كرجا ڤراهو: دوا كڤتن يڠ تيڠڬل، اونتوق ماس ايت، د اتس ساتو كاڤل، دان دوا راكن كتوا د اتس كاڤل يڠ لا؞ين.
