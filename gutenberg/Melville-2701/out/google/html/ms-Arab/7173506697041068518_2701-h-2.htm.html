باب 4. كونترڤن.

اڤابيلا ترباڠون ڤاڬي ايسوقڽ كيرا-كيرا سياڠ، ساي منموكن تاڠن Queequeg دليمڤركن ك اتس ساي دڠن چارا يڠ ڤاليڠ چينتا دان ڤنوه كاسيه سايڠ. اندا همڤير مڠڠڬڤ ساي تله منجادي استريڽ. كونترڤن اداله درڤد ڤاڤن تندا، ڤنوه دڠن ڤرسڬي دان سڬيتيڬ بروارنا ڤرتي كچيل يڠ ڬنجيل; دان ڬڠڬمنڽ اين دتاتو د سلوروهڽ دڠن لابيرين كريتن يڠ تيدق ترهيڠڬ دري سأورڠ توكوه، تياد دوا بهاڬين يڠ مان اداله ساتو بايڠن يڠ تڤت—دسببكن ساي راس كران دي منجاڬ ڬڠڬمنڽ د لاوت سچارا تيدق برتوجوان د باوه ماتاهاري دان بايڠ-بايڠ، ليهير باجوڽ دڬوليڠكن سچارا تيدق تراتور ڤد ڤلباڬاي ماس—ڬڠڬمڽ يڠ سام، ساي كاتاكن، كليهاتن سڤرتي سترڤ يڠ سام اونتوق دنيا ايت سليموت. ميمڠ، سباهاڬينڽ برباريڠ د اتسڽ سڤرتي يڠ دلاكوكن اوليه ڬڠڬم اڤابيلا ساي ڤرتام كالي ترباڠون، ساي سوكر اونتوق مڠتاهويڽ دري سليڤر، مريك بڬيتو مڠادوكن ورنا مريك برسام; دان هاڽ دڠن راس برت دان تكنن يڠ ساي بوليه تاهو بهاوا Queequeg مملوق ساي.

سنساسي ساي اداله ڤليق. ايذينكن ساي چوبا منرڠكنڽ. كتيك ساي كچيل، ساي ايڠت دڠن باءيق كادأن يڠ اڬق سام يڠ منيمڤا ساي; سام اد اي اداله كڽاتأن اتاو ايمڤين، ساي تيدق ڤرنه داڤت مننتوكن سڤنوهڽ. كادأنڽ اداله اين. ساي تله مموتوڠ ببراڤ كاڤر اتاو يڠ لاءين—ساي فيكير اي چوبا مرڠكق ك اتس چرمين، سڤرتي يڠ ساي ليهت سديكيت ڤڽاڤو ببراڤ هاري سبلومڽ; دان ايبو تيري ساي يڠ، اڤاته لاڬي، سلالو منچلا ساي، اتاو مڠهانتر ساي ك تمڤت تيدور تنڤا ماكن مالم،—ايبو ساي مڽرڠ ساي دڠن كاكي كلوار دري چرمين دان ممباوا ساي ك تمڤت تيدور، والاوڤون هاڽ ڤوكول دوا ڤتڠ ڤد 21 جون، هاري ترڤنجڠ دالم تاهون د هيميسفيرا كامي. ساي براس ساڠت مڠريكن. تتاڤي تياد بنتوان اونتوق ايت، جادي ك اتس تڠڬ ساي ڤرڬي ك بيليق كچيل ساي د تيڠكت تيڬ، ممبوك باجو ديري ساي سچارا ڤرلاهن-لاهن اڬر ممبونوه ماس، دان دڠن نفس ڤاهيت سمڤأي ك انتارا لمبارن.

ساي برباريڠ د سان دڠن سوكر مڠيرا بهاوا انم بلس جم سڤنوهڽ هاروس برلالو سبلوم ساي بوليه برهارڤ اونتوق كبنرن. انم بلس جم د اتس كاتيل! كچيل بلاكڠ ساي ساكيت اونتوق مميكيركنڽ. دان اي جوڬ بڬيتو ريڠن; ماتهاري برسينر د تيڠكڤ، دان ڬلومبڠ جورولاتيه يڠ هيبت د جالن-جالن، دان بوڽي سوارا ڬاي د سلوروه رومه. ساي براس سماكين تروق دان تروق—اخيرڽ ساي بڠكيت، برڤاكاين، دان تورون دڠن لمبوت دالم كاكي ساي يڠ برتودوڠ، منچاري ايبو تيري ساي، دان تيبا-تيبا ملمڤركن ديري ساي د كاكيڽ، مموهون كڤداڽ سباڬاي نعمت ترتنتو اونتوق ممبريكن ساي ڬلينچير يڠ باءيق اتس ساله لاكو ساي; اڤ سهاج ميمڠ تتاڤي مڠوتوق ساي اونتوق برباريڠ د اتس تيدور سلاما ماس يڠ تيدق ترتاهن. تتاڤي دي اداله ايبو تيري يڠ ترباءيق دان ڤاليڠ برسوڠڬوه-سوڠڬوه، دان كمبالي ساي ترڤقسا ڤرڬي ك بيليق ساي. سلاما ببراڤ جم ساي برباريڠ د سان ترجاڬ، مراسا لبيه تروق درڤد يڠ ڤرنه ساي لاكوكن سجق ايت، بهكن دري مصيبة تربسر ستروسڽ. اخيرڽ ساي ڤستي تله جاتوه ك دالم ميمڤي بوروق يڠ برمسئله دري تيدور; دان ڤرلاهن-لاهن ترباڠون دريڽ—سڤاروه ترڤنچيل دالم ايمڤين—ساي ممبوك مات ساي، دان بيليق سبلوم دڤنوهي ماتهاري كيني تله دبلڠكڤي دالم كڬلاڤن لوار. سرتا-مرتا ساي مراساي كجوتن يڠ برجالن ملالوءي سلوروه كراڠك ساي; تياد اڤ يڠ اكن دليهت، دان تياد اڤ يڠ اكن ددڠر; تتاڤي تاڠن ڬايب سأوله-اوله دلتقكن ڤد تاڠن ساي. تاڠن ساي ترڬنتوڠ د اتس كونترڤن، دان بنتوق اتاو فانتوم يڠ تيدق برنام، تيدق داڤت دبايڠكن، سوڽي، يڠ مان تاڠن ايت دميليكي، نمڤقڽ دودوق راڤت د تبيڠ كاتيل ساي. اونتوق اڤ يڠ دليهت اوسيا-اوسيا يڠ برتمبه اوسيا، ساي برباريڠ د سان، دبكوكن دڠن كتاكوتن يڠ ڤاليڠ مڠريكن، تيدق براني مڠهيلڠكن تاڠن ساي; نامون ڤرنه برڤيكير بهاوا جك ساي بوليه هاڽ مڠهيلڠكنڽ ساتو اينچي، منترا يڠ مڠريكن ايت اكن دڤچهكن. ساي تيدق تاهو باڬايمانا كسدرن اين اخيرڽ ملونچور دري ساي; تتاڤي ترباڠون ڤد وقتو ڤاڬي، ساي مڠيڠاتي سمواڽ دڠن ڬترن، دان سلاما ببراڤ هاري دان ميڠڬو دان بولن سلڤس ايت ساي كهيلڠن ديري دالم اوسها يڠ ممبيڠوڠكن اونتوق منرڠكن ميستري ايت. تيدق، سهيڠڬ جم اين، ساي سريڠ مڠهينا ديري ساي دڠنڽ.

سكارڠ، هيلڠكن راس تاكوت يڠ مڠريكن، دان سينسيتيۏيتي ساي دالم مراساي تاڠن ڬايب دالم ديري ساي امت سروڤا، دالم كڤلباڬايان مريك، كڤد يڠ ساي عالمي اڤابيلا ترباڠون دان مليهت ڬڠڬمن ڤاڬن Queequeg دليمڤركن د سكيتر ساي. تتاڤي ڤد اخيرڽ سموا ڤريستيوا مالم يڠ لالو برلاكو سچارا سدرهان، ساتو ڤرساتو، دالم رياليتي تتڤ، دان كمودين ساي هاڽ ترباريڠ هيدوڤ كڤد ڤريديقسي كوميس. كارنا والاوڤون ساي چوبا مڠهيلڠكن تاڠنڽ—ممبوك كونچي ڬڠڬم ڤڠنتين للاكيڽ—تتاڤي، تيدور سڤرتيمان دي، دي ماسيه مملوق ساي دڠن كوات، سأوله-اوله تياد اڤ-اڤ كچوالي كماتين يڠ بوليه ممبهاڬيكن كامي بردوا. ساي سكارڠ براوسها اونتوق ممبڠكيتكنڽ—“كويق!”—تتاڤي ساتو-ساتوڽ جواڤنڽ اداله ڬڠڬوان. ساي كمودين برڬوليق، ليهير ساي براس سأوله-اوله د دالم كولر كود; دان تيبا-تيبا تراس سديكيت ڬوريسن. ممبواڠ كونترڤن، ترداڤت توماهوك يڠ تيدور د سيسي بيادب، سأوله-اوله اي اداله بايي برموك ڤيساو. ڤنچوچي مولوت يڠ چوكوڤ اينده، بنر-بنر، فيكير ساي; تيدور د سيني د سبواه رومه يڠ ڤليق ڤد وقتو سياڠ، دڠن سأورڠ كانيبال دان سأورڠ توماهوك! “كويكوڬ!—اتس نام كبايكن، كويكوڬ، باڠونله!” اخيرڽ، دڠن باڽق ڬڠڬوان، دان ڤنجلسن يڠ كوات دان تيدق برڤوتوس اسا تنتڠ كتيدقڤستينڽ مملوق سأورڠ راكن للاكي دالم ڬاي ڤركهوينن ايت، ساي برجاي مڠلواركن ڬرون; دان سكارڠ، دي مناريق باليق تاڠنڽ، مڠهيلڠكن ديريڽ سڤرتي انجيڠ نيوفاوندلند هاڽ دري اير، دان دودوق د اتس كاتيل، كراس سڤرتي توڠكت ڤايوڠ، مليهت ساي، دان مڠڬوسوق ماتڽ سأوله-اوله دي تيدق مڠيڠاتي باڬايمانا ساي داتڠ ك سان، والاوڤون كسدرن يڠ ريمبون اونتوق مڠتاهوي سسواتو تنتڠ ساي ڤرلاهن-لاهن نمڤق برلاكو. سمنتارا ايت، ساي برباريڠ دڠن ديام ممڤرهاتيكنڽ، تنڤا راس شك يڠ سريوس سكارڠ، دان برڤڬڠ تڬوه اونتوق ممڤرهاتيكن سچارا سمڤيت مخلوق يڠ بڬيتو ايڠين تاهو. اڤابيلا، اخيرڽ، فيكيرنڽ سأوله-اوله تله دبوات اونتوق مڽنتوه واتق راكن تيدورڽ، دان دي منجادي، سأوله-اوله، برداماي دڠن كڽاتأن ايت; دي ملونچور ك اتس لنتاي، دان دڠن تندا دان بوڽي ترتنتو ممبريكن ساي اونتوق ممهمي بهاوا، جك اي مڽنڠكن ساي، دي اكن برڤاكاين ترلبيه دهولو دان كمودين منيڠڬلكن ساي اونتوق برڤاكاين سلڤس ايت، منيڠڬلكن سلوروه اڤرتمن كڤد ديري ساي سنديري. فيكير ساي، Queequeg، دالم كادأن، اين اداله اوۏرچر يڠ ساڠت برتمدون; تتاڤي، كبنرنڽ، اورڠ-اورڠ بيادب اين ممڤوڽاءي راس كچنتيكن يڠ سمولا جادي، كاتاكن اڤ يڠ اندا ماهوكن; اي امت مڠحيرانكن بتاڤ ڤد داسرڽ مريك سوڤن. ساي ممبريكن ڤوجيان خصوص اين كڤد Queequeg، كران دي ملاين ساي دڠن بڬيتو سوڤن دان ڤرتيمبڠن، سداڠكن ساي برساله دڠن ككاسرن يڠ ساڠت بسر; مناتڤڽ دري كاتيل، دان مليهت سموا ڬرقن تاندسڽ; اونتوق ماس ايت راس ايڠين تاهو ساي سماكين باءيق دري ڤمبياقن ساي. والاو باڬايماناڤون، سأورڠ للاكي سڤرتي Queequeg يڠ اندا تيدق ليهت ستياڤ هاري، دي دان چارا-چاراڽ اداله برنيلاي لوار بياسا بركاءيتن.

دي مولا برڤاكاين د اتس دڠن مماكاي توڤي بوروڠ ڤوڠڬوقڽ، يڠ ساڠت تيڠڬي، د ڤيڠڬير، دان كمودين—ماسيه كورڠ سلوارڽ—دي ممبورو بوتولڽ. اڤ د باوه لاڠيت يڠ دي لاكوكن، ساي تيدق بوليه ممبريتاهو، تتاڤي ڤرڬرقن ستروسڽ اداله اونتوق مڠهنچوركن ديريڽ سنديري—بوت د تاڠن، دان توڤي—د باوه كاتيل; اڤابيلا، دري ڤلباڬاي ڬڠڬوان دان ڬڠڬوان يڠ ڬانس، ساي مڽيمڤولكن بهاوا دي بكرجا كراس اونتوق ممبوات ديريڽ سنديري; والاوڤون تياد اوندڠ-اوندڠ كڤرچايأن يڠ ڤرنه ساي دڠر، ادكه مان-مان للاكي دڤرلوكن اونتوق منجادي ڤريبادي اڤابيلا مماكاي بوتولڽ. تتاڤي Queequeg، ادكه اندا مليهت، اداله مخلوق دالم ڤريڠكت ڤراليهن—بوكن اولت اتاو راما-راما. دي چوكوڤ برتمدون اونتوق ممڤاميركن كچنتيكنڽ دڠن چارا يڠ ڤاليڠ ڤليق. ڤنديديقنڽ بلوم سلساي لاڬي. بلياو اداله سأورڠ مهاسيسوا. جك دي تيدق برتمدون سديكيت، دي موڠكين تيدق اكن مڽوسهكن ديريڽ دڠن بوتول سام سكالي; تتاڤي سلڤس ايت، جك دي تيدق ماسيه سأورڠ يڠ بيادب، دي تيدق اكن ڤرنه برميمڤي اونتوق ماسوق ك باوه كاتيل اونتوق مماكايڽ. اخيرڽ، دي مونچول دڠن توڤيڽ يڠ ساڠت ترڤوتوس دان ترڤوتوس د اتس ماتڽ، دان مولا برڬرق دان برڬرق د سكيتر بيليق، سأوله-اوله، تيدق باڽق تربياسا دڠن بوتول، سڤاسڠ كوليت لمبوڽ يڠ لمبڤ دان بركدودوقن—موڠكين تيدق دبوات مڠيكوت ڤسانن جوڬ—اڬق منچوچوت دان مڽيقساڽ ڤد ڤرمولأن ڤاڬي يڠ ڤاهيت دان سجوق.

مليهت، سكارڠ، بهاوا تياد لڠسير ك تيڠكڤ، دان بهاوا جالن يڠ ساڠت سمڤيت، رومه يڠ برتنتڠن ممرينتهكن ڤمندڠن يڠ جلس ك دالم بيليق، دان ممڤرهاتيكن سماكين باڽق ڤارس يڠ تيدق سوڤن يڠ دبوات اوليه كويق، برڬرق-ڬرق دڠن سديكيت لاڬي كچوالي توڤي دان بوتولڽ; ساي مموهون كڤداڽ سبايق موڠكين، اونتوق ممڤرچڤتكن تاندسڽ سديكيت، دان تراوتماڽ اونتوق ماسوق ك دالم سلوارڽ سچڤت موڠكين. دي مماتوهي، دان كمودين تروس ممباسوه ديري. ڤد وقتو ايت ڤد وقتو ڤاڬي مان-مان نصراني اكن ممباسوه موكڽ; تتاڤي Queequeg، يڠ مڠجوتكن ساي، ممواسكن ديري دڠن ممباتسي وضوءڽ ك دادا، ڬڠڬم، دان تاڠنڽ. دي كمودين مماكاي باجو ڤيڠڬڠڽ، دان مڠمبيل سڤوتوڠ سابوڠ كراس د اتس ميجا تڠه ڤنچوچي، منچلوڤكنڽ ك دالم اير دان مولا مڠهيلڠكن موكڽ. ساي سدڠ منونتون اونتوق مليهت د مان دي مڽيمڤن ڤيساوڽ، اڤابيلا لو دان ليهت، دي مڠمبيل هارڤون دري سودوت كاتيل، ملڤسكن ستوك كايو ڤنجڠ، ممبوك ساروڠ كڤالا، مڠهيلڠكنڽ سديكيت ڤد بوتڽ، دان ملڠكه ك اتس چرمين د دينديڠ، ممولاكن ڤڠيكيس يڠ ڬيات، اتاو لبيه تڤت لاڬي مڠيكيس هارڤون ڤيڤيڽ. فيكير ساي، Queequeg، اين مڠڬوناكن ڤرالاتن ماكنن ترباءيق Rogers دڠن ڤمبالسن. سلڤس ايت ساي ترتاڽ-تاڽ اڤاكه كورڠ ڤد اوڤراسي اين اڤابيلا ساي تاهو تنتڠ اڤ بسي هالوس كڤالا هارڤون يڠ دبوات، دان بتاڤ تاجمڽ تڤي لوروس ڤنجڠ يڠ سنتياس دسيمڤن.

باقي تاندسڽ تيدق لاما لاڬي دچاڤاي، دان دي دڠن بڠڬڽ برباريس كلوار دري بيليق، بربلنجا دالم جاكيت موڽيت جوروترا يڠ هيبت، دان مماكاي هرڤونڽ سڤرتي باتون مارشال.





باب 5. ماكنن ڤاڬي.

ساي دڠن چڤت مڠيكوتيڽ، دان تورون ك دالم بيليق بار مڠهوبوڠي توان رومه يڠ ترسڽوم دڠن ساڠت مڽنڠكن. ساي تيدق مڠحرمتي راس جاهت ترهادڤڽ، والاوڤون دي تله برڬادوه دڠن ساي تيدق سديكيتڤون دالم حال راكن تيدور ساي.

نامون، كتاوا يڠ باءيق اداله ڤركارا يڠ ساڠت باءيق، دان ترلالو جارڠ ڤركارا يڠ باءيق; سماكين باڽق كاسيه سايڠڽ. جادي، جك مان-مان سأورڠ للاكي، دالم ديريڽ سنديري، ممڤوڽاءي بارڠ اونتوق ڬولوڠن يڠ باءيق كڤد سسياڤا، بيارله دي تيدق ترباليق، تتاڤي بيارله دي دڠن ڬمبيرا ممبنركن ديريڽ بربلنجا دان دبلنجاكن دڠن چارا ايت. دان للاكي يڠ ممڤوڽاءي سسواتو يڠ بوليه دكتاوا تنتڠڽ، ڤستيكن اد لبيه باڽق دالم ديري للاكي ايت درڤد يڠ موڠكين اندا فيكيركن.

“ڬروب، هو!” سكارڠ مناڠيس توان رومه، ممبوك ڤينتو، دان كامي ڤرڬي اونتوق بربوك ڤواسا.

مريك بركات بهاوا للاكي يڠ تله مليهت دنيا، دڠن ايت منجادي چوكوڤ ڽامن دالم ڤريلاكو، چوكوڤ مميليقي ديري دالم شريكت. تيدق سلالو، والاوڤون: ليديارد، ڤنومڤڠ نيو ايڠلند يڠ هيبت، دان موڠو ڤارك، يڠ سكوتلند; درڤد سموا للاكي، مريك ممڤوڽاءي كڤرچايأن يڠ ڤاليڠ سديكيت د دالم بيليق. تتاڤي موڠكين هاڽ ملواتي سيبيريا دالم ڬلڠڬڠ يڠ دتاريق اوليه انجيڠ سڤرتي يڠ دلاكوكن اوليه ليديارد، اتاو برجالن-جالن برسنديرين يڠ ڤنجڠ دڠن ڤروت كوسوڠ، د تڠه-تڠه نيڬرو افريكا، يڠ مروڤاكن جومله ڤرسمبهن مونڬو يڠ ميسكين—ڤرجالنن جنيس اين، ساي كاتاكن، موڠكين بوكنله چارا يڠ ڤاليڠ باءيق اونتوق منچاڤأي ڤموليهن سوسيال يڠ تيڠڬي. نامون، باڬي كبڽقانڽ، ڤركارا سماچم ايت بوليه دڤراوليهي د مان-مان.

ريفليكسي-ريفليكسي اين هاڽ د سيني دسببكن اوليه كادأن يڠ سلڤس كامي سموا دودوق د ميجا، دان ساي برسياڤ-سياڤ اونتوق مندڠر ببراڤ چريتا يڠ باءيق تنتڠ ڤمبوروان ڤاوس; دڠن كجوتن ساي يڠ تيدق كچيل، همڤير ستياڤ للاكي مڠكلكن ديام يڠ مندالم. دان بوكن ايت سهاج، تتاڤي مريك كليهاتن مالو. يا، د سيني اد ساتو سيت انجيڠ لاوت، راماي درڤد مريك تنڤا سديكيتڤون راس مالو تله مناءيقي ڤاوس بسر د لاوت تيڠڬي—سڤنوه اورڠ اسيڠ باڬي مريك—دان برتمڤور دڠن مريك ماتي تنڤا مڠيجڤ; دان نامون، د سيني مريك دودوق د ميجا ماكن ڤاڬي سوسيال—سمواڽ ڤڠڬالقن يڠ سام، سمواڽ سليرا يڠ سام—ممندڠ سكليليڠ دڠن مالو-مالو كڤد ساتو سام لا؞ين سأوله-اوله مريك تيدق ڤرنه كلوار دري ڤنداڠن ببراڤ ڬدوڠ كمبيڠ د كالڠن ڬونوڠ هيجاو. ڤمندڠن يڠ مناريق; بوروڠ-بوروڠ يڠ مالو اين، ڤارا ڤهلاون ڤاوس يڠ مالو-مالو اين!

تتاڤي باڬي كويكڬ—مڠاڤ، كويكڬ دودوق د سان د كالڠن مريك—د كڤالا ميجا جوڬ، اي بڬيتو كبتولن; سجوق سڤرتي ايس كريم. اونتوق ڤستيكن ساي تيدق بوليه بركات باڽق تنتڠ ڤمبياقنڽ. ڤڠاڬوم تربسرڽ تيدق داڤت ممبنركن دڠن ڤنوه راس چينتاڽ ممباواڽ ك دالم ماكنن ڤاڬي برسامڽ، دان مڠڬوناكنڽ د سان تنڤا مجليس; منچاڤأي ميجا دڠنڽ، كڤد بهاي يڠ اكن برلاكو ڤد راماي كڤالا، دان مڠهيلڠكن بيفستيق ك ارهڽ. تتاڤي ايت سوده تنتو دلاكوكن دڠن ساڠت سرونوك اوليهڽ، دان ستياڤ اورڠ تاهو بهاوا دالم اڠݢرن كبڽقان اورڠ، اونتوق ملاكوكن اڤ سهاج دڠن سرونوق اداله ملاكوكنڽ دڠن ڤنوه هاتي.

كامي تيدق اكن برچريتا تنتڠ سموا كأيستيميوأن Queequeg د سيني; باڬايمانا دي مڠيلقكن كوڤي دان رول ڤانس، دان منراڤكن ڤرهاتينڽ يڠ تيدق ترڤچه كڤد بيفستيق، يڠ دلاكوكن سچارا جارڠ. چوكوڤ، بهوا اڤابيلا ماكن ڤاڬي سلساي دي مناريق ديري سڤرتي يڠ لا؞ين ك دالم بيليق عوام، مڠهيدوڤكن ڤاڤرن توماهوكڽ، دان دودوق د سان دڠن ديام منچرنا دان مروكوق دڠن توڤيڽ يڠ تيدق داڤت دڤيسهكن، اڤابيلا ساي كلوار اونتوق برجالن-جالن.





باب 6. جالن.

جك ساي تله ترڤڠڬيل ڤد مولاڽ مليهت سأورڠ اينديۏيدو يڠ بڬيتو لوار بياسا سڤرتي Queequeg يڠ برڤوتر د كالڠن مشاركت يڠ سوڤن د سبواه بندر يڠ برتمدون، كچواين ايت تيدق لاما كمودين برڤينده اڤابيلا ساي برجالن-جالن سياڠ ڤرتام د جالن-جالن نيو بيدفورد.

د جالن راي دكت ڤلابوهن، مان-مان ڤلابوهن لاوت يڠ چوكوڤ بسر اكن سريڠ مناوركن اونتوق مليهت ڤنمڤيلن يڠ ڤاليڠ انيه دري بهاڬين اسيڠ. بهكن د جالن برودواي دان چستنوت، ڤارا ڤلاير ميديترانيا كادڠ-كادڠ اكن منوندوقكن وانيتا-وانيتا يڠ تاكوت. جالن ريڬينت تيدق تيدق دكنلي اوليه لاسكار دان ملايو; دان د بومباي، د اڤولو ڬرين، يانكي هيدوڤ سريڠ مناكوتكن ڤندودوق اصلي. تتاڤي نيو بيدفورد مڠالهكن سموا واتر ستريت دان واڤڤيڠ. د تمڤت-تمڤت يڠ تراخير دسبوتكن اين، اندا هاڽ مليهت ڤارا ڤلاير; تتاڤي د نيو بيدفورد، ڤارا ڤمبونوه سبنر برديري بربوال د سودوت-سودوت جالن; بيادب سچارا تروس-ترڠ; راماي د انتارا مريك ماسيه ممباوا تولڠ-تولڠ مريك داڬيڠ يڠ تيدق سوچي. اي ممبواتكن ڤنداڠن يڠ لبيه اسيڠ.

تتاڤي، سلاين Feegeeans، Tongatobooarrs، Erromanggoans، Pannangians، دان Brighggians، دان، سلاين سڤيسيايس ليار دري كراتن ڤاوس يڠ تيدق دڤرهاتيكن برڬرق د جالن-جالن، اندا اكن مليهت ڤمندڠن-ڤمندڠن لا؞ين يڠ ماسيه لبيه مناريق، تنتوڽ لبيه كوميس. ستياڤ ميڠڬو تيبا د بندر اين ڤلباڬاي اورڠ ۏيرمونت هيجاو دان للاكي نيو هامڤشاير، سمواڽ هاڠوس اونتوق منداڤتكن كأونتوڠن دان كڬميلڠن دالم ڤرايكنن. مريك كباڽقكنڽ مودا، بربنتوق كوات; راكن-راكن يڠ تله منوتوڤ هوتن، دان كيني برحسرت اونتوق منجاتوهكن كاڤق دان مرمڤس تومبق ڤاوس. راماي يڠ هيجاو سڤرتي ڬونوڠ هيجاو دري مان مريك داتڠ. دالم ببراڤ ڤركارا اندا اكن مميكيركنڽ تتاڤي ببراڤ جم لاماڽ. ليهتله د سان! للاكي ايت برڬرق د سودوت. دي مماكاي توڤي بوروڠ ڤوڠڬوق دان كوتق بركوليت ڬلڤ، دڤڬڠ دڠن ڤيڠڬڠ ڤلاير دان ڤيساو ساروڠ. د سيني داتڠ سأورڠ لاڬي دڠن سو'-ويستر دان جوبه بومبازين.

تياد داندي يڠ دبياقكن د بندر اكن دبنديڠكن دڠن يڠ دبياقكن د نڬارا—مقصود ساي سأورڠ داندي بومكين يڠ سام سكالي—سأورڠ راكن يڠ، ڤد هاري-هاري انجيڠ، اكن مموتوڠ دوا ايكر تانهڽ دالم ساروڠ تاڠن كوليت روسا كران تاكوت مڠهيلڠكن تاڠنڽ. سكارڠ اڤابيلا سأورڠ ڤنچوري نڬارا سڤرتي اين مڠمبيلڽ ك دالم كڤالاڽ اونتوق ممبوات ريڤوتاسي يڠ تركنل، دان مڽرتاي ڤرايكنن ڤاوس يڠ هيبت، اندا ڤرلو مليهت ڤركارا-ڤركارا كوميس يڠ دلاكوكنڽ اڤابيلا سمڤأي ك ڤلابوهن لاوت. دالم برچاكڤ تنتڠ ڤاكاين لاوتڽ، دي ممسن بوتڠ-بوتڠ لونچور ك ڤاكاين ڤيڠڬڠڽ; تالي ڤد سلوار كنۏسڽ. اه، هي-سيد يڠ ميسكين! بتاڤ ڤاهيتڽ اكن ڤچه تالي-تالي ايت دالم اڠين ريبوت ڤرتام، اڤابيلا اندا ددوروڠ، تالي-تالي، بوتڠ، دان سمواڽ، ك باوه تكق ريبوت.

تتاڤي جاڠن فيكيركن بهاوا بندر يڠ تركنل اين هاڽ ممڤوڽاءي ڤارا ڤڽرڠ، كانيبال، دان بومكين اونتوق دتونجوقكن كڤد ڤڠونجوڠڽ. تيدق سام سكالي. ماسيه نيو بيدفورد اداله تمڤت يڠ انيه. جك بوكن كران كامي ڤارا ڤنياڬ ڤاوس، تانه ايت ڤد هاري اين موڠكين اكن برادا دالم كادأن يڠ سام سڤرتي ڤنتاي لابرادور. سڤرتي يڠ اد، بهاڬين-بهاڬين نڬارا بلاكڠڽ چوكوڤ اونتوق مناكوتكن سساورڠ، مريك كليهاتن بڬيتو برتولڠ. بندر ايت سنديري موڠكين مروڤاكن تمڤت يڠ ڤاليڠ سايڠ اونتوق تيڠڬل، د سلوروه نيو ايڠلند. اي اداله تانه ميڽق، چوكوڤ بنر: تتاڤي تيدق سڤرتي كنعان; سبواه تانه، جوڬ، جاڬوڠ دان اير ڬندوم. جالن-جالن تيدق برجالن دڠن سوسو; تيدق جوڬ ڤد موسيم سمي مريك ممبوكڽ دڠن تلور سڬر. نامون، والاوڤون بڬيتو، تياد د مان-مان د سلوروه اميريك اندا اكن منموكن لبيه باڽق رومه سڤرتي ڤاتريسيا; تامن دان تامن لبيه ميواه، درڤد د نيو بيدفورد. دري مان مريك داتڠ؟ باڬايمانا دتانم د اتس سكوريا نڬارا يڠ ڤرنه روسق اين؟

ڤرڬيله دان ليهتله هرڤون لامبڠ بسي د سكيتر رومه بسر يڠ تيڠڬي د سان، دان ڤرتاڽان اندا اكن دجواب. يا؛ سموا رومه براني دان تامن بوڠا اين براصل دري لاوتن اتلانتيق، ڤاسيفيك، دان اينديا. ساتو دان سموا، مريك دڤوكول دان دسرتاكن ك اتس ك سيني دري داسر لاوت. بوليهكه هير الايك ساندر ملاكوكن ڤريستاسي سڤرتي ايت؟

د نيو بيدفورد، باڤ-باڤ، كات مريك، ممبريكن ڤاوس اونتوق هديه كڤد انق-انق ڤرمڤوان مريك، دان ممبهاڬيكن انق-انق ساودارا ڤرمڤوان مريك دڠن ببراڤ ڤورڤويس سڤوتوڠ. اندا هاروس ڤرڬي ك نيو بيدفورد اونتوق مليهت مجليس ڤركهوينن يڠ چنتيق; كران، كات مريك، مريك ممڤوڽاءي ريسيرۏوار ميڽق د ستياڤ رومه، دان ستياڤ مالم دڠن تيدق برتڠڬوڠجواب ممباكر ڤنجڠ مريك دالم ليلين سڤيرماچيتي.

ڤد موسيم ڤانس، بندر اين مانيس دليهت; ڤنوه دڠن ماڤل يڠ هالوس—جالن-جالن ڤنجڠ هيجاو دان امس. دان ڤد بولن اوڬوس، تيڠڬي د اودارا، ڤوكوق-ڤوكوق كود يڠ اينده دان برليمڤه، دري سڬي كاندلبرا، مناوركن كڤد ڤارا ڤلڠڬن كرون تڬق يڠ منيڠكت دري بوڠا-بوڠا يڠ بركومڤول. جادي سني يڠ مها كواس; يڠ د باڽق دايره نيو بيدفورد تله مڠيندوكسي تراس-تراس بوڠا يڠ چره د اتس باتو-باتو سمڤه يڠ تانمن يڠ دبواڠ ك اره هاري اخير ڤنچيڤتأن.

دان وانيتا-وانيتا نيو بيدفورد، مريك بركمبڠ سڤرتي بوڠا ماور ميره مريك سنديري. تتاڤي بوڠا ماور هاڽ ممبوڠا ڤد موسيم ڤانس; سداڠكن كارنا هالوس د ڤيڤي مريك اداله ككال سڤرتي چهاي ماتهاري د لاڠيت كتوجوه. د تمڤت لا؞ين مننديڠي بوڠا مريك ايت، اندا تيدق بوليه، كچوالي د سالم، د مان مريك ممبريتاهو ساي بهاوا ڬاديس-ڬاديس مودا مڠهيروڤ كستوري سڤرتي ايت، كسايڠن ڤلاير مريك مڠهيدوڤكن مريك ببراڤ باتو دري ڤنتاي، سأوله-اوله مريك مڠهمڤيري مولوكا يڠ برباو بوكنڽ ڤاسير ڤوريتاني.





باب 7. ڬريجا.

د نيو بيدفورد يڠ سام ترداڤت ڬريجا ڤاوس، دان سديكيت سهاج ڤارا ڤنچوري يڠ موود، يڠ تيدق لاما لاڬي اكن ڤرڬي ك لاوتن هيندي اتاو ڤاسيفيك، يڠ ڬاڬل ملاكوكن لاوتن ڤد هاري احد ك تمڤت ايت. ساي يقين بهاوا ساي تيدق ملاكوكنڽ.

ڤولڠ دري ڤرجالنن ڤاڬي ڤرتام ساي، ساي سكالي لاڬي برڬرق ك اتس توڬسن ايستيميوا اين. لاڠيت تله بروبه دري يڠ جلس، سجوق يڠ چره، كڤد هوجن لبت دان كابوت يڠ ممندو. ممبوڠكوس ديري دالم جاكيت بربولو-بولو دري كاءين يڠ دڤڠڬيل كوليت بوروڠ، ساي برجواڠ اونتوق مننتڠ ريبوت يڠ تڬس. ماسوق، ساي منموكن جماعه كچيل يڠ برسيبار دري ڤارا ڤلاير، دان استري دان جندا ڤارا ڤلاير. سوڽي يڠ ترڤوتوس بركواس، هاڽ دڤچهكن ڤد كتيك-كتيك ترتنتو اوليه جريتن ريبوت. ستياڤ ڤڽمبه يڠ ديام سأوله-اوله سڠاج دودوق ترڤيسه درڤد يڠ لا؞ين، سأوله-اوله ستياڤ كسديهن يڠ ديام اداله ترڤيسه دان تيدق بوليه دڤركنلكن. ڤنديتا ايت بلوم تيبا لاڬي; دان د سان ڤولاو-ڤولاو للاكي دان ڤرمڤوان يڠ سوڽي اين دودوق دڠن تڬس ممڤرهاتيكن ببراڤ لوح مرمر، دڠن سمڤادن هيتم، يڠ دڤاسڠ د دينديڠ د كدوا-دوا بله ميمبر. تيڬ درڤد مريك منجالنكن سسواتو سڤرتي برايكوت، تتاڤي ساي تيدق برڤورا-ڤورا ممتيق:—

دسوچيكن اونتوق ايڠتن جوهن تالبوت، يڠ، ڤد اوسيا لاڤن بلس تاهون، هيلڠ د لاوت، دكت ڤولاو ديسولاسي، د لوار ڤتاڬونيا، 1 نوۏيمبر 1836. لوح اين دديريكن اونتوق ايڠتنڽ اوليه كاكقڽ.

دسوچيكن كڤد ايڠتن روبرت لوڠ، ويلليس ايللي، ناتان كولمن، ولتر كاني، سيت ميسي، دان ساموايل ڬليڬ، ممبنتوق ساله ساتو كرجا ڤراهو كاڤل ايليزا يڠ دتاريك كلوار دري ڤنداڠن اوليه ڤاوس، ڤد ڤاسير ڤاسير د ديسيمبر 31 st ، 1839. مرمر اين دلتقكن د سيني اوليه راكن-راكن كاڤل مريك يڠ ماسيه هيدوڤ.

دسوچيكن اونتوق ايڠتن المرحوم كڤتن ايزيكيل هردي، يڠ د دالم بوسور ڤراهوڽ دبونوه اوليه ڤاوس سڤيرم د ڤنتاي جڤون، 3 اوڬوس 1833. لوح اين دديريكن اونتوق ايڠتنڽ اوليه جنداڽ.

مڠهيلڠكن هوجن لبت دري توڤي دان جاكيت برڬلڤ ايس، ساي دودوق دكت ڤينتو، دان برڤاليڠ ك اره سيسي ترڤڠڬيل مليهت كويق دكت ساي. ترججس اوليه كسوڠڬوهن ڤمندڠن، ترداڤت ڤنداڠن يڠ مڠجوتكن دڠن راس ايڠين تاهو يڠ تيدق ڤرچاي د موكڽ. بيادب اين اداله ساتو-ساتوڽ اورڠ يڠ حاضر يڠ سأوله-اوله ممڤرهاتيكن ڤينتو ماسوق ساي; كران دي اداله ساتو-ساتوڽ يڠ تيدق بوليه ممباچ، دان، اوليه ايت، تيدق ممباچ توليسن-توليسن يڠ سجوق ايت د دينديڠ. سام اد مان-مان كرابت ڤارا ڤلاوت يڠ نامڽ مونچول د سان كيني برادا د كالڠن جماعه، ساي تيدق تاهو; تتاڤي بڬيتو باڽق كمالڠن يڠ تيدق دريكودكن دالم ڤرايكنن، دان بڬيتو جلس ببراڤ وانيتا يڠ حاضر مماكاي واجه جك تيدق ڤرهياسن ببراڤ كسديهن يڠ تيدق برهنتي، سهيڠڬاكن ساي مراسا يقين بهاوا د سيني د هادڤن ساي تله دكومڤولكن مريك، يڠ دالم هاتي مريك يڠ تيدق سمبوه ڤمندڠن لوح-لوح يڠ سوكر ايت دڠن سيمڤاتي مڽببكن لوك-لوك لاما برداراه سمولا.

اوه! كامو يڠ اورڠ ماتيڽ ترباريڠ د باوه رومڤوت هيجاو; سياڤا يڠ برديري د كالڠن بوڠا بوليه بركات—د سيني، د سيني ترلتق كسايڠنكو; كامو تيدق تاهو كهنچورن يڠ برلاكو د دالم دادا سڤرتي اين. بتاڤ ڤاهيتڽ كوسوڠ دالم ڬلڠڬڠ برباتس هيتم ايت يڠ تيدق منوتوڤي ابو! بتاڤ ڤوتوس اساڽ دالم توليسن-توليسن يڠ تيدق برڬرق ايت! اڤ ككوسوڠن يڠ مماتيكن دان ككافيرن يڠ تيدق دأرهكن دالم ڬاريس-ڬاريس يڠ سأوله-اوله مڠهنچوركن سموا ايمان، دان منولق كهيدوڤن سمولا كڤد مخلوق يڠ تله هيلڠ تنڤا تمڤت تنڤا قبور. موڠكين جوڬ لوح-لوح ايت برديري د دالم ڬوا ڬاجه سڤرتي د سيني.

دالم اڤ ڤنچاتتن مخلوق هيدوڤ، اورڠ ماتي مأنسي دماسوقكن; مڠاڤ ڤريبهاس اونيۏرسل مڠاتكن تنتڠ مريك، بهاوا مريك تيدق منچريتاكن چريتا، والاوڤون مڠاندوڠي لبيه باڽق رهسيا درڤد ڬودوين ساندس; باڬايمانا ڤولا كڤد نامڽ يڠ سمالم برڤينده ك دنيا يڠ لا؞ين، كامي مڠمبيل ڤركاتان يڠ بڬيتو سيڬنيفيكن دان كافير، دان نامون تيدق ممبريكنڽ حق، جك دي هاڽ برلڤس ك هينديا ترڤنچيل د بومي هيدوڤ اين; مڠاڤ شريكت اينسورنس هيدوڤ ممباير ڤمبواڠن كماتين اتس اورڠ يڠ تيدق اكن ماتي; دالم اڤ يڠ ابادي، لومڤوه يڠ تيدق مڠهيلڠكن، دان ترنس يڠ مماتيكن، تيدق برهارڤ، نامون ترلتقڽ آدم انتيق يڠ تله منيڠڬل دنيا انم ڤولوه ابد يڠ لالو; باڬايمانا كيت ماسيه منولق اونتوق دهيبوركن كران مريك يڠ كيت ككلكن تيڠڬل دالم كبهاڬيءن يڠ تيدق تركاتاكن; مڠاڤ سموا اورڠ هيدوڤ بڬيتو براوسها اونتوق منديامكن سموا اورڠ ماتي; اوليه ايت، هاڽ ديسكريڤسي تنتڠ ڤڠتوقن قبور اكن مڠجوتكن سلوروه كوتا. سموا ڤركارا اين تيدق تنڤا معنى.

تتاڤي ايمان، سڤرتي سأيكور بوروڠ ڤوڠڬوق، ممبري ماكن د كالڠن قبور، دان بهكن دري كراڬوان-كراڬوان يڠ ماتي اين دي مڠومڤولكن هارڤنڽ يڠ ڤاليڠ ڤنتيڠ.

اي تيدق ڤرلو دچريتاكن، دڠن ڤراسان اڤ، ڤد مالم ڤرجالنن نانتوكيت، ساي مليهت لوح-لوح مرمر ايت، دان اوليه چهاي كجم ڤد هاري يڠ ڬلڤ دان مڽديهكن ايت ممباچ نصيب ڤارا ڤمبورو ڤاوس يڠ تله ڤرڬي سبلوم ساي. يا، اسماعيل، نصيب يڠ سام موڠكين اكن ترجادي ڤدمو. تتاڤي اڤاته لاڬي ساي منجادي ڬمبيرا لاڬي. ايندوك سا يڠ مڽرونوقكن اونتوق مناءيقي كاڤل، ڤلواڠ يڠ باءيق اونتوق ڤروموسي، نمڤقڽ—اي، ڤراهو داڤور اكن منجاديكن ساي سأورڠ يڠ ابادي اوليه برۏيت. يا، اد كماتين دالم ڤرنياڬان ڤمبوروان ڤاوس اين—ڤمبونوهن كاچاو بيلاو يڠ ڤانتس تنڤا كات-كات دري سأورڠ للاكي ك دالم ككالهن. تتاڤي باڬايمانا كمودين؟ ساي فيكير كيت تله ساله فهم دالم حال هيدوڤ دان ماتي اين. ساي برڤنداڤت بهاوا اڤ يڠ مريك ڬلر بايڠ-بايڠ ساي د سيني د بومي اداله باهن ساي يڠ سبنر. ساي برڤنداڤت بهاوا دالم مليهت ڤركارا-ڤركارا روحاني، كيت ترلالو باڽق سڤرتي تيرم يڠ ممڤرهاتيكن ماتهاري ملالوءي اير، دان برڤنداڤت بهاوا اير يڠ تبل اداله اودارا يڠ ڤاليڠ تيڤيس. ساي فيكير توبوه ساي هاڽله لييس دري ديري ساي يڠ لبيه باءيق. سبنرڽ امبيل توبوهكو يڠ اكن، امبيلله اكو بركات، ايت بوكن اكو. دان اوليه ايت تيڬ سوارا ڬمبيرا اونتوق نانتوكيت; دان داتڠ ڤراهو داڤور دان بادن داڤور اڤابيلا مريك ماهو، اونتوق مناهن جيوا ساي، جوۏ سنديري تيدق بوليه.





باب 8. منبر.

ساي تيدق دودوق لاما سبلوم سأورڠ للاكي يڠ بركقواتن ترتنتو ماسوق; سڬرا اڤابيلا ڤينتو يڠ دڤوكول اوليه ريبوت تربڠ كمبالي اڤابيلا مڠاكويڽ، ڤنداڠن ڤانتس ترهادڤڽ اوليه سلوروه جماعه، چوكوڤ ممبوقتيكن بهاوا للاكي توا يڠ باءيق اين اداله ڤنديتا. يا، ايت اداله باڤ ماڤل يڠ تركنل، دڤڠڬيل اوليه ڤارا ڤنياڬ ڤاوس، د انتارا مريك دي اداله ڤيليهن يڠ ساڠت بسر. بلياو ڤرنه منجادي سأورڠ ڤلاير دان سأورڠ ڤڽرڠ ڤد ماس موداڽ، تتاڤي سلاما برتاهون-تاهون تله مندديكاسيكن هيدوڤڽ كڤد كمنترين. ڤد ماس يڠ ساي سكارڠ منوليس، ايه ماڤل برادا دالم موسيم سجوق يڠ تاهن ڤد اوسيا توا يڠ صيحت; اوسيا توا يڠ سأوله-اوله برڬابوڠ منجادي ڤمودا يڠ بربوڠا كدوا، كران د انتارا سموا ڤچهن كدوتنڽ، ترداڤت ڤنراڠن ريڠن ترتنتو دري بوڠا يڠ بارو بركمبڠ—هيجاو موسيم سمي يڠ مڠينتاي والاوڤون د باوه ثلجي فيبرواري. تياد سسياڤا يڠ تله مندڠر سجارهڽ سبلوم اين، بوليه اونتوق كالي ڤرتام مليهت باڤ ماڤل تنڤا مينت يڠ ڤاليڠ تيڠڬي، كارنا ترداڤت كچنتيكن كڤرچايأن ترتنتو يڠ ترچنتوم تنتڠڽ، يڠ بوليه دڤرتيكايكن كڤد كهيدوڤن ماريتيم يڠ برڤڠالمن يڠ تله دڤيمڤينڽ. اڤابيلا دي ماسوق، ساي ڤرهاتيكن بهاوا دي تيدق ممباوا ڤايوڠ، دان تنتوڽ تيدق داتڠ دالم كريتاڽ، كارنا توڤي ترڤولينڽ تورون دڠن هوجن لبت يڠ منچايركن، دان جاكيت كاءين جوروتراڽ يڠ هيبت سأوله-اوله همڤير مڽرڠڽ ك لنتاي دڠن برت اير يڠ تله دسراڤڽ. نامون، توڤي دان كوليت دان كاسوت اتس دكلواركن ساتو ڤرساتو، دان دڬنتوڠ د سديكيت رواڠ د سودوت يڠ برسبلهن; اڤابيلا، برڤاكاين سوت يڠ لايق، دي دڠن ديام-ديام مڠهمڤيري ميمبر.

سڤرتي كبڽقان ميمبر لاما، اي اداله يڠ ساڠت تيڠڬي، دان كارنا تڠڬ بياسا ك تيڠڬي يڠ بڬيتو اكن، دڠن سودوتڽ يڠ ڤنجڠ دڠن لنتاي، مڠونتركن دڠن سريوس كاوسن ڬريجا يڠ سوده كچيل، ارسايتق، نمڤقڽ، تله برتيندق اتس ڤتونجوق باڤ ماڤل، دان مڽلساءيكن ميمبر تنڤا تڠڬ، مڠݢنتيكن تڠڬ يڠ تڬق، سڤرتي يڠ دڬوناكن د سيسي كاڤل دري ڤراهو د لاوت. استري سأورڠ كڤتن ڤنچوري ڤاوس تله ممبريكن ڬريجا دڠن سڤاسڠ تالي-تالي ورڬ ميره يڠ چنتيق اونتوق تڠڬ اين، يڠ، دڠن كڤالاڽ يڠ چنتيق، دان دچلوڤ دڠن ورنا ماهوڬني، سلوروه كونتريۏن، مڠمبيل كيرا جنيس ڬريجا ايت، سام سكالي تيدق نمڤق دالم راس يڠ بوروق. برهنتي سكجڤ د كاكي تڠڬ، دان دڠن كدوا تاڠنڽ ممڬڠ تومبول هياسن تالي-تالي، ايه ماڤل مليهت ك اتس، دان كمودين دڠن كچكڤن يڠ بنر-بنر سأكن-اكن ڤلاير تتاڤي ماسيه ڤنوه ڤڠحرمتن، تاڠن د اتس تاڠن، مناءيقي تڠڬ سأوله-اوله مناءيقي ڤونچق اوتام كاڤلڽ.

بهاڬين-بهاڬين تڬق تڠڬ سيسي اين، سڤرتي يڠ بياساڽ برلاكو دڠن يڠ برڬرق، اداله درڤد تالي يڠ دڤربوات درڤد كاءين، هاڽ بولت-بولتڽ درڤد كايو، سهيڠڬ ڤد ستياڤ لڠكه ترداڤت ساتو سامبوڠن. ڤد ڤنداڠن ڤرتام ساي ك ميمبر، اي تيدق ترلڤس درڤد ساي بهاوا والاو باڬايماناڤون سسواي اونتوق كاڤل، سامبوڠن اين ڤد اينسيدن سكارڠ نمڤقڽ تيدق ڤرلو. كران ساي تيدق برسديا اونتوق مليهت ايه ماڤل سلڤس منچاڤأي كڤنجڠن، ڤرلاهن-لاهن برڤوسيڠ، دان ممبوڠكوق د اتس ميمبر، سڠاج مڽرڠ تڠڬ لڠكه دمي لڠكه، سهيڠڬ سلوروهڽ دسيمڤن د دالم، منيڠڬلكنڽ تيدق داڤت دڤوكول د كويبك كچيلڽ.
