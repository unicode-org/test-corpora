مڠناءي ڤڬاواي-ڤڬاواي كاڤل ڤاوس، اين نمڤقڽ سباڬاي تمڤت يڠ باءيق اونتوق منمڤتكن سديكيت كأيستيميوأن دوميستيك د اتس كاڤل، يڠ تيمبول دري كوجودن كلس ڤڬاواي هرڤون، كلس يڠ تيدق دكنلي تنتو ساج د مان-مان ماريتيم يڠ لا؞ين سلاين ارماد ڤاوس.

سكارڠ، ڤربيذأن بسر يڠ دبوات انتارا ڤڬاواي دان للاكي د لاوت، اداله اين—يڠ ڤرتام هيدوڤ د بلاكڠ، يڠ تراخير ك هادڤن. اوليه ايت، دالم كاڤل ڤاوس دان ڤداڬڠ سام اد، ڤاسڠن ايت ممڤوڽاءي كواترس مريك دڠن كڤتن; دان بڬيتو جوڬ، د كبڽقان ڤارا ڤمبورو ڤاوس اميريك، ڤارا ڤمبورو هارڤون دتمڤتكن د بهاڬين سلڤس كاڤل. ارتيڽ، مريك مڠمبيل ماكنن مريك د كابين كڤتن، دان تيدور د تمڤت يڠ سچارا تيدق لڠسوڠ بركومونيكاسي دڠنڽ.

والاوڤون تيمڤوه ڤنجڠ ڤلايرن ڤمڠكين ڤاوس سلاتن ﴿سجاءوه اين يڠ ڤاليڠ لاما درڤد سموا ڤلايرن سكارڠ اتاو ڤرنه دبوات اوليه مأنسي﴾، بهاي-بهاي يڠ انيه دريڽ، دان مشاركت كڤنتيڠن يڠ برلاكو د كالڠن سبواه شريكت، يڠ سمواڽ، تيڠڬي اتاو رنده، برڬنتوڠ اونتوق كأونتوڠن مريك، بوكن ڤد اوڤه تتڤ، تتاڤي ڤد نصيب برسام مريك، برسام دڠن كڤرچايأن، كچكڤن دان كرجا كراس مريك; والاوڤون سموا ڤركارا اين دالم ببراڤ كيس چندروڠ ملاهيركن ديسيڤلين يڠ كورڠ تڬس بربنديڠ ڤد ڤداڬڠ عمومڽ; نامون، تيدق كيرا بتاڤ باڽقڽ سڤرتي كلوارڬ لاما ميسوڤوتاميا، ڤارا ڤمبورو ڤاوس اين موڠكين، دالم ببراڤ كادأن ڤريميتيف، هيدوڤ برسام; اونتوق سموا ايت، لواران يڠ تڤت، سكوراڠ-كورڠڽ، دري كوارتر-ديك جارڠ دڤچهكن سچارا ماتيريال، دان دالم كادأن تيدق دبواڠ. ميمڠ، باڽق كاڤل-كاڤل نانتوكيت يڠ مان اندا اكن مليهت كڤتنڽ برڤراڠ د ڬلڠڬڠڽ دڠن كڬميلڠن يڠ تيدق داڤت دلالو؞ي اوليه مان-مان تنترا لاءوت; تيدق، ممڤراوليهي همڤير سام باڽق ڤڠحرمتن لواران سأوله-اوله دي مماكاي اوڠو ايمڤاير، دان بوكن كاءين ڤيلوت يڠ ڤاليڠ روسق.

دان والاوڤون درڤد سموا للاكي، كڤتن ڤقود يڠ موود اداله يڠ ڤاليڠ كورڠ دبريكن كڤد انداين يڠ ڤاليڠ ريڠن ايت; دان والاوڤون ساتو-ساتوڽ ڤڠحرمتن يڠ ڤرنه دڤرلوكنڽ، اداله كطاعتن يڠ ترسيرت دان سڬرا; والاوڤون دي تيدق ممرلوكن سأورڠ ڤون اونتوق مڠلواركن كاسوت دري كاكيڽ سبلوم ملڠكه ك اتس ڬلڠڬڠ سوكو; دان والاوڤون اد ماس-ماس اڤابيلا، دسببكن كادأن-كادأن يڠ انيه يڠ بركاءيتن دڠن ڤريستيوا-ڤريستيوا سلڤس اين اونتوق دڤرجلسكن، بلياو مڠهوبوڠيڽ دڠن اصطلاح-اصطلاح يڠ لوار بياسا، سام اد دري سڬي ڤنوه ڤنوه اتاو دالم terrorem ، اتاو سباڬايڽ; نامون كڤتن اهب ڤون تيدق سام سكالي تيدق ممڤرهاتيكن بنتوق دان ڤڠڬوناان اوتام لاوت.

جوڬ، موڠكين، اي اكن ڬاڬل اونتوق دڤرهاتيكن ڤد اخيرڽ، بهاوا د بلاكڠ بنتوق دان ڤڠڬوناان ترسبوت، سأوله-اوله، دي تركادڠ منوتوڤ ديريڽ; سچارا تيدق سڠاج مڠڬوناكنڽ اونتوق توجوان-توجوان يڠ لاءين دان لبيه ڤريبادي درڤد يڠ صح دتوجوكن اونتوق دڤرتاهنكن. كسلطانن ترتنتو اوتقڽ، يڠ سباليقڽ دالم تاهڤ يڠ باءيق تتڤ تيدق دڽاتاكن; ملالوءي بنتوق-بنتوق ترسبوت، سلطانيسمى يڠ سام منجادي اينكارن دالم كراجأن ديكتاتور يڠ تيدق داڤت دڤرتاهنكن. اونتوق منجادي كأوڠڬولن اينتليكتوال سأورڠ للاكي اڤ يڠ اي ماهوكن، اي تيدق بوليه ڤرنه مڠڠڬڤ ككواسأن ڤركتيكل، يڠ ترداڤت اتس للاكي-للاكي يڠ لا؞ين، تنڤا بنتوان سسواتو جنيس سني لواران دان ڤنراڤن، سنتياس، دالم ديري مريك سنديري، كورڠ لبيه رنده دان اساس. اينله، يڠ سلاماڽ منجاڬ ڤوترا-ڤوترا توهن يڠ سبنر دري ايمڤاير دري ڤنچروبوهن دنيا; دان منيڠڬلكن كحرمتن ترتيڠڬي يڠ بوليه دبريكن اوليه اودارا اين، كڤد للاكي-للاكي يڠ منجادي لبيه تركنال ملالوءي ككوراڠن مريك يڠ تيدق ترهيڠڬ كڤد ڤيليهن يڠ ترسمبوڽي دري سڬلينتير اينرت الهي، درڤد ملالوءي كأوڠڬولن مريك يڠ تيدق دراڬوكن اتس تاهڤ ماتي مشاركت. كباءيقن يڠ بڬيتو بسر مڠهالڠ دالم ڤركارا-ڤركارا كچيل اين اڤابيلا كڤرچايأن ڤوليتيك يڠ ملمڤاو ملابوركنڽ، سهيڠڬ دالم ببراڤ كادأن دراج بهكن كڤد كبودوهن بودوه مريك تله ممبريكن ڤوتينسي. تتاڤي اڤابيلا، سڤرتي دالم كيس نيكولاوس زار، مهكوتا برڬلڠ ايمڤاير ڬيوڬرافي مڠليليڠي اوتق ايمڤاير; كمودين، ڬولوڠن ڤليبيا برڬنتوڠ د هادڤن ڤموستن يڠ هيبت. تيدق جوڬ، دراماتيس تراڬيس يڠ اكن مڠݢمبركن كتيدقستابيلن فاناتي دالم ڤڽاڤو ڤنوه دان ڬرقن لڠسوڠ، اكن ڤرنه ملوڤاكن ساتو ڤتونجوق، يڠ سچارا تيدق لڠسوڠ بڬيتو ڤنتيڠ دالم سنيڽ، سڤرتي يڠ سكارڠ دسبوتكن.

تتاڤي اهب، كڤتن ساي، ماسيه برڬرق د هادڤن ساي دالم سموا كڬمبيراان دان كڬمبيراان نانتوكيتڽ; دان دالم ايڤيسود اين يڠ مڽنتوه مهاراج دان راج، ساي تيدق بوليه مڽمبوڽيكن بهاوا ساي هاڽ اد كاءيتن دڠن ڤمبورو ڤاوس توا يڠ ميسكين سڤرتيڽ; دان، اوليه ايت، سموا ڤرهياسن دان ڤرومهن يڠ هيبت لوار دتولق كڤد ساي. اوه، اهب! اڤ يڠ اكن منجادي هيبت دالم ديريمو، اي هاروس ڤرلو دڤوتوڠ دري لاڠيت، دان دسلڠڬاراكن اونتوق د دالم، دان دڤاميركن د اودارا يڠ تيدق برتوبوه!





باب 34. ميجا كابين.

سوده تڠه هاري; دان دوڬ-بوي، ڤڠوروس، منوندوقكن موكڽ يڠ ڤوتيه دري كابين-سكوتل، مڠومومكن ماكن مالم كڤد توان دان توانڽ; يڠ، دودوق د دالم ڤراهو سوكو لي، بارو سهاج مڠمبيل ڤمرهاتين ماتهاري; دان كيني دڠن بيسو مڠيرا ڬاريس لينتڠ ڤد لوح يڠ لينتڠ دان بربنتوق ميدل، يڠ دسيمڤن اونتوق توجوان هارين ايت د بهاڬين اتس كاكي ڬاديڠڽ. دري كتيدقڤرهاتينڽ يڠ لڠكڤ كڤد بريتا، اندا اكن برڤنداڤت بهاوا اهب يڠ موود تيدق مندڠر كات-كات رندهڽ. تتاڤي سكارڠ، منڠكڤ كافن ميزن، دي مڠايوه ديريڽ ك ڬلڠڬڠ، دان دڠن سوارا يڠ رات دان تيدق ڬمبيرا، بركات، “ماكن مالم، توان ستاربق،“ مڠهيلڠ ك دالم كابين.

اڤابيلا ڬلومبڠ تراخير لڠكه سلطانڽ تله ماتي، دان ستاربق، امير ڤرتام، ممڤوڽاءي سڬالا السن اونتوق مڠڠڬڤ بهاوا دي تله دودوق، مك ستاربق ترباڠون دري كتنڠنڽ، مڠمبيل ببراڤ ڤولڠن د سڤنجڠ ڤاڤن، دان، سلڤس مڠينتيڤ ك دالم بيناكل، بركات، دڠن سديكيت ڤنوتوڤن كسنڠن، “ماكن مالم، تورون، دان تورون. امير كدوا برڬمبيرا تنتڠ ڤڽليويڠن سكتيكا، دان كمودين مڠهيلڠكن سديكيت ڤڠڬنتي اوتام، اونتوق مليهت سام اد اي اكن باءيق-باءيق سهاج دڠن تالي ڤنتيڠ ايت، دي جوڬ مڠمبيل ببن لاما، دان دڠن “ماكن مالم، توان فلاسك،“ يڠ ڤانتس، مڠيكوتي ڤنداهولوڽ.

تتاڤي امير كتيڬ، كيني مليهت ديريڽ سنديري د اتس ڬلڠڬڠ سوكو، نمڤقڽ براس ريڠن درڤد ببراڤ ڤڠاواسن يڠ ايڠين تاهو; كران، منوندوقكن سڬالا جنيس ڤڠيجڤن تاهو ك ڤلباڬاي اره، دان منندڠ كاسوتڽ، دي منيمبق ك دالم ڬلومبڠ تندوق يڠ تاجم تتاڤي تيدق بربوڽي تڤت د اتس كڤالا ڬرند ترك; دان كمودين، دڠن چكڤ يڠ چكڤ، ملونتركن توڤيڽ ك اتس ميزنتوڤ اونتوق راك، دي تورون برڬولق سجاوه سكوراڠ-كورڠڽ اڤابيلا دي تتڤ كليهاتن دري ڬلڠڬڠ، ممبولتكن سموا ڤراسمين يڠ لا؞ين، دڠن ممباوا بلاكڠ دڠن موزيک. تتاڤي سبلوم ملڠكه ك ڤينتو كابين د باوه، دي برهنتي، مڠهانتر موك بارو سڤنوهڽ، دان، كمودين، فلاسك كچيل يڠ بيبس دان هيلير مماسوقكي كحاضيرن راج اهاب، دالم واتق ابجيچتوس، اتاو همبا.

اي تيدق لاما سلڤس اوروسن ڤاڤرن، بهوا ڤد سواتو ڤاڬي تيدق لاما سلڤس بربوك ڤواسا، اهب، سڤرتي بياساڽ، مناءيقي ڬلڠڬڠ كابين ك ڬلڠڬڠ. د سان كبڽقان كڤتن لاوت بياساڽ برجالن ڤد جم ايت، سباڬايمان توان-توان نڬارا، سلڤس ماكنن يڠ سام، مڠمبيل ببراڤ ڬيليرن د تامن.

تيدق لاما كمودين، لڠكهڽ يڠ ستابيل دان برورنا ڬاديڠ ددڠر، اڤابيلا دي برجالن-جالن د اتس ڤاڤن-ڤاڤن يڠ بڬيتو بياسا دڠن لڠكه-لڠكهڽ، سهيڠڬ سمواڽ برلوبڠ، سڤرتي باتو-باتو ڬيولوڬيس، دڠن تندا-تندا ڤليق ڤرجالننڽ. ادكه اندا جوڬ مليهت دڠن تڬس، ڤد كڤالا يڠ برتولڠ روسوق دان برلوبڠ ايت; د سان جوڬ، اندا اكن مليهت تاڤق كاكي يڠ ماسيه لبيه اسيڠ—تاڤق كاكي ڤميكيرنڽ يڠ تيدق تيدور دان سلالو برڬرق.

تتاڤي ڤد كسمڤتن يڠ دڤرسؤالكن، ليكوق-ليكوق ايت كليهاتن لبيه دالم، بهكن لڠكهڽ يڠ ڬرون ڤد ڤاڬي ايت منيڠڬلكن تندا يڠ لبيه دالم. دان، بڬيتو ڤنوه دڠن ڤميكيرنڽ اداله اهب، سهيڠڬ ڤد ستياڤ ڤولڠن سراڬم يڠ دبواتڽ، سكارڠ د تياڠ اوتام دان سكارڠ د بينكل، اندا همڤير بوليه مليهت ڤميكيرن ايت برتوكر دالم ديريڽ سماس دي برڤوسيڠ، دان برڬرق دالم ديريڽ سماس دي برڬرق; بڬيتو سڤنوهڽ مميليقيڽ، ميمڠ، سهيڠڬ سمواڽ تتاڤي دسڠكا اچوان دالمن ستياڤ ڤرڬرقن لوار.

“ادكه مننداكن دي، فلاسك؟” بيسيق ستوب; “انق بوروڠ يڠ اد دالم ديريڽ مڠهيلڠكن چڠكرڠ. 'تويل اكن كلوار تيدق لاما لاڬي.”

جم-جم برلالو;—اهاب كيني ترتوتوڤ د دالم كابينڽ; انون، برجالن د اتس ڬلڠڬڠ، دڠن كتيدقڤواسن هاتي يڠ سام اينتنس دالم اسڤيكڽ.

اي همڤير ڤنوتوڤن هاري. تيبا-تيبا دي برهنتي د ڤينتو ڤاڬر، دان مماسوقكن كاكي تولڠڽ ك دالم لوبڠ اوڬر د سان، دان دڠن ساتو تاڠن ممڬڠ كافن، دي ممرينتهكن ستاربكس مڠهانتر سموا اورڠ ك بلاكڠ.

"انچئ!" كات ڤاسڠن ايت، ترڤڠڬيل دڠن ڤرينته يڠ جارڠ اتاو تيدق ڤرنه دبريكن د اتس كاڤل كچوالي دالم ببراڤ كيس لوار بياسا.

“هانتر سموا اورڠ ك بلاكڠ،“ اولڠ اهب. “كڤالا تياڠ، اد! تورون!”

اڤابيلا سلوروه شريكت كاڤل بركومڤول، دان دڠن موك يڠ ايڠين تاهو دان تيدق سڤنوهڽ تيدق ريساو، ممڤرهاتيكنڽ، كارنا دي تيدق سام دڠن اوفوك چواچ اڤابيلا ريبوت اكن تيمبول، اهب، ستله دڠن ڤانتس مليهت ك اتس بينتيڠ، دان كمودين مڠهيلڠكن ماتڽ ك كالڠن كرجا، برمولا دري ڤنداڠنڽ; دان سأوله-اوله تياد سأورڠ ڤون جيوا يڠ برادا د سكيترڽ، منروسكن ڤولڠڽ يڠ برت د اتس ڬلڠڬڠ. دڠن كڤالا يڠ بڠكيت دان توڤي يڠ سڤاروه ترڤوتوس، دي تروس برجالن، تيدق مڠيرا بيسيكن يڠ مڠجوتكن د كالڠن للاكي-للاكي ايت; سهيڠڬ ستوب دڠن برهاتي-هاتي بربيسيق كڤد فلاسك، بهاوا اهب ڤستي تله ممڠڬيل مريك ك سان اونتوق توجوان مڽقسيكن ڤريستاسي ڤجالن كاكي. تتاڤي اين تيدق برتاهن لاما. برهنتي دڠن ڬيڬيه، دي مناڠيس:—

“اڤ يڠ كامو لاكوكن اڤابيلا كامو مليهت ڤاوس، للاكي؟”

“ڽاڽي اونتوقڽ!” اداله ريسڤون ايمڤلوسيف دري سكوير سوارا يڠ دكلوب.

"باءيق!" منجريت اهب، دڠن ڤرستوجوان يڠ ليار دالم ناداڽ; ممڤرهاتيكن انيماسي يڠ ڤنوه ڤرهاتين د مان ڤرتاڽانڽ يڠ تيدق دجڠكاكن تله ملمڤركن مريك سچارا مڬنيت.

“دان اڤ يڠ كامو لاكوكن ستروسڽ، للاكي؟”

“تورونكنله، دان سلڤسڽ!”

“دان لاڬو اڤ يڠ كامو تاريق، للاكي؟”

“ڤاوس ماتي اتاو ڤراهو داڤور!”

سماكين ڤليق دان ڬانس ڬمبيرا دان مڽتوجوي، برتمبه واجه للاكي توا ايت ڤد ستياڤ سروان; سمنتارا ڤارا ڤلاير مولا مليهت دڠن ڤنوه راس ايڠين تاهو كڤد ساتو سام لا؞ين، سأوله-اوله مڠاڬومي باڬايمانا مريك سنديري منجادي بڬيتو تروجا دڠن ڤرتاڽان-ڤرتاڽان يڠ سأوله-اوله تيدق برتوجوان.

تتاڤي، مريك سموا برڬمبيرا لاڬي، سباڬايمان اهب، يڠ كيني سڤاروه برڤوتر د دالم لوبڠ ڤيوڤوتڽ، دڠن ساتو تاڠن منچاڤأي تيڠڬي ك اتس كافن، دان دڠن كراس، همڤير مڠڬڠڬوڽ، مڠهوبوڠي مريك بڬيتو:—

“سموا ڤارا ڤمندو تياڠ سبلوم اين تله مندڠر ساي ممبري ڤرينته تنتڠ ڤاوس ڤوتيه. ليهتله! ادكه اندا مليهت اونس امس سڤنڽول اين؟“—مڠڠكت ساتو دويت شيليڠ يڠ لواس دان چره كڤد ماتهاري—“اين اداله ساتو ڤوتوڠن انم بلس دولر، للاكي. ادكه اندا مليهتڽ؟ توان ستاربق، بريكن ساي يون توب-مال.”

كتيك ڤاسڠن ايت منداڤتكن تولوق، اهب، تنڤا برچاكڤ، ڤرلاهن-لاهن مڠڬوسوق ڤوتوڠن امس ايت ك اتس روكوق جاكيتڽ، سأوله-اوله اونتوق منيڠكتكن كچنتيكنڽ، دان تنڤا مڠڬوناكن سبارڠ ڤركاتان سمنتارا ايت برڬمبيرا دڠن رنده كڤد ديريڽ سنديري، مڠحاصيلكن سوارا يڠ بڬيتو ڤليق دان تيدق داڤت دڤركاتاكن سهيڠڬ اي دسڠكا ميكانيكال دري ڬيڬيه روداڽ دي.

منريما توڤ-مول دري ستاربق، دي ماجو ك اره تياڠ اوتام دڠن تولوق دأڠكت د ساتو تاڠن، ممڤاميركن امس دڠن تاڠن يڠ لا؞ين، دان دڠن سوارا يڠ تيڠڬي بركات: “سياڤا سهاج دري كالين يڠ مڠڠكت ساي سأورڠ ڤاوس بركڤالا ڤوتيه دڠن كڤالا يڠ بركدودوقن دان رحيم يڠ بڠكوق يڠ مڠڠكت ساي ڤاوس بركڤالا ڤوتيه، سياڤا سهاج يڠ مڠڠكت كڤالا ساي يڠ بركڤالا ڤوتيه; دڠن تيڬ لوبڠ يڠ دڤوتوڠ د ڤاڤن كاننڽ—ليهتله، سياڤا سهاج دري كاليان يڠ مڠڠكت ساي ڤاوس ڤوتيه يڠ سام، دي اكن ممڤوڽاءي اونس امس اين، انق-انق ساي!”

“هوزا! هوزا!” منجريت ڤارا ڤلاوت، سڤرتي دڠن ترڤولن يڠ برڬرق مريك مڠوچڤكن سلامت داتڠ كڤد ڤربواتن مموكول امس ك تياڠ.

"اين اداله ڤاوس ڤوتيه، كاتكو،" سامبوڠ لاڬي اهب، كتيك دي ملمڤركن توڤمول: "ڤاوس ڤوتيه. كوليت مات كامو اونتوقڽ، للاكي; ليهت تاجم اونتوق اير ڤوتيه; جك كامو مليهت هاڽ ڬيليمبوڠ، ڽاڽيكنله."

سموا اين سمنتارا تاشتڬو، داڬڬو، دان كويق تله مليهت دڠن مينت دان كجوتن يڠ لبيه اينتنسيف درڤد يڠ لا؞ين، دان ڤد ڤڽبوتن كاكي يڠ بركدودوقن دان رحيم يڠ بڠكيت مريك تله برمولا سأوله-اوله ماسيڠ-ماسيڠ دڤڠاروهي سچارا ترڤيسه اوليه ببراڤ ڤڠيڠاتن ترتنتو.

“كڤتن اهب،“ كات تاشتڬو، “ڤاوس ڤوتيه ايت ڤستي سام دڠن يڠ دڤڠڬيل اوليه سستڠه اورڠ سباڬاي موبي ديك.”

“موبي ديك؟” كات اهب. “ادكه اندا تاهو ڤاوس ڤوتيه ايت، تاش؟”

“ادكه دي برڤواس هاتي سديكيت، توان، سبلوم دي تورون؟” كات ڬاي-هيدر دڠن سڠاج.

“دان ادكه دي جوڬ ممڤوڽاءي ڤنچوري يڠ ايڠين تاهو،“ كات داڬڬو، “ساڠت روميت، والاوڤون اونتوق ڤرماچيتي، دان چڤت يڠ هيبت، كڤتن اهب؟”

“دان دي ممڤوڽاءي ساتو، دوا، تيڬ—اوه! باڽق بسي يڠ باءيق دالم ديريڽ برسمبوڽي، جوڬ، كڤتن،“ كات كويكوڬ دڠن تيدق برسسواين، “سموا تويسك-تي بي-تويسق، سڤرتي دي—دي—“ برڬڠڬو دڠن كراس اونتوق ساتو ڤركاتان، دان مڠهيلڠكن تاڠنڽ بولت دان بولت سأوله-اوله ممبوك بوتول—“سڤرتي دي—دي—”

“ڤمبورو!” منجريت اهب، “ايه، كويكوڬ، هرڤون-هارڤون ايت ترلتق سمواڽ برڤوتر دان ترڤوتوس دالم ديريڽ; ايه، داڬڬو، مونچوڠڽ اداله يڠ بسر، سڤرتي ڬندوم يڠ سلوروهڽ، دان ڤوتيه سڤرتي تيمبونن بولو نانتوكيت كامي سلڤس ڤموتوڠن كمبيڠ تاهونن يڠ هيبت; ايه، تاشتڬو، دان دي برڤچه-بله سڤرتي ڤچهن دان كماتين. شيطان! للاكي، ايت اداله موبي ديك يڠ تله اندا ليهت—موبي ديك—موبي ديك!”

“كڤتن اهب،“ كات ستاربق، يڠ، برسام ستوب دان فلاسك، سلاما اين ممڤرهاتيكن اتسنڽ دڠن كجوتن يڠ سماكين منيڠكت، تتاڤي اخيرڽ نمڤق ترڤڠاروه دڠن ڤميكيرن يڠ اڬق منجلسكن سموا كعجاءيبن ايت. “كڤتن اهب، ساي ڤرنه مندڠر تنتڠ موبي ديك—تتاڤي بوكن موبي ديك يڠ ملڤسكن كاكي اندا؟”

“سياڤا يڠ ممبريتاهو اندا ايت؟” مناڠيس اهب; كمودين برهنتي سكتيكا، “ايه، ستاربق; ايه، هاتي ساي سموا; اي اداله موبي ديك يڠ مڠچيواكن ساي; موبي ديك يڠ ممباوا ساي ك توڠڬق ماتي اين يڠ ساي برديري سكارڠ. ايه، ايه،“ دي برتاڽ دڠن اير مات بيناتڠ يڠ هيبت، كوات، سڤرتي موس يڠ ترججس هاتي; "ايه، ايه! ايت اداله ڤاوس ڤوتيه يڠ ترلقسان ايت يڠ مڠهنچوركن ساي; منجاديكن ساي ڤڬڠ يڠ ميسكين سلاماڽ دان ساتو هاري!" كمودين ملمڤركن كدوا-دوا تاڠنڽ، دڠن ڤنچابولن يڠ تيدق ترهيڠڬ، دي برتاڽ: “ايه، ايه! دان اكو اكن مڠجرڽ د سكيتر ڬود هوڤ، دان د سكيتر تندوق، دان د سكيتر نورواي ميلستروم، دان د سكيتر اڤي-اڤي ڤموسنهن سبلوم اكو مڽرهكنڽ. دان اينله يڠ تله كامو هانتر، للاكي! اونتوق مڠجر ڤاوس ڤوتيه ايت د كدوا-دوا بله تانه، د سلوروه سيسي بومي، دي اكن مڠجرڽ داره دان ڬوليڠن كلوار. اڤ يڠ كامو كاتاكن، للاكي، اكن كامو مڽامبوڠ تاڠن ڤدڽ، سكارڠ؟

“ايه، ايه!” تروق ڤارا ڤنوڠڬڠ دان ڤارا ڤلاوت، برلاري لبيه دكت كڤد للاكي توا يڠ ترڬڠڬو ايت: “مات يڠ تاجم اونتوق ڤاوس ڤوتيه; تومبق يڠ تاجم اونتوق موبي ديك!”

“توهن ممبركاتي اندا،“ دي سأوله-اوله سڤاروه منڠيس دان سڤاروه برتاڽ. "توهن ممبركاتي اندا، للاكي-للاكي. ستوارد! ڤرڬيله ملوكيس اوكورن ڬروڬ يڠ بسر. تتاڤي اڤ يڠ موك ڤنجڠ اين، توان ستاربق; تيدقكه اندا اكن مڠجر ڤاوس ڤوتيه؟ بوكن ڤرماءينن اونتوق موبي ديك؟"

"ساي اداله ڤرماءينن اونتوق رحيمڽ يڠ بڠكوق، دان اونتوق رحيم ماتي جوڬ، كڤتن اهب، جك اي عاديل داتڠ دالم جالن ڤرنياڬان يڠ كامي ايكوتي; تتاڤي ساي داتڠ ك سيني اونتوق ممبورو ڤاوس، بوكن بالسن ڤڠليما ساي. براڤ باڽق توڠ يڠ اكن دبريكن اوليه ڤمبالسن اندا والاوڤون اندا منداڤتكنڽ، كڤتن اهب اي تيدق اكن منداڤتكن باڽق د ڤاسرن نانتو كامي؟"

"ڤاسر نانتوكيت! هوت! تتاڤي داتڠله لبيه دكت، ستاربق; اندا ممرلوكن لاڤيسن يڠ لبيه رنده سديكيت. جك واڠ منجادي ڤڠوكورن، مان، دان ڤارا اكاونتن تله مڠيرا رومه ڤڠهيتوڠن مريك يڠ هيبت د ڬلوب، دڠن مڠيكتڽ دڠن ڬينيا، ساتو كڤد ستياڤ تيڬ بهاڬين اينچي; مك، ايذينكن ساي ممبريتاهو اندا، بهاوا بالسن ساي اكن منيمبولكن ڤريميوم يڠ بسر د سيني!
