Ngan| |sigurado| |nga| |may|-|ada| |pinakatalagsaon| |nga| |aringasa| |nga| |nahitatabo| |ha| |sulod| |—pirme| |nga| |pag|-|uwang| |ngan| |pag|-|atras|,| |ngan| |ha| |panapanahon| |usa| |nga| |daku| |nga| |pag|-|uyog|,| |nga| |baga| |hin| |ginbari| |an| |usa| |nga| |plato| |o| |kaldero|.
