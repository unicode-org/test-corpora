Porko| |kaj| |Pipro

"|Ne| |utilas| |frapi|,|"| |diris| |la| |Lakeo|,| |"|kaj| |tio| |pro| |du| |kialoj|.| |Unue|,| |ĉar| |mi| |estas| |sur| |la| |sama| |flanko| |de| |la| |pordo| |kiel| |vi|;| |due|,| |ĉar| |ili| |faras| |tian| |bruon| |interne|,| |neniu| |povus| |aŭdi| |vin|.|"

"|Bonvolu| |do|,|"| |diris| |Alicio|,| |"|kiel| |mi| |eniru|?|"

“|Sed| |eble| |li| |ne| |povas| |eviti|,|”| |ŝi| |diris| |al| |si|;

“|—aŭ| |la| |venontan| |tagon|,| |eble|,|”| |la| |Lakeo| |daŭrigis| |per| |la| |sama| |tono|,| |ĝuste| |kvazaŭ| |nenio| |estus| |okazinta|.
"|Kiel| |mi| |eniru|?|"

"|Ĉu| |vi| |entute| |eniru|?|"

"|Tio| |estas| |la| |unua| |demando|,| |vi| |scias|.|"

"|Estas| |vere| |terure|,|"| |ŝi| |murmuris| |al| |si|,| |"|kiel| |ĉiuj| |estaĵoj| |kverelas|.| |Sufiĉas| |por| |frenezigi| |unu|!|"
La| |Lakeo| |ŝajnis| |opinii| |tion| |bona| |okazo| |por| |ripeti| |sian| |rimarkon|,| |kun| |varioj|.
"|Mi| |sidos| |ĉi| |tie|,|"| |li| |diris|,| |"|ĉe| |kaj| |ekstere|,| |dum| |tagoj| |kaj| |tagoj|.|"
"|Sed| |kion| |mi| |faru|?|"

"|Ho|,| |ne| |utilas| |paroli| |kun| |li|,|"| |diris| |Alicio| |senespere|:| |"|li| |estas| |tute| |idiota|!|"

"|Certe| |estas| |tro| |da| |pipro| |en| |tiu| |supo|!|"

"|Bonvolu| |diri| |al| |mi|,|"| |diris| |Alicio|,| |iom| |timeme|,| |ĉar| |ŝi| |ne| |estis| |tute| |certa|,| |ĉu| |estas| |bona| |maniero| |por| |ŝi| |paroli| |unue|,| |"|kial| |via| |kato| |tiel| |ridas|?|"
"|Ĝi| |estas| |Ĉeŝira| |kato|,|"| |diris| |la| |Dukino|,| |"|kaj| |tial|.| |Porko|!|"

"|Mi| |ne| |sciis|,| |ke| |Cheshire|-|katoj| |ĉiam| |rikanis|;| |fakte|,| |mi| |ne| |sciis|,| |ke| |katoj| |povas| |rikani|.|"
“|Ili| |ĉiuj| |povas|,|”| |diris| |la| |Dukino|;
"|Kaj| |la| |plej| |multaj| |el| |ili| |faras|.|"
“|Mi| |ne| |scias| |pri| |iu| |ajn|,|”| |Alico| |diris| |tre| |ĝentile|,| |sentante| |tre| |kontenta| |pro| |konversacio|.
“|Vi| |ne| |scias| |multon|,|”| |diris| |la| |Dukino|;
"|kaj| |tio| |estas| |fakto|.|"

Dum| |ŝi| |provis| |fiksi| |unu|,| |la| |kuiristino| |deprenis| |la| |kaldronon| |da| |supo| |de| |la| |fajro|,| |kaj| |tuj| |eklaboris| |ĵeti| |ĉion| |en| |sia| |atingo| |al| |la| |Dukino| |kaj| |la| |bebo|—|la| |fajroferoj| |estis| |unue|;

"|Ho|,| |bonvolu| |atenti|,| |kion| |vi| |faras|!|"

"|Ho|,| |tie| |iras| |lia| |altvalora| |nazo|!|"

"|Se| |ĉiuj| |zorgus| |pri| |siaj| |aferoj|,|"| |la| |Dukino| |diris| |raŭka| |grumblo|,| |"|la| |mondo| |ĉirkaŭirus| |interkonsenton| |pli| |rapide| |ol| |ĝi|.|"

"|Nur| |pensu| |pri| |kia| |laboro| |ĝi| |farus| |tage| |kaj| |nokte|!| |Vi| |vidas|,| |ke| |la| |tero| |bezonas| |dudek| |kvar| |horojn| |por| |turniĝi| |laŭ| |sia| |akso—|"
"|Parolante| |pri| |hakiloj|,|"| |diris| |la| |Dukino|,| |"|dehaku| |ŝian| |kapon|!|"

"|Mi| |neniam| |povis| |elteni| |figurojn|!|"
